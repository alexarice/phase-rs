6pub unsafe trait Ptr: Sized + Copy {
7	type Item;
8
9	fn get_ptr(ptr: *mut Self::Item) -> Self;
10
11	fn null() -> Self;
12
13	unsafe fn offset_from(self, origin: Self) -> isize;
14
15	unsafe fn add(self, offset: usize) -> Self;
16	unsafe fn sub(self, offset: usize) -> Self;
17
18	unsafe fn read(self) -> Self::Item;
19	unsafe fn write(self, item: Self::Item);
20
21	unsafe fn copy_nonoverlapping(src: Self, dst: Self, len: usize);
22	unsafe fn reverse(ptr: Self, len: usize);
23
24	unsafe fn swap(a: Self, b: Self) {
25		let a_item = a.read();
26		let b_item = b.read();
27
28		a.write(b_item);
29		b.write(a_item);
30	}
31
32	unsafe fn swap_idx(self, i: usize, j: usize) {
33		Self::swap(self.add(i), self.add(j));
34	}
35}

148impl<P: Ptr> Drop for InsertionHole<P> {
149	#[inline(always)]
150	fn drop(&mut self) {
151		// SAFETY: This is a helper class. Please refer to its usage for correctness. Namely, one
152		// must be sure that `src` and `dst` does not overlap as required by
153		// `ptr::copy_nonoverlapping` and are both valid for writes.
154		unsafe {
155			P::copy_nonoverlapping(self.src, self.dest, 1);
156		}
157	}

160unsafe fn insert_tail<P: Ptr, F>(v: P, v_len: usize, is_less: &mut F)
161where
162	F: FnMut(P, P) -> bool,
163{
164	debug_assert!(v_len >= 2);
165
166	let arr_ptr = v;
167	let i = v_len - 1;
168
169	// SAFETY: caller must ensure v is at least len 2.
170	unsafe {
171		// See insert_head which talks about why this approach is beneficial.
172		let i_ptr = arr_ptr.add(i);
173
174		// It's important that we use i_ptr here. If this check is positive and we continue,
175		// We want to make sure that no other copy of the value was seen by is_less.
176		// Otherwise we would have to copy it back.
177		if is_less(i_ptr, i_ptr.sub(1)) {
178			// It's important, that we use tmp for comparison from now on. As it is the value that
179			// will be copied back. And notionally we could have created a divergence if we copy
180			// back the wrong value.
181			let tmp = core::mem::ManuallyDrop::new(P::read(i_ptr));
182			let tmp = P::get_ptr((&*tmp) as *const P::Item as *mut P::Item);
183			// Intermediate state of the insertion process is always tracked by `hole`, which
184			// serves two purposes:
185			// 1. Protects integrity of `v` from panics in `is_less`.
186			// 2. Fills the remaining hole in `v` in the end.
187			//
188			// Panic safety:
189			//
190			// If `is_less` panics at any point during the process, `hole` will get dropped and
191			// fill the hole in `v` with `tmp`, thus ensuring that `v` still holds every object it
192			// initially held exactly once.
193			let mut hole = InsertionHole {
194				src: tmp,
195				dest: i_ptr.sub(1),
196			};
197			P::copy_nonoverlapping(hole.dest, i_ptr, 1);
198
199			// SAFETY: We know i is at least 1.
200			for j in (0..(i - 1)).rev() {
201				let j_ptr = arr_ptr.add(j);
202				if !is_less(tmp, j_ptr) {
203					break;
204				}
205
206				P::copy_nonoverlapping(j_ptr, hole.dest, 1);
207				hole.dest = j_ptr;
208			}
209			// `hole` gets dropped and thus copies `tmp` into the remaining hole in `v`.
210		}
211	}
212}

214unsafe fn insert_head<P: Ptr, F>(v: P, v_len: usize, is_less: &mut F)
215where
216	F: FnMut(P, P) -> bool,
217{
218	debug_assert!(v_len >= 2);
219
220	// SAFETY: caller must ensure v is at least len 2.
221	unsafe {
222		if is_less(v.add(1), v.add(0)) {
223			let arr_ptr = v;
224
225			// There are three ways to implement insertion here:
226			//
227			// 1. Swap adjacent elements until the first one gets to its final destination. However, this way we
228			//    copy data around more than is necessary. If elements are big structures (costly to copy), this
229			//    method will be slow.
230			//
231			// 2. Iterate until the right place for the first element is found. Then shift the elements
232			//    succeeding it to make room for it and finally place it into the remaining hole. This is a good
233			//    method.
234			//
235			// 3. Copy the first element into a temporary variable. Iterate until the right place for it is
236			//    found. As we go along, copy every traversed element into the slot preceding it. Finally, copy
237			//    data from the temporary variable into the remaining hole. This method is very good. Benchmarks
238			//    demonstrated slightly better performance than with the 2nd method.
239			//
240			// All methods were benchmarked, and the 3rd showed best results. So we chose that one.
241			let tmp = core::mem::ManuallyDrop::new(P::read(arr_ptr));
242			let tmp = P::get_ptr((&*tmp) as *const P::Item as *mut P::Item);
243
244			// Intermediate state of the insertion process is always tracked by `hole`, which
245			// serves two purposes:
246			// 1. Protects integrity of `v` from panics in `is_less`.
247			// 2. Fills the remaining hole in `v` in the end.
248			//
249			// Panic safety:
250			//
251			// If `is_less` panics at any point during the process, `hole` will get dropped and
252			// fill the hole in `v` with `tmp`, thus ensuring that `v` still holds every object it
253			// initially held exactly once.
254			let mut hole = InsertionHole {
255				src: tmp,
256				dest: arr_ptr.add(1),
257			};
258			P::copy_nonoverlapping(arr_ptr.add(1), arr_ptr.add(0), 1);
259
260			for i in 2..v_len {
261				if !is_less(v.add(i), tmp) {
262					break;
263				}
264				P::copy_nonoverlapping(arr_ptr.add(i), arr_ptr.add(i - 1), 1);
265				hole.dest = arr_ptr.add(i);
266			}
267			// `hole` gets dropped and thus copies `tmp` into the remaining hole in `v`.
268		}
269	}
270}

273pub(super) fn insertion_sort_shift_left<P: Ptr, F: FnMut(P, P) -> bool>(v: P, v_len: usize, offset: usize, is_less: &mut F) {
274	let len = v_len;
275
276	// Using assert here improves performance.
277	core::assert!(offset != 0 && offset <= len);
278
279	// Shift each element of the unsorted region v[i..] as far left as is needed to make v sorted.
280	for i in offset..len {
281		// SAFETY: we tested that `offset` must be at least 1, so this loop is only entered if len
282		// >= 2. The range is exclusive and we know `i` must be at least 1 so this slice has at
283		// >least len 2.
284		unsafe {
285			insert_tail(v, i + 1, is_less);
286		}
287	}
288}

290#[inline(never)]
291fn insertion_sort_shift_right<P: Ptr, F: FnMut(P, P) -> bool>(v: P, v_len: usize, offset: usize, is_less: &mut F) {
292	let len = v_len;
293
294	// Using assert here improves performance.
295	core::assert!(offset != 0 && offset <= len && len >= 2);
296
297	// Shift each element of the unsorted region v[..i] as far left as is needed to make v sorted.
298	for i in (0..offset).rev() {
299		// SAFETY: we tested that `offset` must be at least 1, so this loop is only entered if len
300		// >= 2.We ensured that the slice length is always at least 2 long. We know that start_found
301		// will be at least one less than end, and the range is exclusive. Which gives us i always
302		// <= (end - 2).
303		unsafe {
304			insert_head(v.add(i), len - i, is_less);
305		}
306	}
307}

309#[cold]
310unsafe fn partial_insertion_sort<P: Ptr, F: FnMut(P, P) -> bool>(v: P, v_len: usize, is_less: &mut F) -> bool {
311	// Maximum number of adjacent out-of-order pairs that will get shifted.
312	const MAX_STEPS: usize = 5;
313	// If the slice is shorter than this, don't shift any elements.
314	const SHORTEST_SHIFTING: usize = 50;
315
316	let len = v_len;
317	let mut i = 1;
318
319	for _ in 0..MAX_STEPS {
320		// SAFETY: We already explicitly did the bound checking with `i < len`.
321		// All our subsequent indexing is only in the range `0 <= index < len`
322		unsafe {
323			// Find the next pair of adjacent out-of-order elements.
324			while i < len && !is_less(v.add(i), v.add(i - 1)) {
325				i += 1;
326			}
327		}
328
329		// Are we done?
330		if i == len {
331			return true;
332		}
333
334		// Don't shift elements on short arrays, that has a performance cost.
335		if len < SHORTEST_SHIFTING {
336			return false;
337		}
338
339		// Swap the found pair of elements. This puts them in correct order.
340		v.swap_idx(i - 1, i);
341
342		if i >= 2 {
343			// Shift the smaller element to the left.
344			insertion_sort_shift_left(v, i, i - 1, is_less);
345
346			// Shift the greater element to the right.
347			insertion_sort_shift_right(v, i, 1, is_less);
348		}
349	}
350
351	// Didn't manage to sort the slice in the limited number of steps.
352	false
353}

356pub unsafe fn heapsort<P: Ptr, F: FnMut(P, P) -> bool>(v: P, v_len: usize, mut is_less: F) {
357	// This binary heap respects the invariant `parent >= child`.
358	let mut sift_down = |v: P, v_len: usize, mut node| {
359		loop {
360			// Children of `node`.
361			let mut child = 2 * node + 1;
362			if child >= v_len {
363				break;
364			}
365
366			// Choose the greater child.
367			if child + 1 < v_len {
368				// We need a branch to be sure not to out-of-bounds index,
369				// but it's highly predictable.  The comparison, however,
370				// is better done branchless, especially for primitives.
371				child += is_less(v.add(child), v.add(child + 1)) as usize;
372			}
373
374			// Stop if the invariant holds at `node`.
375			if !is_less(v.add(node), v.add(child)) {
376				break;
377			}
378
379			// Swap `node` with the greater child, move one step down, and continue sifting.
380			v.swap_idx(node, child);
381			node = child;
382		}
383	};
384
385	// Build the heap in linear time.
386	for i in (0..v_len / 2).rev() {
387		sift_down(v, v_len, i);
388	}
389
390	// Pop maximal elements from the heap.
391	for i in (1..v_len).rev() {
392		v.swap_idx(0, i);
393		sift_down(v, i, 0);
394	}
395}

397unsafe fn partition_in_blocks<P: Ptr, F: FnMut(P, P) -> bool>(v: P, v_len: usize, pivot: P, is_less: &mut F) -> usize {
398	// Number of elements in a typical block.
399	const BLOCK: usize = 128;
400
401	// The partitioning algorithm repeats the following steps until completion:
402	//
403	// 1. Trace a block from the left side to identify elements greater than or equal to the pivot.
404	// 2. Trace a block from the right side to identify elements smaller than the pivot.
405	// 3. Exchange the identified elements between the left and right side.
406	//
407	// We keep the following variables for a block of elements:
408	//
409	// 1. `block` - Number of elements in the block.
410	// 2. `start` - Start pointer into the `offsets` array.
411	// 3. `end` - End pointer into the `offsets` array.
412	// 4. `offsets` - Indices of out-of-order elements within the block.
413
414	// The current block on the left side (from `l` to `l.add(block_l)`).
415	let mut l = v;
416	let mut block_l = BLOCK;
417	let mut start_l = core::ptr::null_mut();
418	let mut end_l = core::ptr::null_mut();
419	let mut offsets_l = [core::mem::MaybeUninit::<u8>::uninit(); BLOCK];
420
421	// The current block on the right side (from `r.sub(block_r)` to `r`).
422	// SAFETY: The documentation for .add() specifically mention that `vec.as_ptr().add(vec.len())`
423	// is always safe
424	let mut r = unsafe { l.add(v_len) };
425	let mut block_r = BLOCK;
426	let mut start_r = core::ptr::null_mut();
427	let mut end_r = core::ptr::null_mut();
428	let mut offsets_r = [core::mem::MaybeUninit::<u8>::uninit(); BLOCK];
429
430	// FIXME: When we get VLAs, try creating one array of length `min(v.len(), 2 * BLOCK)` rather
431	// than two fixed-size arrays of length `BLOCK`. VLAs might be more cache-efficient.
432
433	// Returns the number of elements between pointers `l` (inclusive) and `r` (exclusive).
434	unsafe fn width<P: Ptr>(l: P, r: P) -> usize {
435		r.offset_from(l) as usize
436	}
437
438	loop {
439		// We are done with partitioning block-by-block when `l` and `r` get very close. Then we do
440		// some patch-up work in order to partition the remaining elements in between.
441		let is_done = width(l, r) <= 2 * BLOCK;
442
443		if is_done {
444			// Number of remaining elements (still not compared to the pivot).
445			let mut rem = width(l, r);
446			if start_l < end_l || start_r < end_r {
447				rem -= BLOCK;
448			}
449
450			// Adjust block sizes so that the left and right block don't overlap, but get perfectly
451			// aligned to cover the whole remaining gap.
452			if start_l < end_l {
453				block_r = rem;
454			} else if start_r < end_r {
455				block_l = rem;
456			} else {
457				// There were the same number of elements to switch on both blocks during the last
458				// iteration, so there are no remaining elements on either block. Cover the
459				// remaining items with roughly equally-sized blocks.
460				block_l = rem / 2;
461				block_r = rem - block_l;
462			}
463			debug_assert!(block_l <= BLOCK && block_r <= BLOCK);
464			debug_assert!(width(l, r) == block_l + block_r);
465		}
466
467		if start_l == end_l {
468			// Trace `block_l` elements from the left side.
469			start_l = offsets_l.as_mut_ptr() as *mut u8;
470			end_l = start_l;
471			let mut elem = l;
472
473			for i in 0..block_l {
474				// SAFETY: The unsafety operations below involve the usage of the `offset`.
475				//         According to the conditions required by the function, we satisfy them
476				// because:
477				//         1. `offsets_l` is stack-allocated, and thus considered separate allocated object.
478				//         2. The function `is_less` returns a `bool`. Casting a `bool` will never overflow `isize`.
479				//         3. We have guaranteed that `block_l` will be `<= BLOCK`. Plus, `end_l` was initially set
480				//            to the begin pointer of `offsets_` which was declared on the stack. Thus, we know that
481				//            even in the worst case (all invocations of `is_less` returns false) we will only be at
482				//            most 1 byte pass the end.
483				//        Another unsafety operation here is dereferencing `elem`.
484				//        However, `elem` was initially the begin pointer to the slice which is
485				// always valid.
486				unsafe {
487					// Branchless comparison.
488					*end_l = i as u8;
489					end_l = end_l.add(!is_less(elem, pivot) as usize);
490					elem = elem.add(1);
491				}
492			}
493		}
494
495		if start_r == end_r {
496			// Trace `block_r` elements from the right side.
497			start_r = offsets_r.as_mut_ptr() as *mut u8;
498			end_r = start_r;
499			let mut elem = r;
500
501			for i in 0..block_r {
502				// SAFETY: The unsafety operations below involve the usage of the `offset`.
503				//         According to the conditions required by the function, we satisfy them
504				// because:
505				//         1. `offsets_r` is stack-allocated, and thus considered separate allocated object.
506				//         2. The function `is_less` returns a `bool`. Casting a `bool` will never overflow `isize`.
507				//         3. We have guaranteed that `block_r` will be `<= BLOCK`. Plus, `end_r` was initially set
508				//            to the begin pointer of `offsets_` which was declared on the stack. Thus, we know that
509				//            even in the worst case (all invocations of `is_less` returns true) we will only be at
510				//            most 1 byte pass the end.
511				//        Another unsafety operation here is dereferencing `elem`.
512				//        However, `elem` was initially `1 * sizeof(T)` past the end and we
513				// decrement it by `1 * sizeof(T)` before accessing it.        Plus,
514				// `block_r` was asserted to be less than `BLOCK` and `elem` will therefore at most
515				// be pointing to the beginning of the slice.
516				unsafe {
517					// Branchless comparison.
518					elem = elem.sub(1);
519					*end_r = i as u8;
520					end_r = end_r.add(is_less(elem, pivot) as usize);
521				}
522			}
523		}
524
525		// Number of out-of-order elements to swap between the left and right side.
526		let count = Ord::min(width(start_l, end_l), width(start_r, end_r));
527
528		if count > 0 {
529			macro_rules! left {
530				() => {
531					l.add(usize::from(*start_l))
532				};
533			}
534			macro_rules! right {
535				() => {
536					r.sub(usize::from(*start_r) + 1)
537				};
538			}
539
540			// Instead of swapping one pair at the time, it is more efficient to perform a cyclic
541			// permutation. This is not strictly equivalent to swapping, but produces a similar
542			// result using fewer memory operations.
543
544			// SAFETY: The use of `ptr::read` is valid because there is at least one element in
545			// both `offsets_l` and `offsets_r`, so `left!` is a valid pointer to read from.
546			//
547			// The uses of `left!` involve calls to `offset` on `l`, which points to the
548			// beginning of `v`. All the offsets pointed-to by `start_l` are at most `block_l`, so
549			// these `offset` calls are safe as all reads are within the block. The same argument
550			// applies for the uses of `right!`.
551			//
552			// The calls to `start_l.offset` are valid because there are at most `count-1` of them,
553			// plus the final one at the end of the unsafe block, where `count` is the minimum
554			// number of collected offsets in `offsets_l` and `offsets_r`, so there is
555			// no risk of there not being enough elements. The same reasoning applies to
556			// the calls to `start_r.offset`.
557			//
558			// The calls to `copy_nonoverlapping` are safe because `left!` and `right!` are
559			// guaranteed not to overlap, and are valid because of the reasoning above.
560			unsafe {
561				let tmp = P::read(left!());
562				let tmp_ptr = P::get_ptr(&tmp as *const P::Item as *mut P::Item);
563				P::copy_nonoverlapping(right!(), left!(), 1);
564
565				for _ in 1..count {
566					start_l = start_l.add(1);
567					P::copy_nonoverlapping(left!(), right!(), 1);
568					start_r = start_r.add(1);
569					P::copy_nonoverlapping(right!(), left!(), 1);
570				}
571
572				P::copy_nonoverlapping(tmp_ptr, right!(), 1);
573				// core::mem::forget(tmp);
574				start_l = start_l.add(1);
575				start_r = start_r.add(1);
576			}
577		}
578
579		if start_l == end_l {
580			// All out-of-order elements in the left block were moved. Move to the next block.
581
582			// block-width-guarantee
583			// SAFETY: if `!is_done` then the slice width is guaranteed to be at least `2*BLOCK`
584			// wide. There are at most `BLOCK` elements in `offsets_l` because of its
585			// size, so the `offset` operation is safe. Otherwise, the debug assertions
586			// in the `is_done` case guarantee that `width(l, r) == block_l + block_r`,
587			// namely, that the block sizes have been adjusted to account
588			// for the smaller number of remaining elements.
589			l = unsafe { l.add(block_l) };
590		}
591
592		if start_r == end_r {
593			// All out-of-order elements in the right block were moved. Move to the previous block.
594
595			// SAFETY: Same argument as [block-width-guarantee]. Either this is a full block
596			// `2*BLOCK`-wide, or `block_r` has been adjusted for the last handful of
597			// elements.
598			r = unsafe { r.sub(block_r) };
599		}
600
601		if is_done {
602			break;
603		}
604	}
605
606	// All that remains now is at most one block (either the left or the right) with out-of-order
607	// elements that need to be moved. Such remaining elements can be simply shifted to the end
608	// within their block.
609
610	if start_l < end_l {
611		// The left block remains.
612		// Move its remaining out-of-order elements to the far right.
613		debug_assert_eq!(width(l, r), block_l);
614		while start_l < end_l {
615			// remaining-elements-safety
616			// SAFETY: while the loop condition holds there are still elements in `offsets_l`, so it
617			// is safe to point `end_l` to the previous element.
618			//
619			// The `ptr::swap` is safe if both its arguments are valid for reads and writes:
620			//  - Per the debug assert above, the distance between `l` and `r` is `block_l` elements, so there
621			//    can be at most `block_l` remaining offsets between `start_l` and `end_l`. This means `r` will
622			//    be moved at most `block_l` steps back, which makes the `r.offset` calls valid (at that point
623			//    `l == r`).
624			//  - `offsets_l` contains valid offsets into `v` collected during the partitioning of the last
625			//    block, so the `l.offset` calls are valid.
626			unsafe {
627				end_l = end_l.sub(1);
628				P::swap(l.add(usize::from(*end_l)), r.sub(1));
629				r = r.sub(1);
630			}
631		}
632		width(v, r)
633	} else if start_r < end_r {
634		// The right block remains.
635		// Move its remaining out-of-order elements to the far left.
636		debug_assert_eq!(width(l, r), block_r);
637		while start_r < end_r {
638			// SAFETY: See the reasoning in [remaining-elements-safety].
639			unsafe {
640				end_r = end_r.sub(1);
641				P::swap(l, r.sub(usize::from(*end_r) + 1));
642				l = l.add(1);
643			}
644		}
645		width(v, l)
646	} else {
647		// Nothing else to do, we're done.
648		width(v, l)
649	}
650}

652pub(super) unsafe fn partition<P: Ptr, F>(v: P, v_len: usize, pivot: usize, is_less: &mut F) -> (usize, bool)
653where
654	F: FnMut(P, P) -> bool,
655{
656	let (mid, was_partitioned) = {
657		// Place the pivot at the beginning of slice.
658		v.swap_idx(0, pivot);
659		let pivot = v;
660		let v = v.add(1);
661		let v_len = v_len - 1;
662
663		// Read the pivot into a stack-allocated variable for efficiency. If a following comparison
664		// operation panics, the pivot will be automatically written back into the slice.
665
666		// SAFETY: `pivot` is a reference to the first element of `v`, so `ptr::read` is safe.
667		let tmp = core::mem::ManuallyDrop::new(unsafe { P::read(pivot) });
668		let tmp = P::get_ptr((&*tmp) as *const P::Item as *mut P::Item);
669		let _pivot_guard = InsertionHole { src: tmp, dest: pivot };
670		let pivot = tmp;
671
672		// Find the first pair of out-of-order elements.
673		let mut l = 0;
674		let mut r = v_len;
675
676		// SAFETY: The unsafety below involves indexing an array.
677		// For the first one: We already do the bounds checking here with `l < r`.
678		// For the second one: We initially have `l == 0` and `r == v.len()` and we checked that `l
679		// < r` at every indexing operation.                     From here we know that `r`
680		// must be at least `r == l` which was shown to be valid from the first one.
681		unsafe {
682			// Find the first element greater than or equal to the pivot.
683			while l < r && is_less(v.add(l), pivot) {
684				l += 1;
685			}
686
687			// Find the last element smaller that the pivot.
688			while l < r && !is_less(v.add(r - 1), pivot) {
689				r -= 1;
690			}
691		}
692
693		(l + partition_in_blocks(v.add(l), r - l, pivot, is_less), l >= r)
694
695		// `_pivot_guard` goes out of scope and writes the pivot (which is a stack-allocated
696		// variable) back into the slice where it originally was. This step is critical in ensuring
697		// safety!
698	};
699
700	// Place the pivot between the two partitions.
701	v.swap_idx(0, mid);
702
703	(mid, was_partitioned)
704}

706pub(super) unsafe fn partition_equal<P: Ptr, F>(v: P, v_len: usize, pivot: usize, is_less: &mut F) -> usize
707where
708	F: FnMut(P, P) -> bool,
709{
710	// Place the pivot at the beginning of slice.
711	v.swap_idx(0, pivot);
712	let pivot = v;
713	let v = v.add(1);
714	let v_len = v_len - 1;
715
716	// Read the pivot into a stack-allocated variable for efficiency. If a following comparison
717	// operation panics, the pivot will be automatically written back into the slice.
718	// SAFETY: The pointer here is valid because it is obtained from a reference to a slice.
719	let tmp = core::mem::ManuallyDrop::new(unsafe { P::read(pivot) });
720	let tmp = P::get_ptr((&*tmp) as *const P::Item as *mut P::Item);
721	let _pivot_guard = InsertionHole { src: tmp, dest: pivot };
722	let pivot = tmp;
723
724	let len = v_len;
725	if len == 0 {
726		return 0;
727	}
728
729	// Now partition the slice.
730	let mut l = 0;
731	let mut r = len;
732	loop {
733		// SAFETY: The unsafety below involves indexing an array.
734		// For the first one: We already do the bounds checking here with `l < r`.
735		// For the second one: We initially have `l == 0` and `r == v.len()` and we checked that `l
736		// < r` at every indexing operation.                     From here we know that `r`
737		// must be at least `r == l` which was shown to be valid from the first one.
738		unsafe {
739			// Find the first element greater than the pivot.
740			while l < r && !is_less(pivot, v.add(l)) {
741				l += 1;
742			}
743
744			// Find the last element equal to the pivot.
745			loop {
746				r -= 1;
747				if l >= r || !is_less(pivot, v.add(r)) {
748					break;
749				}
750			}
751
752			// Are we done?
753			if l >= r {
754				break;
755			}
756
757			// Swap the found pair of out-of-order elements.
758			let ptr = v;
759			P::swap(ptr.add(l), ptr.add(r));
760			l += 1;
761		}
762	}
763
764	// We found `l` elements equal to the pivot. Add 1 to account for the pivot itself.
765	l + 1
766
767	// `_pivot_guard` goes out of scope and writes the pivot (which is a stack-allocated variable)
768	// back into the slice where it originally was. This step is critical in ensuring safety!
769}

771#[cold]
772pub(super) unsafe fn break_patterns<P: Ptr>(v: P, v_len: usize) {
773	let len = v_len;
774	if len >= 8 {
775		let mut seed = len;
776		let mut gen_usize = || {
777			// Pseudorandom number generator from the "Xorshift RNGs" paper by George Marsaglia.
778			if usize::BITS <= 32 {
779				let mut r = seed as u32;
780				r ^= r << 13;
781				r ^= r >> 17;
782				r ^= r << 5;
783				seed = r as usize;
784				seed
785			} else {
786				let mut r = seed as u64;
787				r ^= r << 13;
788				r ^= r >> 7;
789				r ^= r << 17;
790				seed = r as usize;
791				seed
792			}
793		};
794
795		// Take random numbers modulo this number.
796		// The number fits into `usize` because `len` is not greater than `isize::MAX`.
797		let modulus = len.next_power_of_two();
798
799		// Some pivot candidates will be in the nearby of this index. Let's randomize them.
800		let pos = len / 4 * 2;
801
802		for i in 0..3 {
803			// Generate a random number modulo `len`. However, in order to avoid costly operations
804			// we first take it modulo a power of two, and then decrease by `len` until it fits
805			// into the range `[0, len - 1]`.
806			let mut other = gen_usize() & (modulus - 1);
807
808			// `other` is guaranteed to be less than `2 * len`.
809			if other >= len {
810				other -= len;
811			}
812
813			v.swap_idx(pos - 1 + i, other);
814		}
815	}
816}

818pub(super) unsafe fn choose_pivot<P: Ptr, F>(v: P, v_len: usize, is_less: &mut F) -> (usize, bool)
819where
820	F: FnMut(P, P) -> bool,
821{
822	// Minimum length to choose the median-of-medians method.
823	// Shorter slices use the simple median-of-three method.
824	const SHORTEST_MEDIAN_OF_MEDIANS: usize = 50;
825	// Maximum number of swaps that can be performed in this function.
826	const MAX_SWAPS: usize = 4 * 3;
827
828	let len = v_len;
829
830	// Three indices near which we are going to choose a pivot.
831	let mut a = len / 4;
832	let mut b = len / 4 * 2;
833	let mut c = len / 4 * 3;
834
835	// Counts the total number of swaps we are about to perform while sorting indices.
836	let mut swaps = 0;
837
838	if len >= 8 {
839		// Swaps indices so that `v[a] <= v[b]`.
840		// SAFETY: `len >= 8` so there are at least two elements in the neighborhoods of
841		// `a`, `b` and `c`. This means the three calls to `sort_adjacent` result in
842		// corresponding calls to `sort3` with valid 3-item neighborhoods around each
843		// pointer, which in turn means the calls to `sort2` are done with valid
844		// references. Thus the `v.get_unchecked` calls are safe, as is the `ptr::swap`
845		// call.
846		let mut sort2 = |a: &mut usize, b: &mut usize| unsafe {
847			if is_less(v.add(*b), v.add(*a)) {
848				core::ptr::swap(a, b);
849				swaps += 1;
850			}
851		};
852
853		// Swaps indices so that `v[a] <= v[b] <= v[c]`.
854		let mut sort3 = |a: &mut usize, b: &mut usize, c: &mut usize| {
855			sort2(a, b);
856			sort2(b, c);
857			sort2(a, b);
858		};
859
860		if len >= SHORTEST_MEDIAN_OF_MEDIANS {
861			// Finds the median of `v[a - 1], v[a], v[a + 1]` and stores the index into `a`.
862			let mut sort_adjacent = |a: &mut usize| {
863				let tmp = *a;
864				sort3(&mut (tmp - 1), a, &mut (tmp + 1));
865			};
866
867			// Find medians in the neighborhoods of `a`, `b`, and `c`.
868			sort_adjacent(&mut a);
869			sort_adjacent(&mut b);
870			sort_adjacent(&mut c);
871		}
872
873		// Find the median among `a`, `b`, and `c`.
874		sort3(&mut a, &mut b, &mut c);
875	}
876
877	if swaps < MAX_SWAPS {
878		(b, swaps == 0)
879	} else {
880		// The maximum number of swaps was performed. Chances are the slice is descending or mostly
881		// descending, so reversing will probably help sort it faster.
882		P::reverse(v, v_len);
883		(len - 1 - b, true)
884	}
885}

887/// sorts `v` using pattern-defeating quicksort, which is *O*(*n* \* log(*n*)) worst-case
888pub unsafe fn quicksort<P: Ptr, F>(v: P, v_len: usize, mut is_less: F)
889where
890	F: FnMut(P, P) -> bool,
891{
892	// Sorting has no meaningful behavior on zero-sized types.
893	if core::mem::size_of::<P::Item>() == 0 {
894		return;
895	}
896
897	// Limit the number of imbalanced partitions to `floor(log2(len)) + 1`.
898	let limit = usize::BITS - v_len.leading_zeros();
899
900	recurse(v, v_len, &mut is_less, None, limit);
901}
902
903unsafe fn recurse<P: Ptr, F: FnMut(P, P) -> bool>(mut v: P, mut v_len: usize, is_less: &mut F, mut pred: Option<P>, mut limit: u32) {
904	// Slices of up to this length get sorted using insertion sort.
905	const MAX_INSERTION: usize = 20;
906
907	// True if the last partitioning was reasonably balanced.
908	let mut was_balanced = true;
909	// True if the last partitioning didn't shuffle elements (the slice was already partitioned).
910	let mut was_partitioned = true;
911
912	loop {
913		let len = v_len;
914
915		// Very short slices get sorted using insertion sort.
916		if len <= MAX_INSERTION {
917			if len >= 2 {
918				insertion_sort_shift_left(v, v_len, 1, is_less);
919			}
920			return;
921		}
922
923		// If too many bad pivot choices were made, simply fall back to heapsort in order to
924		// guarantee `O(n * log(n))` worst-case.
925		if limit == 0 {
926			heapsort(v, v_len, is_less);
927			return;
928		}
929
930		// If the last partitioning was imbalanced, try breaking patterns in the slice by shuffling
931		// some elements around. Hopefully we'll choose a better pivot this time.
932		if !was_balanced {
933			break_patterns(v, v_len);
934			limit -= 1;
935		}
936
937		// Choose a pivot and try guessing whether the slice is already sorted.
938		let (pivot, likely_sorted) = choose_pivot(v, v_len, is_less);
939
940		// If the last partitioning was decently balanced and didn't shuffle elements, and if pivot
941		// selection predicts the slice is likely already sorted...
942		if was_balanced && was_partitioned && likely_sorted {
943			// Try identifying several out-of-order elements and shifting them to correct
944			// positions. If the slice ends up being completely sorted, we're done.
945			if partial_insertion_sort(v, v_len, is_less) {
946				return;
947			}
948		}
949
950		// If the chosen pivot is equal to the predecessor, then it's the smallest element in the
951		// slice. Partition the slice into elements equal to and elements greater than the pivot.
952		// This case is usually hit when the slice contains many duplicate elements.
953		if let Some(p) = pred {
954			if !is_less(p, v.add(pivot)) {
955				let mid = partition_equal(v, v_len, pivot, is_less);
956
957				// Continue sorting elements greater than the pivot.
958				v = v.add(mid);
959				v_len -= mid;
960				continue;
961			}
962		}
963
964		// Partition the slice.
965		let (mid, was_p) = partition(v, v_len, pivot, is_less);
966		was_balanced = Ord::min(mid, len - mid) >= len / 8;
967		was_partitioned = was_p;
968
969		// Split the slice into `left`, `pivot`, and `right`.
970		let left = v;
971		let left_len = mid;
972		let right = v.add(mid);
973		let right_len = v_len - mid;
974		let pivot = right;
975		let right = right.add(1);
976		let right_len = right_len - 1;
977
978		// Recurse into the shorter side only in order to minimize the total number of recursive
979		// calls and consume less stack space. Then just continue with the longer side (this is
980		// akin to tail recursion).
981		if left_len < right_len {
982			recurse(left, left_len, is_less, pred, limit);
983			v = right;
984			v_len = right_len;
985			pred = Some(pivot);
986		} else {
987			recurse(right, right_len, is_less, Some(pivot), limit);
988			v = left;
989			v_len = left_len;
990		}
991	}
992}

994pub unsafe fn sort_unstable_by<P: Ptr>(ptr: P, len: usize, compare: impl FnMut(P, P) -> core::cmp::Ordering) {
995	let mut compare = compare;
996	quicksort(
997		ptr,
998		len,
999		#[inline(always)]
1000		|a, b| compare(a, b) == core::cmp::Ordering::Less,
1001	);
1002}

1004pub unsafe fn sort_indices<I: crate::Index, T>(indices: &mut [I], values: &mut [T]) {
1005	let len = indices.len();
1006	debug_assert!(values.len() == len);
1007
1008	sort_unstable_by((indices.as_mut_ptr(), values.as_mut_ptr()), len, |(i, _), (j, _)| (*i).cmp(&*j));
1009}
