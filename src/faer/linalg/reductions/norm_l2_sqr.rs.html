7#[inline(always)]
8#[math]
9fn norm_l2_sqr_simd<'N, T: ComplexField>(data: ColRef<'_, T, Dim<'N>, ContiguousFwd>) -> T::Real {
10	struct Impl<'a, 'N, T: ComplexField> {
11		data: ColRef<'a, T, Dim<'N>, ContiguousFwd>,
12	}
13
14	impl<'N, T: ComplexField> pulp::WithSimd for Impl<'_, 'N, T> {
15		type Output = T::Real;
16
17		#[inline(always)]
18		fn with_simd<S: pulp::Simd>(self, simd: S) -> Self::Output {
19			let Self { data } = self;
20			let simd = SimdCtx::<T, S>::new(T::simd_ctx(simd), data.nrows());
21
22			let zero = simd.splat(&zero());
23
24			let mut acc0 = RealReg(zero);
25			let mut acc1 = RealReg(zero);
26			let mut acc2 = RealReg(zero);
27			let mut acc3 = RealReg(zero);
28
29			let (head, body4, body1, tail) = simd.batch_indices::<4>();
30			if let Some(i0) = head {
31				let x0 = simd.read(data, i0);
32				acc0 = simd.abs2_add(x0, acc0);
33			}
34			for [i0, i1, i2, i3] in body4 {
35				let x0 = simd.read(data, i0);
36				let x1 = simd.read(data, i1);
37				let x2 = simd.read(data, i2);
38				let x3 = simd.read(data, i3);
39
40				acc0 = simd.abs2_add(x0, acc0);
41				acc1 = simd.abs2_add(x1, acc1);
42				acc2 = simd.abs2_add(x2, acc2);
43				acc3 = simd.abs2_add(x3, acc3);
44			}
45			for i0 in body1 {
46				let x0 = simd.read(data, i0);
47				acc0 = simd.abs2_add(x0, acc0);
48			}
49			if let Some(i0) = tail {
50				let x0 = simd.read(data, i0);
51				acc0 = simd.abs2_add(x0, acc0);
52			}
53
54			acc0 = RealReg(simd.add(acc0.0, acc1.0));
55			acc2 = RealReg(simd.add(acc2.0, acc3.0));
56			acc0 = RealReg(simd.add(acc0.0, acc2.0));
57
58			simd.reduce_sum_real(acc0)
59		}
60	}
61
62	dispatch!(Impl { data }, Impl, T)
63}

65#[math]
66fn norm_l2_sqr_simd_pairwise_rows<T: ComplexField>(data: ColRef<'_, T, usize, ContiguousFwd>) -> T::Real {
67	if data.nrows() <= LINEAR_IMPL_THRESHOLD {
68		with_dim!(N, data.nrows());
69
70		norm_l2_sqr_simd(data.as_row_shape(N))
71	} else {
72		let split_point = ((data.nrows() + 1) / 2).next_power_of_two();
73		let (head, tail) = data.split_at_row(split_point);
74		let acc0 = norm_l2_sqr_simd_pairwise_rows(head);
75		let acc1 = norm_l2_sqr_simd_pairwise_rows(tail);
76
77		acc0 + acc1
78	}
79}

81#[math]
82fn norm_l2_sqr_simd_pairwise_cols<T: ComplexField>(data: MatRef<'_, T, usize, usize, ContiguousFwd>) -> T::Real {
83	if data.ncols() == 1 {
84		norm_l2_sqr_simd_pairwise_rows(data.col(0))
85	} else {
86		let split_point = ((data.ncols() + 1) / 2).next_power_of_two();
87		let (head, tail) = data.split_at_col(split_point);
88		let acc0 = norm_l2_sqr_simd_pairwise_cols(head);
89		let acc1 = norm_l2_sqr_simd_pairwise_cols(tail);
90
91		acc0 + acc1
92	}
93}

95#[math]
96pub fn norm_l2_sqr<T: ComplexField>(mut mat: MatRef<'_, T>) -> T::Real {
97	if mat.ncols() > 1 && mat.col_stride().unsigned_abs() == 1 {
98		mat = mat.transpose();
99	}
100	if mat.row_stride() < 0 {
101		mat = mat.reverse_rows();
102	}
103
104	if mat.nrows() == 0 || mat.ncols() == 0 {
105		zero()
106	} else {
107		let m = mat.nrows();
108		let n = mat.ncols();
109
110		if try_const! { T::SIMD_CAPABILITIES.is_simd() } {
111			if let Some(mat) = mat.try_as_col_major() {
112				if try_const! { T::IS_NATIVE_C32 } {
113					let mat: MatRef<'_, Complex<f32>, usize, usize, ContiguousFwd> = unsafe { crate::hacks::coerce(mat) };
114					let mat = unsafe {
115						MatRef::<'_, f32, usize, usize, ContiguousFwd>::from_raw_parts(
116							mat.as_ptr() as *const f32,
117							2 * mat.nrows(),
118							mat.ncols(),
119							ContiguousFwd,
120							mat.col_stride().wrapping_mul(2),
121						)
122					};
123					return unsafe { crate::hacks::coerce(norm_l2_sqr_simd_pairwise_cols::<f32>(mat)) };
124				} else if try_const! { T::IS_NATIVE_C64 } {
125					let mat: MatRef<'_, Complex<f64>, usize, usize, ContiguousFwd> = unsafe { crate::hacks::coerce(mat) };
126					let mat = unsafe {
127						MatRef::<'_, f64, usize, usize, ContiguousFwd>::from_raw_parts(
128							mat.as_ptr() as *const f64,
129							2 * mat.nrows(),
130							mat.ncols(),
131							ContiguousFwd,
132							mat.col_stride().wrapping_mul(2),
133						)
134					};
135					return unsafe { crate::hacks::coerce(norm_l2_sqr_simd_pairwise_cols::<f64>(mat)) };
136				} else {
137					return norm_l2_sqr_simd_pairwise_cols(mat);
138				}
139			}
140		}
141
142		let mut acc = zero();
143		for j in 0..n {
144			for i in 0..m {
145				acc = acc + abs2(mat[(i, j)]);
146			}
147		}
148		acc
149	}
150}

152#[cfg(test)]
153mod tests {
154	use super::*;
155	use crate::{Col, Mat, assert, unzip, zip};
156
157	#[test]
158	fn test_norm_l2_sqr() {
159		let relative_err = |a: f64, b: f64| (a - b).abs() / f64::max(a.abs(), b.abs());
160
161		for (m, n) in [(9, 10), (1023, 5), (42, 1)] {
162			for factor in [0.0, 1.0, 1e30, 1e120, 1e-30, 1e-250] {
163				let mat = Mat::from_fn(m, n, |i, j| factor * ((i + j) as f64));
164				let mut target = 0.0;
165				zip!(mat.as_ref()).for_each(|unzip!(x)| {
166					target += x * x;
167				});
168
169				if target == 0.0 {
170					assert!(norm_l2_sqr(mat.as_ref()) == target);
171				} else {
172					assert!(relative_err(norm_l2_sqr(mat.as_ref()), target) < 1e-14);
173				}
174			}
175		}
176
177		let mat = Col::from_fn(10000000, |_| 0.3);
178		let target = 0.3 * 0.3 * 10000000.0f64;
179		assert!(relative_err(norm_l2_sqr(mat.as_ref().as_mat()), target) < 1e-14);
180	}
