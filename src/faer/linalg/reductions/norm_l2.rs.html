7#[inline(always)]
8#[math]
9fn norm_l2_simd<'N, T: ComplexField>(data: ColRef<'_, T, Dim<'N>, ContiguousFwd>) -> [T::Real; 3] {
10	struct Impl<'a, 'N, T: ComplexField> {
11		data: ColRef<'a, T, Dim<'N>, ContiguousFwd>,
12	}
13
14	impl<'N, T: ComplexField> pulp::WithSimd for Impl<'_, 'N, T> {
15		type Output = [T::Real; 3];
16
17		#[inline(always)]
18		fn with_simd<S: pulp::Simd>(self, simd: S) -> Self::Output {
19			let Self { data } = self;
20			let simd = SimdCtx::<T, S>::new(T::simd_ctx(simd), data.nrows());
21
22			let zero = simd.splat(&zero());
23
24			let sml = simd.splat_real(&sqrt_min_positive());
25			let big = simd.splat_real(&sqrt_max_positive());
26
27			let mut acc0_sml = RealReg(zero);
28			let mut acc1_sml = RealReg(zero);
29			let mut acc0_med = RealReg(zero);
30			let mut acc1_med = RealReg(zero);
31			let mut acc0_big = RealReg(zero);
32			let mut acc1_big = RealReg(zero);
33
34			let (head, body2, body1, tail) = simd.batch_indices::<2>();
35
36			if let Some(i0) = head {
37				let x0 = simd.read(data, i0);
38
39				acc0_sml = simd.abs2_add(simd.mul_real(x0, sml), acc0_sml);
40				acc0_med = simd.abs2_add(x0, acc0_med);
41				acc0_big = simd.abs2_add(simd.mul_real(x0, big), acc0_big);
42			}
43			for [i0, i1] in body2 {
44				let x0 = simd.read(data, i0);
45				let x1 = simd.read(data, i1);
46
47				acc0_sml = simd.abs2_add(simd.mul_real(x0, sml), acc0_sml);
48				acc1_sml = simd.abs2_add(simd.mul_real(x1, sml), acc1_sml);
49
50				acc0_med = simd.abs2_add(x0, acc0_med);
51				acc1_med = simd.abs2_add(x1, acc1_med);
52
53				acc0_big = simd.abs2_add(simd.mul_real(x0, big), acc0_big);
54				acc1_big = simd.abs2_add(simd.mul_real(x1, big), acc1_big);
55			}
56			for i0 in body1 {
57				let x0 = simd.read(data, i0);
58
59				acc0_sml = simd.abs2_add(simd.mul_real(x0, sml), acc0_sml);
60				acc0_med = simd.abs2_add(x0, acc0_med);
61				acc0_big = simd.abs2_add(simd.mul_real(x0, big), acc0_big);
62			}
63			if let Some(i0) = tail {
64				let x0 = simd.read(data, i0);
65
66				acc0_sml = simd.abs2_add(simd.mul_real(x0, sml), acc0_sml);
67				acc0_med = simd.abs2_add(x0, acc0_med);
68				acc0_big = simd.abs2_add(simd.mul_real(x0, big), acc0_big);
69			}
70
71			acc0_sml = RealReg(simd.add(acc0_sml.0, acc1_sml.0));
72			acc0_big = RealReg(simd.add(acc0_big.0, acc1_big.0));
73			acc0_med = RealReg(simd.add(acc0_med.0, acc1_med.0));
74
75			[
76				simd.reduce_sum_real(acc0_sml),
77				simd.reduce_sum_real(acc0_med),
78				simd.reduce_sum_real(acc0_big),
79			]
80		}
81	}
82
83	dispatch!(Impl { data }, Impl, T)
84}

86#[math]
87fn norm_l2_simd_pairwise_rows<T: ComplexField>(data: ColRef<'_, T, usize, ContiguousFwd>) -> [T::Real; 3] {
88	if data.nrows() <= LINEAR_IMPL_THRESHOLD {
89		with_dim!(N, data.nrows());
90
91		norm_l2_simd(data.as_row_shape(N))
92	} else {
93		let split_point = ((data.nrows() + 1) / 2).next_power_of_two();
94		let (head, tail) = data.split_at_row(split_point);
95		let acc0 = norm_l2_simd_pairwise_rows(head);
96		let acc1 = norm_l2_simd_pairwise_rows(tail);
97
98		[add(acc0[0], acc1[0]), add(acc0[1], acc1[1]), add(acc0[2], acc1[2])]
99	}
100}

102#[math]
103fn norm_l2_simd_pairwise_cols<T: ComplexField>(data: MatRef<'_, T, usize, usize, ContiguousFwd>) -> [T::Real; 3] {
104	if data.ncols() == 1 {
105		norm_l2_simd_pairwise_rows(data.col(0))
106	} else {
107		let split_point = ((data.ncols() + 1) / 2).next_power_of_two();
108		let (head, tail) = data.split_at_col(split_point);
109		let acc0 = norm_l2_simd_pairwise_cols(head);
110		let acc1 = norm_l2_simd_pairwise_cols(tail);
111
112		[add(acc0[0], acc1[0]), add(acc0[1], acc1[1]), add(acc0[2], acc1[2])]
113	}
114}

116#[math]
117pub fn norm_l2_x3<T: ComplexField>(mut mat: MatRef<'_, T>) -> [T::Real; 3] {
118	if mat.ncols() > 1 && mat.col_stride().unsigned_abs() == 1 {
119		mat = mat.transpose();
120	}
121	if mat.row_stride() < 0 {
122		mat = mat.reverse_rows();
123	}
124
125	if mat.nrows() == 0 || mat.ncols() == 0 {
126		[zero(), zero(), zero()]
127	} else {
128		let m = mat.nrows();
129		let n = mat.ncols();
130
131		if try_const! { T::SIMD_CAPABILITIES.is_simd() } {
132			if let Some(mat) = mat.try_as_col_major() {
133				if try_const! { T::IS_NATIVE_C32 } {
134					let mat: MatRef<'_, Complex<f32>, usize, usize, ContiguousFwd> = unsafe { crate::hacks::coerce(mat) };
135					let mat = unsafe {
136						MatRef::<'_, f32, usize, usize, ContiguousFwd>::from_raw_parts(
137							mat.as_ptr() as *const f32,
138							2 * mat.nrows(),
139							mat.ncols(),
140							ContiguousFwd,
141							mat.col_stride().wrapping_mul(2),
142						)
143					};
144					return unsafe { crate::hacks::coerce(norm_l2_simd_pairwise_cols::<f32>(mat)) };
145				} else if try_const! { T::IS_NATIVE_C64 } {
146					let mat: MatRef<'_, Complex<f64>, usize, usize, ContiguousFwd> = unsafe { crate::hacks::coerce(mat) };
147					let mat = unsafe {
148						MatRef::<'_, f64, usize, usize, ContiguousFwd>::from_raw_parts(
149							mat.as_ptr() as *const f64,
150							2 * mat.nrows(),
151							mat.ncols(),
152							ContiguousFwd,
153							mat.col_stride().wrapping_mul(2),
154						)
155					};
156					return unsafe { crate::hacks::coerce(norm_l2_simd_pairwise_cols::<f64>(mat)) };
157				} else {
158					return norm_l2_simd_pairwise_cols(mat);
159				}
160			}
161		}
162
163		let sml = min_positive::<T::Real>();
164		let big = max_positive::<T::Real>();
165		let mut acc = zero();
166		for j in 0..n {
167			for i in 0..m {
168				acc = hypot(acc, abs(mat[(i, j)]));
169			}
170		}
171		acc = abs2(acc);
172		[sml * acc, copy(acc), big * acc]
173	}
174}

176#[math]
177pub fn norm_l2<T: ComplexField>(mat: MatRef<'_, T>) -> T::Real {
178	let [acc_sml, acc_med, acc_big] = norm_l2_x3(mat);
179
180	let sml = sqrt_min_positive();
181	let big = sqrt_max_positive();
182
183	if acc_sml >= one() {
184		sqrt(acc_sml) * big
185	} else if acc_med >= one() {
186		sqrt(acc_med)
187	} else {
188		sqrt(acc_big) * sml
189	}
190}

197	#[test]
198	fn test_norm_l2() {
199		let relative_err = |a: f64, b: f64| (a - b).abs() / f64::max(a.abs(), b.abs());
200
201		for (m, n) in [(9, 10), (1023, 5), (42, 1)] {
202			for factor in [0.0, 1.0, 1e30, 1e250, 1e-30, 1e-250] {
203				let mat = Mat::from_fn(m, n, |i, j| factor * ((i + j) as f64));
204				let mut target = 0.0;
205				zip!(mat.as_ref()).for_each(|unzip!(x)| {
206					target = f64::hypot(*x, target);
207				});
208
209				if factor == 0.0 {
210					assert!(norm_l2(mat.as_ref()) == target);
211				} else {
212					assert!(relative_err(norm_l2(mat.as_ref()), target) < 1e-14);
213				}
214			}
215		}
216
217		let mat = Col::from_fn(10000000, |_| 0.3);
218		let target = (0.3 * 0.3 * 10000000.0f64).sqrt();
219		assert!(relative_err(norm_l2(mat.as_ref().as_mat()), target) < 1e-14);
220	}
