20#[math]
21#[allow(dead_code)]
22fn tridiag_to_mat<T: RealField>(diag: ColRef<'_, T, usize, ContiguousFwd>, offdiag: ColRef<'_, T, usize, ContiguousFwd>) -> Mat<T, usize, usize> {
23	let n = diag.nrows();
24	let mut m = Mat::zeros(n, n);
25
26	{
27		let mut m = m.as_mut();
28		for i in 0..n {
29			(m[(i, i)] = copy(diag[i]));
30			if i + 1 < n {
31				(m[(i + 1, i)] = copy(offdiag[i]));
32				(m[(i, i + 1)] = copy(offdiag[i]));
33			}
34		}
35	}
36	m
37}

39#[math]
40pub(crate) fn qr_algorithm<T: RealField>(
41	diag: ColMut<'_, T, usize, ContiguousFwd>,
42	offdiag: ColMut<'_, T, usize, ContiguousFwd>,
43	u: Option<MatMut<'_, T, usize, usize>>,
44) -> Result<(), EvdError> {
45	let n = diag.nrows();
46
47	let mut u = u;
48	let mut diag = diag;
49	let mut offdiag = offdiag;
50
51	if let Some(mut u) = u.rb_mut() {
52		u.fill(zero());
53		u.diagonal_mut().fill(one());
54	}
55
56	if n <= 1 {
57		return Ok(());
58	}
59
60	let eps = eps();
61	let sml = min_positive();
62
63	if n == 2 {
64		let a = copy(diag[0]);
65		let d = copy(diag[1]);
66		let b = copy(offdiag[0]);
67
68		let half = from_f64(0.5);
69
70		let t0 = hypot(a - d, b * from_f64(2.0)) * half;
71		let t1 = (a + d) * half;
72		let r0 = t1 - t0;
73		let r1 = t1 + t0;
74
75		let tol = max(abs(r0), abs(r1)) * eps;
76
77		if let Some(mut u) = u.rb_mut() {
78			if r1 - r0 <= tol {
79				u[(0, 0)] = one();
80				u[(1, 0)] = zero();
81				u[(0, 1)] = zero();
82				u[(1, 1)] = one();
83			} else if abs(b) <= tol {
84				if diag[0] < diag[1] {
85					u[(0, 0)] = one();
86					u[(1, 0)] = zero();
87					u[(0, 1)] = zero();
88					u[(1, 1)] = one();
89				} else {
90					u[(0, 0)] = zero();
91					u[(1, 0)] = one();
92					u[(0, 1)] = one();
93					u[(1, 1)] = zero();
94				}
95			} else {
96				let tau = ((d - a) / b) * half;
97				let mut t = recip(abs(tau) + hypot(one(), tau));
98				if tau < zero() {
99					t = -t;
100				}
101
102				let c = hypot(one(), t);
103				let s = c * t;
104
105				let r = hypot(c, s);
106				let c = c / r;
107				let s = s / r;
108
109				let r0_r = (c * a - s * b) / c;
110				if abs(r0 - r0_r) < r1 - r0_r {
111					u[(0, 0)] = copy(c);
112					u[(1, 0)] = -(s);
113					u[(0, 1)] = copy(s);
114					u[(1, 1)] = copy(c);
115				} else {
116					u[(0, 1)] = copy(c);
117					u[(1, 1)] = -(s);
118					u[(0, 0)] = copy(s);
119					u[(1, 0)] = copy(c);
120				}
121			}
122		}
123
124		(diag[0] = r0);
125		(diag[1] = r1);
126
127		return Ok(());
128	}
129
130	let max = max(diag.norm_max(), offdiag.norm_max());
131	if max == zero() {
132		return Ok(());
133	}
134	let max_inv = recip(max);
135
136	for x in diag.rb_mut().iter_mut() {
137		(*x = *x * max_inv);
138	}
139	for x in offdiag.rb_mut().iter_mut() {
140		*x = *x * max_inv;
141	}
142
143	let mut end = n - 1;
144	let mut start = 0;
145
146	let max_iters = Ord::max(30, nbits::<T>() / 2).saturating_mul(n).saturating_mul(n);
147
148	for iter in 0..max_iters {
149		for i in start..end {
150			if abs(offdiag[i]) < sml || abs(offdiag[i]) < eps * hypot(diag[i], diag[i + 1]) {
151				(offdiag[i] = zero());
152			}
153		}
154
155		while end > 0 && (offdiag[end - 1] == zero()) {
156			end -= 1;
157		}
158
159		if end == 0 {
160			break;
161		}
162
163		if iter + 1 == max_iters {
164			for x in diag.rb_mut().iter_mut() {
165				*x = *x * max;
166			}
167			for x in offdiag.rb_mut().iter_mut() {
168				*x = *x * max;
169			}
170
171			return Err(EvdError::NoConvergence);
172		}
173
174		start = end - 1;
175		while start > 0 && !(offdiag[start - 1] == zero()) {
176			start -= 1;
177		}
178
179		{
180			// Wilkinson shift
181			let td = (diag[end - 1] - diag[end]) * from_f64(0.5);
182			let e = copy(offdiag[end - 1]);
183			let mut mu = copy(diag[end]);
184
185			if td == zero() {
186				mu = mu - abs(e);
187			} else if !(e == zero()) {
188				let e2 = abs2(e);
189				let h = hypot(td, e);
190				let h = if td > zero() { copy(h) } else { -h };
191				if e2 == zero() {
192					mu = mu - e / ((td + h) / e)
193				} else {
194					mu = mu - e2 / (td + h)
195				};
196			}
197
198			let mut x = diag[start] - mu;
199			let mut z = copy(offdiag[start]);
200
201			let mut k = start;
202			while k < end && !(z == zero()) {
203				let rot = JacobiRotation::make_givens(copy(x), copy(z));
204
205				// T = G' T G
206				let sdk = rot.s * diag[k] + rot.c * offdiag[k];
207				let dkp1 = rot.s * offdiag[k] + rot.c * diag[k + 1];
208
209				(diag[k] = rot.c * (rot.c * diag[k] - rot.s * offdiag[k]) - rot.s * (rot.c * offdiag[k] - rot.s * diag[k + 1]));
210				(diag[k + 1] = rot.s * sdk + rot.c * dkp1);
211				(offdiag[k] = rot.c * sdk - rot.s * dkp1);
212
213				if k > start {
214					(offdiag[k - 1] = rot.c * offdiag[k - 1] - rot.s * z);
215				}
216
217				x = copy(offdiag[k]);
218
219				if k < end - 1 {
220					z = -rot.s * offdiag[k + 1];
221					(offdiag[k + 1] = rot.c * offdiag[k + 1]);
222				}
223
224				if let Some(u) = u.rb_mut() {
225					rot.apply_on_the_right_in_place(u.two_cols_mut(k + 1, k));
226				}
227				k += 1;
228			}
229		}
230	}
231
232	for i in 0..n - 1 {
233		let mut idx = i;
234		let mut min = copy(diag[i]);
235
236		for k in i + 1..n {
237			if diag[k] < min {
238				idx = k;
239				min = copy(diag[k]);
240			}
241		}
242
243		if idx != i {
244			let (a, b) = (copy(diag[i]), copy(diag[idx]));
245			(diag[i] = b);
246			(diag[idx] = a);
247			if let Some(mut u) = u.rb_mut() {
248				swap_cols_idx(u.rb_mut(), i, idx);
249			}
250		}
251	}
252
253	for x in diag.rb_mut().iter_mut() {
254		(*x = *x * max);
255	}
256
257	Ok(())
258}

260#[math]
261fn secular_eq<T: RealField>(shift: T, mu: T, d: ColRef<'_, T, usize, ContiguousFwd>, z: ColRef<'_, T, usize, ContiguousFwd>, rho_recip: T) -> T {
262	with_dim!(n, d.nrows());
263	let d = d.as_row_shape(n);
264	let z = z.as_row_shape(n);
265
266	let mut res = rho_recip;
267
268	for i in n.indices() {
269		let d = copy(d[i]);
270		let z = copy(z[i]);
271
272		res = res + z * (z / ((d - shift) - mu));
273	}
274
275	res
276}

278#[math]
279fn batch_secular_eq<const N: usize, T: RealField>(
280	shift: &[T; N],
281	mu: &[T; N],
282	d: ColRef<'_, T, usize, ContiguousFwd>,
283	z: ColRef<'_, T, usize, ContiguousFwd>,
284	rho_recip: T,
285) -> [T; N] {
286	with_dim!(n, d.nrows());
287	let d = d.as_row_shape(n);
288	let z = z.as_row_shape(n);
289
290	let mut res = [(); N].map(|_| copy(rho_recip));
291	for i in n.indices() {
292		let d = copy(d[i]);
293		let z = copy(z[i]);
294
295		for ((res, mu), shift) in res.iter_mut().zip(mu.iter()).zip(shift.iter()) {
296			*res = *res + z * (z / ((d - *shift) - *mu));
297		}
298	}
299	res
300}

302#[math]
303fn compute_eigenvalues<T: RealField>(
304	mut mus: ColMut<'_, T, usize, ContiguousFwd>,
305	mut shifts: ColMut<'_, T, usize, ContiguousFwd>,
306	d: ColRef<'_, T, usize, ContiguousFwd>,
307	z: ColRef<'_, T, usize, ContiguousFwd>,
308	rho: T,
309	non_deflated: usize,
310) {
311	let n = non_deflated;
312	let full_n = d.nrows();
313
314	let rho_recip = recip(rho);
315
316	for i in 0..n {
317		let left = copy(d[i]);
318		let last = i == n - 1;
319
320		let right = if last { d[i] + rho * z.squared_norm_l2() } else { copy(d[i + 1]) };
321
322		let d = d.subrows(0, n);
323		let z = z.subrows(0, n);
324
325		let (shift, mu) = secular_eq_root_finder(
326			&|shift, mu| secular_eq(shift, mu, d, z, copy(rho_recip)),
327			&|shift, mu| batch_secular_eq(shift, mu, d, z, copy(rho_recip)),
328			left,
329			right,
330			last,
331		);
332
333		(shifts[i] = shift);
334		(mus[i] = mu);
335	}
336	for i in n..full_n {
337		(shifts[i] = zero());
338		(mus[i] = copy(d[i]));
339	}
340}

342#[math]
343fn divide_and_conquer_recurse<T: RealField>(
344	mut diag: ColMut<'_, T, usize, ContiguousFwd>,
345	mut offdiag: ColMut<'_, T, usize, ContiguousFwd>,
346	mut u: MatMut<'_, T, usize, usize>,
347
348	par: Par,
349
350	pl_before: &mut [usize],
351	pl_after: &mut [usize],
352	pr: &mut [usize],
353	run_info: &mut [usize],
354
355	mut z: ColMut<'_, T, usize, ContiguousFwd>,
356	mut permuted_diag: ColMut<'_, T, usize, ContiguousFwd>,
357	mut permuted_z: ColMut<'_, T, usize, ContiguousFwd>,
358	mut householder: ColMut<'_, T, usize, ContiguousFwd>,
359	mut mus: ColMut<'_, T, usize, ContiguousFwd>,
360	mut shifts: ColMut<'_, T, usize, ContiguousFwd>,
361	mut repaired_u: MatMut<'_, T, usize, usize, ContiguousFwd>,
362	mut tmp: MatMut<'_, T, usize, usize, ContiguousFwd>,
363	qr_fallback_threshold: usize,
364) -> Result<(), EvdError> {
365	let n = diag.nrows();
366	let qr_fallback_threshold = Ord::max(qr_fallback_threshold, 4);
367
368	if n <= qr_fallback_threshold {
369		return qr_algorithm(diag, offdiag, Some(u));
370	}
371
372	let n1 = n / 2;
373	let mut rho = copy(offdiag[n1 - 1]);
374	let (mut diag0, mut diag1) = diag.rb_mut().split_at_row_mut(n1);
375	let (offdiag0, offdiag1) = offdiag.rb_mut().split_at_row_mut(n1 - 1);
376	let offdiag1 = offdiag1.split_at_row_mut(1).1;
377
378	(diag0[n1 - 1] = diag0[n1 - 1] - abs(rho));
379	(diag1[0] = diag1[0] - abs(rho));
380
381	let (mut u0, _, _, mut u1) = u.rb_mut().split_at_mut(n1, n1);
382	{
383		let (pl_before0, pl_before1) = pl_before.split_at_mut(n1);
384		let (pl_after0, pl_after1) = pl_after.split_at_mut(n1);
385		let (pr0, pr1) = pr.split_at_mut(n1);
386		let (run_info0, run_info1) = run_info.split_at_mut(n1);
387		let (z0, z1) = z.rb_mut().split_at_row_mut(n1);
388		let (permuted_diag0, permuted_diag1) = permuted_diag.rb_mut().split_at_row_mut(n1);
389		let (permuted_z0, permuted_z1) = permuted_z.rb_mut().split_at_row_mut(n1);
390		let (householder0, householder1) = householder.rb_mut().split_at_row_mut(n1);
391		let (mus0, mus1) = mus.rb_mut().split_at_row_mut(n1);
392		let (shift0, shift1) = shifts.rb_mut().split_at_row_mut(n1);
393
394		let (repaired_u0, _, _, repaired_u1) = repaired_u.rb_mut().split_at_mut(n1, n1);
395		let (tmp0, _, _, tmp1) = tmp.rb_mut().split_at_mut(n1, n1);
396
397		let mut r0 = Ok(());
398		let mut r1 = Ok(());
399		join_raw(
400			|par| {
401				r0 = divide_and_conquer_recurse(
402					diag0,
403					offdiag0,
404					u0.rb_mut(),
405					par,
406					pl_before0,
407					pl_after0,
408					pr0,
409					run_info0,
410					z0,
411					permuted_diag0,
412					permuted_z0,
413					householder0,
414					mus0,
415					shift0,
416					repaired_u0,
417					tmp0,
418					qr_fallback_threshold,
419				);
420			},
421			|par| {
422				r1 = divide_and_conquer_recurse(
423					diag1,
424					offdiag1,
425					u1.rb_mut(),
426					par,
427					pl_before1,
428					pl_after1,
429					pr1,
430					run_info1,
431					z1,
432					permuted_diag1,
433					permuted_z1,
434					householder1,
435					mus1,
436					shift1,
437					repaired_u1,
438					tmp1,
439					qr_fallback_threshold,
440				);
441			},
442			par,
443		);
444		r0?;
445		r1?;
446	}
447
448	let mut repaired_u = repaired_u.subrows_mut(0, n);
449	let mut tmp = tmp.subrows_mut(0, n);
450
451	//     [Q0   0] ([D0   0]            ) [Q0.T     0]
452	// T = [ 0  Q1]×([ 0  D1] + rho×z×z.T)×[   0  Q1.T]
453	//
454	// we compute the permutation Pl_before that sorts diag([D0 D1])
455	// we apply householder transformations to segments of permuted_z that correspond to
456	// consecutive almost-equal eigenvalues
457	// we compute the permutation Pl_after that places the deflated eigenvalues at the end
458	//
459	// we compute the EVD of:
460	// Pl_after × H × Pl_before × (diag([D0 D1]) + rho×z×z.T) × Pl_before.T × H.T × Pl_after.T =
461	// Q×W×Q.T
462	//
463	// we sort the eigenvalues in W
464	// Pr × W × Pr.T = E
465	//
466	// diag([D0 D1]) + rho×z×z.T = Pl_before^-1×H^-1×Pl_after^-1×Q×Pr × E × ...
467
468	let (mut z0, mut z1) = z.rb_mut().split_at_row_mut(n1);
469
470	z0.copy_from(u0.rb().row(n1 - 1).transpose());
471	if rho < zero() {
472		z!(z1.rb_mut(), u1.rb().row(0).transpose()).for_each(|uz!(z, u)| *z = -*u);
473	} else {
474		z1.copy_from(u1.rb().row(0).transpose());
475	}
476
477	let inv_sqrt2 = sqrt(from_f64(0.5));
478	z!(z.rb_mut()).for_each(|uz!(x)| *x = *x * inv_sqrt2);
479
480	rho = abs(rho * from_f64(2.0));
481
482	// merge two sorted diagonals
483	{
484		let mut i = 0usize;
485		let mut j = 0usize;
486		for p in &mut *pl_before {
487			if i == n1 {
488				*p = n1 + j;
489				j += 1;
490			} else if (j == n - n1) || diag[i] < diag[n1 + j] {
491				*p = i;
492				i += 1;
493			} else {
494				*p = n1 + j;
495				j += 1;
496			}
497		}
498	}
499
500	// permuted_diag = Pl * diag * Pl.T
501	// permuted_z = Pl * diag
502	for (i, &pl_before) in pl_before.iter().enumerate() {
503		(permuted_diag[i] = copy(diag[pl_before]));
504	}
505	for (i, &pl_before) in pl_before.iter().enumerate() {
506		(permuted_z[i] = copy(z[pl_before]));
507	}
508
509	let dmax = permuted_diag.norm_max();
510	let zmax = permuted_z.norm_max();
511
512	let eps = eps::<T>();
513	let tol = from_f64::<T>(8.0) * eps * max(dmax, zmax);
514
515	if rho * zmax <= tol {
516		// fill uninitialized values of u with zeros
517		// apply Pl to u on the right
518		// copy permuted_diag to diag
519		// return
520
521		let (mut tmp_tl, mut tmp_tr, mut tmp_bl, mut tmp_br) = tmp.rb_mut().split_at_mut(n1, n1);
522		tmp_tl.copy_from(u0.rb());
523		tmp_br.copy_from(u1.rb());
524		tmp_tr.fill(zero());
525		tmp_bl.fill(zero());
526
527		for (j, &pl_before) in pl_before.iter().enumerate() {
528			u.rb_mut().col_mut(j).copy_from(tmp.rb().col(pl_before));
529		}
530
531		for (j, diag) in diag.iter_mut().enumerate() {
532			(*diag = copy(permuted_diag[j]));
533		}
534
535		return Ok(());
536	}
537
538	for i in 0..n {
539		let zi = copy(permuted_z[i]);
540		if abs(rho * zi) <= tol {
541			permuted_z[i] = zero();
542		}
543	}
544
545	let mut applied_householder = false;
546	let mut idx = 0;
547	while idx < n {
548		let mut run_len = 1;
549
550		let d_prev = copy(permuted_diag[idx]);
551		while idx + run_len < n {
552			if permuted_diag[idx + run_len] - d_prev <= tol {
553				permuted_diag[idx + run_len] = copy(d_prev);
554				run_len += 1;
555			} else {
556				break;
557			}
558		}
559
560		run_info[idx..][..run_len].fill(run_len);
561		if run_len > 1 {
562			applied_householder = true;
563
564			let mut householder = householder.rb_mut().subrows_mut(idx, run_len);
565			let mut permuted_z = permuted_z.rb_mut().subrows_mut(idx, run_len);
566
567			householder.copy_from(permuted_z.rb());
568
569			let (tail, head) = householder.rb_mut().split_at_row_mut(run_len - 1);
570			let head = head.at_mut(0);
571			let HouseholderInfo { tau, .. } = make_householder_in_place(head, tail.as_dyn_stride_mut().reverse_rows_mut());
572			permuted_z.fill(zero());
573			(permuted_z[run_len - 1] = copy(head));
574			*head = tau;
575		}
576
577		idx += run_len;
578	}
579
580	// move deflated eigenvalues to the end
581	let mut writer_deflated = 0;
582	let mut writer_non_deflated = 0;
583	for reader in 0..n {
584		let z = copy(permuted_z[reader]);
585		let d = copy(permuted_diag[reader]);
586
587		if z == zero() {
588			// deflated value, store in diag
589			(diag[writer_deflated] = d);
590			pr[writer_deflated] = reader;
591			writer_deflated += 1;
592		} else {
593			(permuted_z[writer_non_deflated] = z);
594			(permuted_diag[writer_non_deflated] = d);
595			pl_after[writer_non_deflated] = reader;
596			writer_non_deflated += 1;
597		}
598	}
599
600	let non_deflated = writer_non_deflated;
601	let deflated = writer_deflated;
602
603	for i in 0..deflated {
604		(permuted_diag[non_deflated + i] = copy(diag[i]));
605		pl_after[non_deflated + i] = pr[i];
606	}
607
608	compute_eigenvalues(
609		mus.rb_mut(),
610		shifts.rb_mut(),
611		permuted_diag.rb(),
612		permuted_z.rb(),
613		copy(rho),
614		non_deflated,
615	);
616
617	// perturb z and rho
618	// we don't actually need rho for computing the eigenvectors so we're not going to perturb it
619
620	// new_zi^2 = prod(wk - di) / prod_{k != i} (dk - di)
621	for i in 0..non_deflated {
622		let di = copy(permuted_diag[i]);
623		let mu_i = copy(mus[i]);
624		let shift_i = copy(shifts[i]);
625
626		// NOTE: order of operations is crucial here
627		let mut prod = mu_i + (shift_i - di);
628
629		(0..i).chain(i + 1..non_deflated).for_each(|k| {
630			let dk = copy(permuted_diag[k]);
631			let mu_k = copy(mus[k]);
632			let shift_k = copy(shifts[k]);
633
634			let numerator = mu_k + (shift_k - di);
635			let denominator = dk - di;
636			prod = prod * (numerator / denominator);
637		});
638
639		let prod = sqrt(abs(prod));
640		let zi = permuted_z.rb_mut().at_mut(i);
641		let new_zi = if *zi < zero() { -prod } else { prod };
642
643		*zi = new_zi;
644	}
645
646	// reuse z to store computed eigenvalues, since it's not used anymore
647	let mut eigenvals = z;
648	for i in 0..n {
649		(eigenvals[i] = mus[i] + shifts[i]);
650	}
651
652	for (i, p) in pr.iter_mut().enumerate() {
653		*p = i;
654	}
655
656	pr.sort_unstable_by(|&i, &j| match PartialOrd::partial_cmp(&eigenvals[i], &eigenvals[j]) {
657		Some(ord) => ord,
658		None => {
659			if is_nan(eigenvals[i]) && is_nan(eigenvals[j]) {
660				core::cmp::Ordering::Equal
661			} else if is_nan(eigenvals[i]) {
662				core::cmp::Ordering::Greater
663			} else {
664				core::cmp::Ordering::Less
665			}
666		},
667	});
668
669	if !applied_householder {
670		for p in pl_after.iter_mut() {
671			*p = pl_before[*p];
672		}
673	}
674
675	// compute singular vectors
676	for (j, &pj) in pr.iter().enumerate() {
677		if pj >= non_deflated {
678			repaired_u.rb_mut().col_mut(j).fill(zero());
679			(repaired_u[(pl_after[pj], j)] = one());
680		} else {
681			let mu_j = copy(mus[pj]);
682			let shift_j = copy(shifts[pj]);
683
684			for (i, &pl_after) in pl_after[..non_deflated].iter().enumerate() {
685				let zi = copy(permuted_z[i]);
686				let di = copy(permuted_diag[i]);
687
688				(repaired_u[(pl_after, j)] = zi / ((di - shift_j) - mu_j));
689			}
690			for &pl_after in &pl_after[non_deflated..non_deflated + deflated] {
691				(repaired_u[(pl_after, j)] = zero());
692			}
693
694			let inv_norm = recip(repaired_u.rb().col(j).norm_l2());
695			z!(repaired_u.rb_mut().col_mut(j)).for_each(|unzip!(x)| *x = *x * inv_norm);
696		}
697	}
698
699	if applied_householder {
700		let mut idx = 0;
701		while idx < n {
702			let run_len = run_info[idx];
703
704			if run_len > 1 {
705				let mut householder = householder.rb_mut().subrows_mut(idx, run_len);
706				let tau = copy(householder[run_len - 1]);
707				(householder[run_len - 1] = one());
708				let householder = householder.rb();
709
710				let mut repaired_u = repaired_u.rb_mut().subrows_mut(idx, run_len);
711
712				let tau_inv = recip(tau);
713
714				for j in 0..n {
715					let mut col = repaired_u.rb_mut().col_mut(j);
716					let dot = tau_inv * dot::inner_prod(householder.as_dyn_stride().transpose(), Conj::No, col.rb().as_dyn_stride(), Conj::No);
717					z!(col.rb_mut(), householder).for_each(|uz!(u, h)| *u = *u - dot * h);
718				}
719			}
720			idx += run_len;
721		}
722
723		for j in 0..n {
724			for (i, &pl_before) in pl_before.iter().enumerate() {
725				(tmp[(pl_before, j)] = copy(repaired_u[(i, j)]));
726			}
727		}
728
729		core::mem::swap(&mut repaired_u, &mut tmp);
730	}
731
732	// multiply u by repaired_u (taking into account uninitialized values and sparsity structure)
733	//     [u0   0]
734	// u = [ 0  u1], u0 is n1×n1, u1 is (n-n1)×(n-n1)
735
736	// compute: u×repaired_u
737	// partition repaired_u
738	//
739	//              [u'_0]
740	// repaired_u = [u'_1], u'_0 is n1×n, u'_1 is (n-n1)×n
741	//
742	//                  [u0×u'_0]
743	// u × repaired_u = [u1×u'_1]
744
745	let (repaired_u_top, repaired_u_bot) = repaired_u.rb().split_at_row(n1);
746	let (tmp_top, tmp_bot) = tmp.rb_mut().split_at_row_mut(n1);
747
748	crate::utils::thread::join_raw(
749		|par| matmul(tmp_top, Accum::Replace, u0.rb(), repaired_u_top, one(), par),
750		|par| matmul(tmp_bot, Accum::Replace, u1.rb(), repaired_u_bot, one(), par),
751		par,
752	);
753
754	u.copy_from(tmp.rb());
755	for i in 0..n {
756		(diag[i] = mus[pr[i]] + shifts[pr[i]]);
757	}
758
759	Ok(())
760}

762#[math]
763pub(crate) fn divide_and_conquer<T: RealField>(
764	diag: ColMut<'_, T, usize, ContiguousFwd>,
765	offdiag: ColMut<'_, T, usize, ContiguousFwd>,
766	u: MatMut<'_, T, usize, usize>,
767
768	par: Par,
769	stack: &mut MemStack,
770	qr_fallback_threshold: usize,
771) -> Result<(), EvdError> {
772	let n = diag.nrows();
773	let (mut pl_before, stack) = stack.make_with(n, |_| 0usize);
774	let (mut pl_after, stack) = stack.make_with(n, |_| 0usize);
775	let (mut pr, stack) = stack.make_with(n, |_| 0usize);
776	let (mut run_info, stack) = stack.make_with(n, |_| 0usize);
777	let (mut z, stack) = unsafe { temp_mat_uninit(n, 1, stack) };
778	let (mut permuted_diag, stack) = unsafe { temp_mat_uninit(n, 1, stack) };
779	let (mut permuted_z, stack) = unsafe { temp_mat_uninit(n, 1, stack) };
780	let (mut householder, stack) = unsafe { temp_mat_uninit(n, 1, stack) };
781	let (mut mus, stack) = unsafe { temp_mat_uninit(n, 1, stack) };
782	let (mut shifts, stack) = unsafe { temp_mat_uninit(n, 1, stack) };
783	let (mut repaired_u, stack) = unsafe { temp_mat_uninit(n, n, stack) };
784	let (mut tmp, _) = unsafe { temp_mat_uninit(n, n, stack) };
785
786	divide_and_conquer_recurse(
787		diag,
788		offdiag,
789		u,
790		par,
791		&mut pl_before,
792		&mut pl_after,
793		&mut pr,
794		&mut run_info,
795		z.as_mat_mut().col_mut(0).try_as_col_major_mut().unwrap(),
796		permuted_diag.as_mat_mut().col_mut(0).try_as_col_major_mut().unwrap(),
797		permuted_z.as_mat_mut().col_mut(0).try_as_col_major_mut().unwrap(),
798		householder.as_mat_mut().col_mut(0).try_as_col_major_mut().unwrap(),
799		mus.as_mat_mut().col_mut(0).try_as_col_major_mut().unwrap(),
800		shifts.as_mat_mut().col_mut(0).try_as_col_major_mut().unwrap(),
801		repaired_u.as_mat_mut().try_as_col_major_mut().unwrap(),
802		tmp.as_mat_mut().try_as_col_major_mut().unwrap(),
803		qr_fallback_threshold,
804	)
805}

807pub(crate) fn divide_and_conquer_scratch<T: ComplexField>(n: usize, par: Par) -> StackReq {
808	let _ = par;
809	StackReq::all_of(&[
810		StackReq::new::<usize>(n).array(4),
811		temp_mat_scratch::<T>(n, 1).array(6),
812		temp_mat_scratch::<T>(n, n).array(2),
813	])
814}
