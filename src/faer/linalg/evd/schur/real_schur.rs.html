9fn lahqr_eig22<T: RealField>(mut a00: T, mut a01: T, mut a10: T, mut a11: T) -> ((T, T), (T, T)) {
10	let half = from_f64::<T>(0.5);
11
12	let s = abs(a00) + abs(a01) + abs(a10) + abs(a11);
13	if s == zero() {
14		return ((zero(), zero()), (zero(), zero()));
15	}
16
17	a00 = a00 / s;
18	a01 = a01 / s;
19	a10 = a10 / s;
20	a11 = a11 / s;
21
22	let tr = (a00 + a11) * half;
23	let det = abs2(a00 - tr) + a01 * a10;
24
25	if det >= zero() {
26		let rtdisc = sqrt(det);
27		((s * (tr + rtdisc), zero()), (s * (tr - rtdisc), zero()))
28	} else {
29		let rtdisc = sqrt(-det);
30		let re = s * tr;
31		let im = s * rtdisc;
32		((copy(re), copy(im)), (re, -im))
33	}
34}

36fn lasy2<T: RealField>(tl: MatRef<'_, T>, tr: MatRef<'_, T>, b: MatRef<'_, T>, x: MatMut<'_, T>) -> T {
37	let mut x = x;
38	let mut info = 0;
39
40	assert!(all(tl.nrows() == 2, tr.nrows() == 2, tl.ncols() == 2, tr.ncols() == 2,));
41
42	let eps = eps::<T>();
43	let smlnum = min_positive::<T>() / eps;
44
45	stack_mat!(btmp, 4, 1, T);
46	stack_mat!(tmp, 4, 1, T);
47
48	stack_mat!(t16, 4, 4, T);
49
50	let mut jpiv = [0usize; 4];
51
52	let mut smin = max(max(abs1(tr[(0, 0)]), abs1(tr[(0, 1)])), max(abs1(tr[(1, 0)]), abs1(tr[(1, 1)])));
53	smin = max(
54		smin,
55		max(max(abs1(tl[(0, 0)]), abs1(tl[(0, 1)])), max(abs1(tl[(1, 0)]), abs1(tl[(1, 1)]))),
56	);
57	smin = max(eps * smin, smlnum);
58
59	t16[(0, 0)] = tl[(0, 0)] - tr[(0, 0)];
60	t16[(1, 1)] = tl[(1, 1)] - tr[(0, 0)];
61	t16[(2, 2)] = tl[(0, 0)] - tr[(1, 1)];
62	t16[(3, 3)] = tl[(1, 1)] - tr[(1, 1)];
63
64	t16[(0, 1)] = copy(tl[(0, 1)]);
65	t16[(1, 0)] = copy(tl[(1, 0)]);
66	t16[(2, 3)] = copy(tl[(0, 1)]);
67	t16[(3, 2)] = copy(tl[(1, 0)]);
68
69	t16[(0, 2)] = -tr[(1, 0)];
70	t16[(1, 3)] = -tr[(1, 0)];
71	t16[(2, 0)] = -tr[(0, 1)];
72	t16[(3, 1)] = -tr[(0, 1)];
73
74	btmp[(0, 0)] = copy(b[(0, 0)]);
75	btmp[(1, 0)] = copy(b[(1, 0)]);
76	btmp[(2, 0)] = copy(b[(0, 1)]);
77	btmp[(3, 0)] = copy(b[(1, 1)]);
78
79	let (mut ipsv, mut jpsv);
80	#[allow(clippy::needless_range_loop)]
81	for i in 0..3usize {
82		ipsv = i;
83		jpsv = i;
84		// Do pivoting to get largest pivot element
85		let mut xmax = zero();
86		for ip in i..4 {
87			for jp in i..4 {
88				if abs1(t16[(ip, jp)]) >= xmax {
89					xmax = abs1(t16[(ip, jp)]);
90					ipsv = ip;
91					jpsv = jp;
92				}
93			}
94		}
95		if ipsv != i {
96			crate::perm::swap_rows_idx(t16.rb_mut(), ipsv, i);
97
98			let temp = copy(btmp[(i, 0)]);
99			btmp[(i, 0)] = copy(btmp[(ipsv, 0)]);
100			btmp[(ipsv, 0)] = temp;
101		}
102		if jpsv != i {
103			crate::perm::swap_cols_idx(t16.rb_mut(), jpsv, i);
104		}
105		jpiv[i] = jpsv;
106		if abs1(t16[(i, i)]) < smin {
107			info = 1;
108			t16[(i, i)] = copy(smin);
109		}
110		for j in i + 1..4 {
111			t16[(j, i)] = t16[(j, i)] / t16[(i, i)];
112			btmp[(j, 0)] = btmp[(j, 0)] - t16[(j, i)] * btmp[(i, 0)];
113			for k in i + 1..4 {
114				t16[(j, k)] = t16[(j, k)] - t16[(j, i)] * t16[(i, k)];
115			}
116		}
117	}
118
119	if abs1(t16[(3, 3)]) < smin {
120		info = 1;
121		t16[(3, 3)] = copy(smin);
122	}
123	let mut scale = one::<T>();
124	let eight = from_f64::<T>(8.0);
125
126	if (eight * smlnum) * abs1(btmp[(0, 0)]) > abs1(t16[(0, 0)])
127		|| (eight * smlnum) * abs1(btmp[(1, 0)]) > abs1(t16[(1, 1)])
128		|| (eight * smlnum) * abs1(btmp[(2, 0)]) > abs1(t16[(2, 2)])
129		|| (eight * smlnum) * abs1(btmp[(3, 0)]) > abs1(t16[(3, 3)])
130	{
131		scale = from_f64::<T>(0.125) / max(max(abs1(btmp[(0, 0)]), abs1(btmp[(1, 0)])), max(abs1(btmp[(2, 0)]), abs1(btmp[(3, 0)])));
132		btmp[(0, 0)] = btmp[(0, 0)] * scale;
133		btmp[(1, 0)] = btmp[(1, 0)] * scale;
134		btmp[(2, 0)] = btmp[(2, 0)] * scale;
135		btmp[(3, 0)] = btmp[(3, 0)] * scale;
136	}
137
138	for i in 0..4usize {
139		let k = 3 - i;
140		let temp = recip(t16[(k, k)]);
141		tmp[(k, 0)] = btmp[(k, 0)] * temp;
142		for j in k + 1..4 {
143			tmp[(k, 0)] = tmp[(k, 0)] - temp * t16[(k, j)] * tmp[(j, 0)];
144		}
145	}
146	for i in 0..3usize {
147		if jpiv[2 - i] != 2 - i {
148			let temp = copy(tmp[(2 - i, 0)]);
149			tmp[(2 - i, 0)] = copy(tmp[(jpiv[2 - i], 0)]);
150			tmp[(jpiv[2 - i], 0)] = temp;
151		}
152	}
153	x[(0, 0)] = copy(tmp[(0, 0)]);
154	x[(1, 0)] = copy(tmp[(1, 0)]);
155	x[(0, 1)] = copy(tmp[(2, 0)]);
156	x[(1, 1)] = copy(tmp[(3, 0)]);
157
158	let _ = info;
159	scale
160}

164fn lahqr_schur22<T: RealField>(mut a: T, mut b: T, mut c: T, mut d: T) -> ((T, T, T, T), (T, T), (T, T), (T, T)) {
165	let half = from_f64::<T>(0.5);
166	let one = one::<T>();
167	let multpl = from_f64::<T>(4.0);
168	let eps = eps::<T>();
169	let safmin = min_positive::<T>();
170	let safmn2 = sqrt((safmin / eps));
171	let safmx2 = recip(safmn2);
172	let mut cs;
173	let mut sn;
174	if c == zero() {
175		cs = one;
176		sn = zero();
177	} else if b == zero() {
178		cs = zero();
179		sn = one;
180		core::mem::swap(&mut d, &mut a);
181		b = -c;
182		c = zero();
183	} else if (a - d == zero()) && (b > zero()) != (c > zero()) {
184		cs = one;
185		sn = zero();
186	} else {
187		let mut temp = a - d;
188		let mut p = temp * half;
189		let bcmax = max(abs(b), abs(c));
190		let mut bcmin = min(abs(b), abs(c));
191		if (b > zero()) != (c > zero()) {
192			bcmin = -bcmin;
193		}
194		let mut scale = max(abs(p), bcmax);
195		let mut z = ((p / scale) * p) + ((bcmax / scale) * bcmin);
196		if z >= (multpl * eps) {
197			let mut __tmp = sqrt(scale) * sqrt(z);
198			if p < zero() {
199				__tmp = -__tmp;
200			}
201			z = p + __tmp;
202			a = d + z;
203			d = d - ((bcmax / z) * bcmin);
204			let tau = sqrt(abs2(c) + abs2(z));
205			cs = z / tau;
206			sn = c / tau;
207			b = b - c;
208			c = zero();
209		} else {
210			let mut sigma = b + c;
211			for _ in 0..20 {
212				scale = max(abs(temp), abs(sigma));
213				if scale >= safmx2 {
214					sigma = sigma * safmn2;
215					temp = temp * safmn2;
216					continue;
217				}
218				if scale <= safmn2 {
219					sigma = sigma * safmx2;
220					temp = temp * safmx2;
221					continue;
222				}
223				break;
224			}
225			p = mul_pow2(temp, half);
226			let mut tau = sqrt(abs2(sigma) + abs2(temp));
227			cs = sqrt(mul_pow2(one + abs(sigma) / tau, half));
228			sn = -p / (tau * cs);
229			if sigma < zero() {
230				sn = -sn;
231			}
232			let aa = a * cs + b * sn;
233			let bb = -a * sn + b * cs;
234			let cc = c * cs + d * sn;
235			let dd = -c * sn + d * cs;
236			a = aa * cs + cc * sn;
237			b = bb * cs + dd * sn;
238			c = -aa * sn + cc * cs;
239			d = -bb * sn + dd * cs;
240			temp = mul_pow2((a + d), half);
241			a = copy(temp);
242			d = copy(temp);
243			if c != zero() && b != zero() && (b > zero()) == (c > zero()) {
244				let sab = sqrt(abs(b));
245				let sac = sqrt(abs(c));
246				p = if c > zero() { sab * sac } else { -sab * sac };
247				tau = recip(sqrt(abs(b + c)));
248				a = temp + p;
249				d = temp - p;
250				b = b - c;
251				c = zero();
252				let cs1 = sab * tau;
253				let sn1 = sac * tau;
254				temp = cs * cs1 - sn * sn1;
255				sn = cs * sn1 + sn * cs1;
256				cs = temp;
257			}
258		}
259	}
260	let (s1, s2) = if c != zero() {
261		let temp = sqrt(abs(b)) * sqrt(abs(c));
262		((copy(a), copy(temp)), (copy(d), (-temp)))
263	} else {
264		((copy(a), zero()), (copy(d), zero()))
265	};
266	((a, b, c, d), s1, s2, (cs, sn))
267}

269fn lahqr_shiftcolumn<T: RealField>(h: MatRef<'_, T>, mut v: ColMut<'_, T>, s1: (T, T), s2: (T, T)) {
270	debug_assert!(h.nrows() == h.ncols());
271	let n = h.nrows();
272	debug_assert!(v.nrows() == n);
273	if n == 2 {
274		let s = abs(h[(0, 0)] - s2.0) + abs(s2.1) + abs(h[(1, 0)]);
275		if s == zero() {
276			v[0] = zero();
277			v[1] = zero();
278		} else {
279			let h10s = h[(1, 0)] / s;
280			let v0 = h10s * h[(0, 1)] + (h[(0, 0)] - s1.0) * ((h[(0, 0)] - s2.0) / s) - s1.1 * (s2.1 / s);
281			let v1 = h10s * (h[(0, 0)] + h[(1, 1)] - s1.0 - s2.0);
282			v[0] = v0;
283			v[1] = v1;
284		}
285	} else {
286		let s = abs(h[(0, 0)] - s2.0) + abs(s2.1) + abs(h[(1, 0)]) + abs(h[(2, 0)]);
287		if s == zero() {
288			v[0] = zero();
289			v[1] = zero();
290			v[2] = zero();
291		} else {
292			let h10s = h[(1, 0)] / s;
293			let h20s = h[(2, 0)] / s;
294			let v0 = (h[(0, 0)] - s1.0) * ((h[(0, 0)] - s2.0) / s) - s1.1 * (s2.1 / s) + h[(0, 1)] * h10s + h[(0, 2)] * h20s;
295			let v1 = h10s * ((h[(0, 0)] + h[(1, 1)]) - s1.0 - s2.0) + h[(1, 2)] * h20s;
296			let v2 = h20s * ((h[(0, 0)] + h[(2, 2)]) - s1.0 - s2.0) + h[(2, 1)] * h10s;
297			v[0] = v0;
298			v[1] = v1;
299			v[2] = v2;
300		}
301	}
302}

305fn schur_move<T: RealField>(mut a: MatMut<T>, mut q: Option<MatMut<T>>, mut ifst: usize, ilst: &mut usize) -> isize {
306	let n = a.nrows();
307	if n == 0 {
308		return 0;
309	}
310	if ifst > 0 && (a[(ifst, ifst - 1)] != zero()) {
311		ifst -= 1;
312	}
313	let mut nbf = 1;
314	if ifst < n - 1 && (a[(ifst + 1, ifst)] != zero()) {
315		nbf = 2;
316	}
317	if *ilst > 0 && (a[(*ilst, *ilst - 1)] != zero()) {
318		*ilst -= 1;
319	}
320	let mut nbl = 1;
321	if (*ilst < n - 1) && (a[(*ilst + 1, *ilst)] != zero()) {
322		nbl = 2;
323	}
324	let mut here = ifst;
325	if ifst < *ilst {
326		if nbf == 2 && nbl == 1 {
327			*ilst -= 1;
328		}
329		if nbf == 1 && nbl == 2 {
330			*ilst += 1;
331		}
332		while here != *ilst {
333			let mut nbnext = 1;
334			if (here + nbf + 1 < n) && (a[(here + nbf + 1, here + nbf)] != zero()) {
335				nbnext = 2;
336			}
337			let ierr = schur_swap(a.rb_mut(), q.rb_mut(), here, nbf, nbnext);
338			if ierr != 0 {
339				*ilst = here;
340				return 1;
341			}
342			here += nbnext;
343		}
344	} else {
345		while here != *ilst {
346			let mut nbnext = 1;
347			if here > 1 && (a[(here - 1, here - 2)] != zero()) {
348				nbnext = 2;
349			}
350			let ierr = schur_swap(a.rb_mut(), q.rb_mut(), here - nbnext, nbnext, nbf);
351			if ierr != 0 {
352				*ilst = here;
353				return 1;
354			}
355			here -= nbnext;
356		}
357	}
358	0
359}

362pub(crate) fn schur_swap<T: RealField>(mut a: MatMut<T>, mut q: Option<MatMut<T>>, j0: usize, n1: usize, n2: usize) -> isize {
363	let n = a.nrows();
364	let epsilon = eps::<T>();
365	let zero_threshold = min_positive::<T>();
366	let j1 = j0 + 1;
367	let j2 = j0 + 2;
368	let j3 = j0 + 3;
369	if n1 == 2 && (a[(j1, j0)] == zero()) {
370		schur_swap(a.rb_mut(), q.rb_mut(), j1, 1, n2);
371		schur_swap(a.rb_mut(), q.rb_mut(), j0, 1, n2);
372		return 0;
373	}
374	if n2 == 2 && (a[(j0 + n1 + 1, j0 + n1)] == zero()) {
375		schur_swap(a.rb_mut(), q.rb_mut(), j0, n1, 1);
376		schur_swap(a.rb_mut(), q.rb_mut(), j1, n1, 1);
377		return 0;
378	}
379	if n1 == 1 && n2 == 1 {
380		let t00 = copy(a[(j0, j0)]);
381		let t11 = copy(a[(j1, j1)]);
382		let temp = copy(a[(j0, j1)]);
383		let temp2 = t11 - t00;
384		let (rot, _) = JacobiRotation::rotg(temp, temp2);
385		a[(j1, j1)] = t00;
386		a[(j0, j0)] = t11;
387		if j2 < n {
388			rot.apply_on_the_right_in_place(a.rb_mut().transpose_mut().get_mut(j2.., ..).two_cols_mut(j0, j1));
389		}
390		if j0 > 0 {
391			rot.apply_on_the_right_in_place(a.rb_mut().get_mut(..j0, ..).two_cols_mut(j0, j1));
392		}
393		if let Some(q) = q.rb_mut() {
394			rot.apply_on_the_right_in_place(q.two_cols_mut(j0, j1));
395		}
396	}
397	if n1 == 1 && n2 == 2 {
398		stack_mat!(b, 3, 2, T);
399		b[(0, 0)] = copy(a[(j0, j1)]);
400		b[(1, 0)] = a[(j1, j1)] - a[(j0, j0)];
401		b[(2, 0)] = copy(a[(j2, j1)]);
402		b[(0, 1)] = copy(a[(j0, j2)]);
403		b[(1, 1)] = copy(a[(j1, j2)]);
404		b[(2, 1)] = a[(j2, j2)] - a[(j0, j0)];
405		let mut v1 = b.rb_mut().col_mut(0);
406		let (head, tail) = v1.rb_mut().split_at_row_mut(1);
407		let HouseholderInfo { tau: tau1, .. } = make_householder_in_place(head.at_mut(0), tail);
408		let tau1 = recip(tau1);
409		let v11 = copy(b[(1, 0)]);
410		let v12 = copy(b[(2, 0)]);
411		let sum = b[(0, 1)] + v11 * b[(1, 1)] + v12 * b[(2, 1)];
412		b[(0, 1)] = b[(0, 1)] - (sum * tau1);
413		b[(1, 1)] = b[(1, 1)] - ((sum * tau1) * v11);
414		b[(2, 1)] = b[(2, 1)] - ((sum * tau1) * v12);
415		let mut v2 = b.rb_mut().col_mut(1).subrows_mut(1, 2);
416		let (head, tail) = v2.rb_mut().split_at_row_mut(1);
417		let HouseholderInfo { tau: tau2, .. } = make_householder_in_place(head.at_mut(0), tail);
418		let tau2 = recip(tau2);
419		let v21 = copy(v2[1]);
420		for j in j0..n {
421			let sum = a[(j0, j)] + v11 * a[(j1, j)] + v12 * a[(j2, j)];
422			a[(j0, j)] = a[(j0, j)] - sum * tau1;
423			a[(j1, j)] = a[(j1, j)] - sum * tau1 * v11;
424			a[(j2, j)] = a[(j2, j)] - sum * tau1 * v12;
425			let sum = a[(j1, j)] + v21 * a[(j2, j)];
426			a[(j1, j)] = a[(j1, j)] - sum * tau2;
427			a[(j2, j)] = a[(j2, j)] - sum * tau2 * v21;
428		}
429		for j in 0..j3 {
430			let sum = a[(j, j0)] + v11 * a[(j, j1)] + v12 * a[(j, j2)];
431			a[(j, j0)] = a[(j, j0)] - sum * tau1;
432			a[(j, j1)] = a[(j, j1)] - sum * tau1 * v11;
433			a[(j, j2)] = a[(j, j2)] - sum * tau1 * v12;
434			let sum = a[(j, j1)] + v21 * a[(j, j2)];
435			a[(j, j1)] = a[(j, j1)] - sum * tau2;
436			a[(j, j2)] = a[(j, j2)] - sum * tau2 * v21;
437		}
438		if let Some(mut q) = q.rb_mut() {
439			for j in 0..n {
440				let sum = q[(j, j0)] + v11 * q[(j, j1)] + v12 * q[(j, j2)];
441				q[(j, j0)] = q[(j, j0)] - sum * tau1;
442				q[(j, j1)] = q[(j, j1)] - sum * tau1 * v11;
443				q[(j, j2)] = q[(j, j2)] - sum * tau1 * v12;
444				let sum = q[(j, j1)] + v21 * q[(j, j2)];
445				q[(j, j1)] = q[(j, j1)] - sum * tau2;
446				q[(j, j2)] = q[(j, j2)] - sum * tau2 * v21;
447			}
448		}
449		a[(j2, j0)] = zero();
450		a[(j2, j1)] = zero();
451	}
452	if n1 == 2 && n2 == 1 {
453		stack_mat!(b, 3, 2, T);
454		b[(0, 0)] = copy(a[(j1, j2)]);
455		b[(1, 0)] = a[(j1, j1)] - a[(j2, j2)];
456		b[(2, 0)] = copy(a[(j1, j0)]);
457		b[(0, 1)] = copy(a[(j0, j2)]);
458		b[(1, 1)] = copy(a[(j0, j1)]);
459		b[(2, 1)] = a[(j0, j0)] - a[(j2, j2)];
460		let mut v1 = b.rb_mut().col_mut(0);
461		let (head, tail) = v1.rb_mut().split_at_row_mut(1);
462		let HouseholderInfo { tau: tau1, .. } = make_householder_in_place(head.at_mut(0), tail);
463		let tau1 = recip(tau1);
464		let v11 = copy(v1[1]);
465		let v12 = copy(v1[2]);
466		let sum = b[(0, 1)] + v11 * b[(1, 1)] + v12 * b[(2, 1)];
467		b[(0, 1)] = b[(0, 1)] - sum * tau1;
468		b[(1, 1)] = b[(1, 1)] - sum * tau1 * v11;
469		b[(2, 1)] = b[(2, 1)] - sum * tau1 * v12;
470		let mut v2 = b.rb_mut().col_mut(1).subrows_mut(1, 2);
471		let (head, tail) = v2.rb_mut().split_at_row_mut(1);
472		let HouseholderInfo { tau: tau2, .. } = make_householder_in_place(head.at_mut(0), tail);
473		let tau2 = recip(tau2);
474		let v21 = copy(v2[1]);
475		for j in j0..n {
476			let sum = a[(j2, j)] + v11 * a[(j1, j)] + v12 * a[(j0, j)];
477			a[(j2, j)] = a[(j2, j)] - sum * tau1;
478			a[(j1, j)] = a[(j1, j)] - sum * tau1 * v11;
479			a[(j0, j)] = a[(j0, j)] - sum * tau1 * v12;
480			let sum = a[(j1, j)] + v21 * a[(j0, j)];
481			a[(j1, j)] = a[(j1, j)] - sum * tau2;
482			a[(j0, j)] = a[(j0, j)] - sum * tau2 * v21;
483		}
484		for j in 0..j3 {
485			let sum = a[(j, j2)] + v11 * a[(j, j1)] + v12 * a[(j, j0)];
486			a[(j, j2)] = a[(j, j2)] - sum * tau1;
487			a[(j, j1)] = a[(j, j1)] - sum * tau1 * v11;
488			a[(j, j0)] = a[(j, j0)] - sum * tau1 * v12;
489			let sum = a[(j, j1)] + v21 * a[(j, j0)];
490			a[(j, j1)] = a[(j, j1)] - sum * tau2;
491			a[(j, j0)] = a[(j, j0)] - sum * tau2 * v21;
492		}
493		if let Some(mut q) = q.rb_mut() {
494			for j in 0..n {
495				let sum = q[(j, j2)] + v11 * q[(j, j1)] + v12 * q[(j, j0)];
496				q[(j, j2)] = q[(j, j2)] - sum * tau1;
497				q[(j, j1)] = q[(j, j1)] - sum * tau1 * v11;
498				q[(j, j0)] = q[(j, j0)] - sum * tau1 * v12;
499				let sum = q[(j, j1)] + v21 * q[(j, j0)];
500				q[(j, j1)] = q[(j, j1)] - sum * tau2;
501				q[(j, j0)] = q[(j, j0)] - sum * tau2 * v21;
502			}
503		}
504		a[(j1, j0)] = zero();
505		a[(j2, j0)] = zero();
506	}
507	if n1 == 2 && n2 == 2 {
508		stack_mat!(d, 4, 4, T);
509		let ad_slice = a.rb().submatrix(j0, j0, 4, 4);
510		d.copy_from(ad_slice);
511		let mut dnorm = zero();
512		z!(d.rb()).for_each(|unzip!(d)| dnorm = max(dnorm, abs((*d))));
513		let eps = epsilon;
514		let small_num = zero_threshold / eps;
515		let thresh = max(((from_f64::<T>(10.0) * eps) * dnorm), small_num);
516		stack_mat!(v, 4, 2, T);
517		let mut x = v.rb_mut().submatrix_mut(0, 0, 2, 2);
518		let (tl, b, _, tr) = d.rb().split_at(2, 2);
519		let scale = lasy2(tl, tr, b, x.rb_mut());
520		v[(2, 0)] = -scale;
521		v[(2, 1)] = zero();
522		v[(3, 0)] = zero();
523		v[(3, 1)] = -scale;
524		let mut v1 = v.rb_mut().col_mut(0);
525		let (head, tail) = v1.rb_mut().split_at_row_mut(1);
526		let HouseholderInfo { tau: tau1, .. } = make_householder_in_place(head.at_mut(0), tail);
527		let tau1 = recip(tau1);
528		let v11 = copy(v1[1]);
529		let v12 = copy(v1[2]);
530		let v13 = copy(v1[3]);
531		let sum = v[(0, 1)] + v11 * v[(1, 1)] + v12 * v[(2, 1)] + v13 * v[(3, 1)];
532		v[(0, 1)] = v[(0, 1)] - sum * tau1;
533		v[(1, 1)] = v[(1, 1)] - sum * tau1 * v11;
534		v[(2, 1)] = v[(2, 1)] - sum * tau1 * v12;
535		v[(3, 1)] = v[(3, 1)] - sum * tau1 * v13;
536		let mut v2 = v.rb_mut().col_mut(1).subrows_mut(1, 3);
537		let (head, tail) = v2.rb_mut().split_at_row_mut(1);
538		let HouseholderInfo { tau: tau2, .. } = make_householder_in_place(head.at_mut(0), tail);
539		let tau2 = recip(tau2);
540		let v21 = copy(v2[1]);
541		let v22 = copy(v2[2]);
542		for j in 0..4 {
543			let sum = d[(0, j)] + v11 * d[(1, j)] + v12 * d[(2, j)] + v13 * d[(3, j)];
544			d[(0, j)] = d[(0, j)] - sum * tau1;
545			d[(1, j)] = d[(1, j)] - sum * tau1 * v11;
546			d[(2, j)] = d[(2, j)] - sum * tau1 * v12;
547			d[(3, j)] = d[(3, j)] - sum * tau1 * v13;
548			let sum = d[(1, j)] + v21 * d[(2, j)] + v22 * d[(3, j)];
549			d[(1, j)] = d[(1, j)] - sum * tau2;
550			d[(2, j)] = d[(2, j)] - sum * tau2 * v21;
551			d[(3, j)] = d[(3, j)] - sum * tau2 * v22;
552		}
553		for j in 0..4 {
554			let sum = d[(j, 0)] + v11 * d[(j, 1)] + v12 * d[(j, 2)] + v13 * d[(j, 3)];
555			d[(j, 0)] = d[(j, 0)] - sum * tau1;
556			d[(j, 1)] = d[(j, 1)] - sum * tau1 * v11;
557			d[(j, 2)] = d[(j, 2)] - sum * tau1 * v12;
558			d[(j, 3)] = d[(j, 3)] - sum * tau1 * v13;
559			let sum = d[(j, 1)] + v21 * d[(j, 2)] + v22 * d[(j, 3)];
560			d[(j, 1)] = d[(j, 1)] - sum * tau2;
561			d[(j, 2)] = d[(j, 2)] - sum * tau2 * v21;
562			d[(j, 3)] = d[(j, 3)] - sum * tau2 * v22;
563		}
564		if max(max(abs(d[(2, 0)]), abs(d[(2, 1)])), max(abs(d[(3, 0)]), abs(d[(3, 1)]))) > thresh {
565			return 1;
566		}
567		for j in j0..n {
568			let sum = a[(j0, j)] + v11 * a[(j1, j)] + v12 * a[(j2, j)] + v13 * a[(j3, j)];
569			a[(j0, j)] = a[(j0, j)] - sum * tau1;
570			a[(j1, j)] = a[(j1, j)] - sum * tau1 * v11;
571			a[(j2, j)] = a[(j2, j)] - sum * tau1 * v12;
572			a[(j3, j)] = a[(j3, j)] - sum * tau1 * v13;
573			let sum = a[(j1, j)] + v21 * a[(j2, j)] + v22 * a[(j3, j)];
574			a[(j1, j)] = a[(j1, j)] - sum * tau2;
575			a[(j2, j)] = a[(j2, j)] - sum * tau2 * v21;
576			a[(j3, j)] = a[(j3, j)] - sum * tau2 * v22;
577		}
578		for j in 0..j0 + 4 {
579			let sum = a[(j, j0)] + v11 * a[(j, j1)] + v12 * a[(j, j2)] + v13 * a[(j, j3)];
580			a[(j, j0)] = a[(j, j0)] - sum * tau1;
581			a[(j, j1)] = a[(j, j1)] - sum * tau1 * v11;
582			a[(j, j2)] = a[(j, j2)] - sum * tau1 * v12;
583			a[(j, j3)] = a[(j, j3)] - sum * tau1 * v13;
584			let sum = a[(j, j1)] + v21 * a[(j, j2)] + v22 * a[(j, j3)];
585			a[(j, j1)] = a[(j, j1)] - sum * tau2;
586			a[(j, j2)] = a[(j, j2)] - sum * tau2 * v21;
587			a[(j, j3)] = a[(j, j3)] - sum * tau2 * v22;
588		}
589		if let Some(mut q) = q.rb_mut() {
590			for j in 0..n {
591				let sum = q[(j, j0)] + v11 * q[(j, j1)] + v12 * q[(j, j2)] + v13 * q[(j, j3)];
592				q[(j, j0)] = q[(j, j0)] - sum * tau1;
593				q[(j, j1)] = q[(j, j1)] - sum * tau1 * v11;
594				q[(j, j2)] = q[(j, j2)] - sum * tau1 * v12;
595				q[(j, j3)] = q[(j, j3)] - sum * tau1 * v13;
596				let sum = q[(j, j1)] + v21 * q[(j, j2)] + v22 * q[(j, j3)];
597				q[(j, j1)] = q[(j, j1)] - sum * tau2;
598				q[(j, j2)] = q[(j, j2)] - sum * tau2 * v21;
599				q[(j, j3)] = q[(j, j3)] - sum * tau2 * v22;
600			}
601		}
602		a[(j2, j0)] = zero();
603		a[(j2, j1)] = zero();
604		a[(j3, j0)] = zero();
605		a[(j3, j1)] = zero();
606	}
607	if n2 == 2 {
608		let ((a00, a01, a10, a11), _, _, (cs, sn)) = lahqr_schur22(copy(a[(j0, j0)]), copy(a[(j0, j1)]), copy(a[(j1, j0)]), copy(a[(j1, j1)]));
609		let rot = JacobiRotation { c: cs, s: sn };
610		a[(j0, j0)] = a00;
611		a[(j0, j1)] = a01;
612		a[(j1, j0)] = a10;
613		a[(j1, j1)] = a11;
614		if j2 < n {
615			rot.apply_on_the_right_in_place(a.rb_mut().transpose_mut().get_mut(j2.., ..).two_cols_mut(j0, j1));
616		}
617		if j0 > 0 {
618			rot.apply_on_the_right_in_place(a.rb_mut().get_mut(..j0, ..).two_cols_mut(j0, j1));
619		}
620		if let Some(q) = q.rb_mut() {
621			rot.apply_on_the_right_in_place(q.two_cols_mut(j0, j1));
622		}
623	}
624	if n1 == 2 {
625		let j0 = j0 + n2;
626		let j1 = j1 + n2;
627		let j2 = j2 + n2;
628		let ((a00, a01, a10, a11), _, _, (cs, sn)) = lahqr_schur22(copy(a[(j0, j0)]), copy(a[(j0, j1)]), copy(a[(j1, j0)]), copy(a[(j1, j1)]));
629		let rot = JacobiRotation { c: cs, s: sn };
630		a[(j0, j0)] = a00;
631		a[(j0, j1)] = a01;
632		a[(j1, j0)] = a10;
633		a[(j1, j1)] = a11;
634		if j2 < n {
635			rot.apply_on_the_right_in_place(a.rb_mut().transpose_mut().get_mut(j2.., ..).two_cols_mut(j0, j1));
636		}
637		if j0 > 0 {
638			rot.apply_on_the_right_in_place(a.rb_mut().get_mut(..j0, ..).two_cols_mut(j0, j1));
639		}
640		if let Some(q) = q.rb_mut() {
641			rot.apply_on_the_right_in_place(q.two_cols_mut(j0, j1));
642		}
643	}
644	0
645}

647#[math]
648fn aggressive_early_deflation<T: RealField>(
649	want_t: bool,
650	mut a: MatMut<'_, T>,
651	mut z: Option<MatMut<'_, T>>,
652	mut s_re: ColMut<'_, T>,
653	mut s_im: ColMut<'_, T>,
654	ilo: usize,
655	ihi: usize,
656	nw: usize,
657	par: Par,
658	mut stack: &mut MemStack,
659	params: SchurParams,
660) -> (usize, usize) {
661	let n = a.nrows();
662	let epsilon = eps::<T>();
663	let zero_threshold = min_positive::<T>();
664	let nw_max = (n - 3) / 3;
665	let eps = epsilon;
666	let small_num = zero_threshold / eps * from_f64::<T>(n as f64);
667	let jw = Ord::min(Ord::min(nw, ihi - ilo), nw_max);
668	let kwtop = ihi - jw;
669	let mut s_spike = if kwtop == ilo { zero() } else { copy(a[(kwtop, kwtop - 1)]) };
670	if kwtop + 1 == ihi {
671		s_re[kwtop] = copy(a[(kwtop, kwtop)]);
672		s_im[kwtop] = zero();
673		let mut ns = 1;
674		let mut nd = 0;
675		if abs(s_spike) <= max(small_num, (eps * abs(a[(kwtop, kwtop)]))) {
676			ns = 0;
677			nd = 1;
678			if kwtop > ilo {
679				a[(kwtop, kwtop - 1)] = zero();
680			}
681		}
682		return (ns, nd);
683	}
684	let mut v = unsafe { a.rb().submatrix(n - jw, 0, jw, jw).const_cast() };
685	let mut tw = unsafe { a.rb().submatrix(n - jw, jw, jw, jw).const_cast() };
686	let mut wh = unsafe { a.rb().submatrix(n - jw, jw, jw, n - 2 * jw - 3).const_cast() };
687	let mut wv = unsafe { a.rb().submatrix(jw + 3, 0, n - 2 * jw - 3, jw).const_cast() };
688	let mut a = unsafe { a.rb().const_cast() };
689	let a_window = a.rb().submatrix(kwtop, kwtop, ihi - kwtop, ihi - kwtop);
690	let mut s_re_window = unsafe { s_re.rb().subrows(kwtop, ihi - kwtop).const_cast() };
691	let mut s_im_window = unsafe { s_im.rb().subrows(kwtop, ihi - kwtop).const_cast() };
692	z!(tw.rb_mut()).for_each_triangular_lower(linalg::zip::Diag::Include, |unzip!(x)| *x = zero());
693	for j in 0..jw {
694		for i in 0..Ord::min(j + 2, jw) {
695			tw[(i, j)] = copy(a_window[(i, j)]);
696		}
697	}
698	v.fill(zero());
699	v.rb_mut().diagonal_mut().fill(one::<T>());
700	let infqr = if true || jw < params.blocking_threshold {
701		lahqr(true, tw.rb_mut(), Some(v.rb_mut()), s_re_window.rb_mut(), s_im_window.rb_mut(), 0, jw)
702	} else {
703		let infqr = multishift_qr(
704			true,
705			tw.rb_mut(),
706			Some(v.rb_mut()),
707			s_re_window.rb_mut(),
708			s_im_window.rb_mut(),
709			0,
710			jw,
711			par,
712			stack.rb_mut(),
713			params,
714		)
715		.0;
716		for j in 0..jw {
717			for i in j + 2..jw {
718				tw[(i, j)] = zero();
719			}
720		}
721		infqr
722	};
723	let infqr = infqr as usize;
724	let mut ns = jw;
725	let nd;
726	let mut ilst = infqr;
727	while ilst < ns {
728		let mut bulge = false;
729		if ns > 1 && (tw[(ns - 1, ns - 2)] != zero()) {
730			bulge = true;
731		}
732		if !bulge {
733			#[allow(clippy::disallowed_names)]
734			let mut foo = abs(tw[(ns - 1, ns - 1)]);
735			if foo == zero() {
736				foo = abs(s_spike);
737			}
738			if abs(s_spike) * abs(v[(0, ns - 1)]) <= max(small_num, (eps * foo)) {
739				ns -= 1;
740			} else {
741				let ifst = ns - 1;
742				schur_move(tw.rb_mut(), Some(v.rb_mut()), ifst, &mut ilst);
743				ilst += 1;
744			}
745		} else {
746			#[allow(clippy::disallowed_names)]
747			let mut foo = abs(tw[(ns - 1, ns - 1)]) + sqrt(abs(tw[(ns - 1, ns - 2)])) * sqrt(abs(tw[(ns - 2, ns - 1)]));
748			if foo == zero() {
749				foo = abs(s_spike);
750			}
751			if max(abs((s_spike * v[(0, ns - 1)])), abs((s_spike * v[(0, ns - 2)]))) <= max(small_num, (eps * foo)) {
752				ns -= 2;
753			} else {
754				let ifst = ns - 2;
755				schur_move(tw.rb_mut(), Some(v.rb_mut()), ifst, &mut ilst);
756				ilst += 2;
757			}
758		}
759	}
760	if ns == 0 {
761		s_spike = zero();
762	}
763	if ns == jw {
764		nd = jw - ns;
765		ns -= infqr;
766		return (ns, nd);
767	}
768	let mut sorted = false;
769	let mut sorting_window_size = jw as isize;
770	while !sorted {
771		sorted = true;
772		let mut ilst = 0isize;
773		let mut i1 = ns;
774		while i1 as isize + 1 < sorting_window_size {
775			let mut n1 = 1;
776			if tw[(i1 + 1, i1)] != zero() {
777				n1 = 2;
778			}
779			if i1 + n1 == jw {
780				ilst -= n1 as isize;
781				break;
782			}
783			let i2 = i1 + n1;
784			let mut n2 = 1;
785			if i2 + 1 < jw && (tw[(i2 + 1, i2)] != zero()) {
786				n2 = 2;
787			}
788			let (ev1, ev2);
789			if n1 == 1 {
790				ev1 = abs(tw[(i1, i1)]);
791			} else {
792				ev1 = abs(tw[(i1, i1)]) + sqrt(abs(tw[(i1 + 1, i1)])) * sqrt(abs(tw[(i1, i1 + 1)]));
793			}
794			if n2 == 1 {
795				ev2 = abs(tw[(i2, i2)]);
796			} else {
797				ev2 = abs(tw[(i2, i2)]) + sqrt(abs(tw[(i2 + 1, i2)])) * sqrt(abs(tw[(i2, i2 + 1)]));
798			}
799			if ev1 >= ev2 {
800				i1 = i2;
801			} else {
802				sorted = false;
803				let ierr = schur_swap(tw.rb_mut(), Some(v.rb_mut()), i1, n1, n2);
804				if ierr == 0 {
805					i1 += n2;
806				} else {
807					i1 = i2;
808				}
809				ilst = i1 as isize;
810			}
811		}
812		sorting_window_size = ilst;
813	}
814	let mut i = 0;
815	while i < jw {
816		let mut n1 = 1;
817		if i + 1 < jw && (tw[(i + 1, i)] != zero()) {
818			n1 = 2;
819		}
820		if n1 == 1 {
821			s_re[kwtop + i] = copy(tw[(i, i)]);
822			s_im[kwtop + i] = zero();
823		} else {
824			let ((s1_re, s1_im), (s2_re, s2_im)) =
825				lahqr_eig22(copy(tw[(i, i)]), copy(tw[(i, i + 1)]), copy(tw[(i + 1, i)]), copy(tw[(i + 1, i + 1)]));
826			s_re[kwtop + i] = s1_re;
827			s_im[kwtop + i] = s1_im;
828			s_re[kwtop + i + 1] = s2_re;
829			s_im[kwtop + i + 1] = s2_im;
830		}
831		i += n1;
832	}
833	if s_spike != zero() {
834		{
835			let mut vv = wv.rb_mut().col_mut(0).subrows_mut(0, ns);
836			for i in 0..ns {
837				vv[i] = conj(v[(0, i)]);
838			}
839			let mut head = copy(vv[0]);
840			let tail = vv.rb_mut().subrows_mut(1, ns - 1);
841			let HouseholderInfo { tau, .. } = make_householder_in_place(&mut head, tail);
842			let beta = copy(head);
843			vv[0] = one::<T>();
844			let tau = recip(tau);
845			{
846				let mut tw_slice = tw.rb_mut().submatrix_mut(0, 0, ns, jw);
847				let (mut tmp, _) = unsafe { temp_mat_uninit(jw, 1, stack) };
848				let mut tmp = tmp.as_mat_mut().transpose_mut();
849				matmul(tmp.rb_mut(), Accum::Replace, vv.rb().adjoint().as_mat(), tw_slice.rb(), one::<T>(), par);
850				matmul(tw_slice.rb_mut(), Accum::Add, vv.rb().as_mat(), tmp.as_ref(), -tau, par);
851			}
852			{
853				let mut tw_slice2 = tw.rb_mut().submatrix_mut(0, 0, jw, ns);
854				let (mut tmp, _) = unsafe { temp_mat_uninit(jw, 1, stack) };
855				let mut tmp = tmp.as_mat_mut();
856				matmul(tmp.rb_mut(), Accum::Replace, tw_slice2.rb(), vv.rb().as_mat(), one::<T>(), par);
857				matmul(tw_slice2.rb_mut(), Accum::Add, tmp.as_ref(), vv.rb().adjoint().as_mat(), -tau, par);
858			}
859			{
860				let mut v_slice = v.rb_mut().submatrix_mut(0, 0, jw, ns);
861				let (mut tmp, _) = unsafe { temp_mat_uninit(jw, 1, stack) };
862				let mut tmp = tmp.as_mat_mut();
863				matmul(tmp.rb_mut(), Accum::Replace, v_slice.rb(), vv.rb().as_mat(), one::<T>(), par);
864				matmul(v_slice.rb_mut(), Accum::Add, tmp.as_ref(), vv.rb().adjoint().as_mat(), -tau, par);
865			}
866			vv[0] = beta;
867		}
868		{
869			let mut householder = wv.rb_mut().col_mut(0).subrows_mut(0, ns - 1);
870			hessenberg::hessenberg_in_place(
871				tw.rb_mut().submatrix_mut(0, 0, ns, ns),
872				householder.rb_mut().as_mat_mut().transpose_mut(),
873				par,
874				stack.rb_mut(),
875				Default::default(),
876			);
877			apply_block_householder_sequence_transpose_on_the_left_in_place_with_conj(
878				tw.rb().submatrix(1, 0, ns - 1, ns - 1),
879				householder.rb().transpose().as_mat(),
880				Conj::Yes,
881				unsafe { tw.rb().submatrix(1, ns, ns - 1, jw - ns).const_cast() },
882				par,
883				stack.rb_mut(),
884			);
885			apply_block_householder_sequence_on_the_right_in_place_with_conj(
886				tw.rb().submatrix(1, 0, ns - 1, ns - 1),
887				householder.rb().transpose().as_mat(),
888				Conj::No,
889				v.rb_mut().submatrix_mut(0, 1, jw, ns - 1),
890				par,
891				stack.rb_mut(),
892			);
893		}
894	}
895	if kwtop > 0 {
896		a[(kwtop, kwtop - 1)] = s_spike * conj(v[(0, 0)]);
897	}
898	for j in 0..jw {
899		for i in 0..Ord::min(j + 2, jw) {
900			a[(kwtop + i, kwtop + j)] = copy(tw[(i, j)]);
901		}
902	}
903	nd = jw - ns;
904	ns -= infqr;
905	let (istart_m, istop_m);
906	if want_t {
907		istart_m = 0;
908		istop_m = n;
909	} else {
910		istart_m = ilo;
911		istop_m = ihi;
912	}
913	if ihi < istop_m {
914		let mut i = ihi;
915		while i < istop_m {
916			let iblock = Ord::min(istop_m - i, wh.ncols());
917			let mut a_slice = a.rb_mut().submatrix_mut(kwtop, i, ihi - kwtop, iblock);
918			let mut wh_slice = wh.rb_mut().submatrix_mut(0, 0, a_slice.nrows(), a_slice.ncols());
919			matmul(wh_slice.rb_mut(), Accum::Replace, v.rb().adjoint(), a_slice.rb(), one::<T>(), par);
920			a_slice.copy_from(wh_slice.rb());
921			i += iblock;
922		}
923	}
924	if istart_m < kwtop {
925		let mut i = istart_m;
926		while i < kwtop {
927			let iblock = Ord::min(kwtop - i, wv.nrows());
928			let mut a_slice = a.rb_mut().submatrix_mut(i, kwtop, iblock, ihi - kwtop);
929			let mut wv_slice = wv.rb_mut().submatrix_mut(0, 0, a_slice.nrows(), a_slice.ncols());
930			matmul(wv_slice.rb_mut(), Accum::Replace, a_slice.rb(), v.rb(), one::<T>(), par);
931			a_slice.copy_from(wv_slice.rb());
932			i += iblock;
933		}
934	}
935	if let Some(mut z) = z.rb_mut() {
936		let mut i = 0;
937		while i < n {
938			let iblock = Ord::min(n - i, wv.nrows());
939			let mut z_slice = z.rb_mut().submatrix_mut(i, kwtop, iblock, ihi - kwtop);
940			let mut wv_slice = wv.rb_mut().submatrix_mut(0, 0, z_slice.nrows(), z_slice.ncols());
941			matmul(wv_slice.rb_mut(), Accum::Replace, z_slice.rb(), v.rb(), one::<T>(), par);
942			z_slice.copy_from(wv_slice.rb());
943			i += iblock;
944		}
945	}
946	(ns, nd)
947}
948#[math]
949fn move_bulge<T: RealField>(mut h: MatMut<'_, T>, mut v: ColMut<'_, T>, s1: (T, T), s2: (T, T)) {
950	let epsilon = eps::<T>();
951	let v0 = real(v[0]);
952	let v1 = copy(v[1]);
953	let v2 = copy(v[2]);
954	let refsum = mul_real(v2, v0) * h[(3, 2)];
955	h[(3, 0)] = -refsum;
956	h[(3, 1)] = -refsum * conj(v1);
957	h[(3, 2)] = h[(3, 2)] - refsum * conj(v2);
958	v[0] = copy(h[(1, 0)]);
959	v[1] = copy(h[(2, 0)]);
960	v[2] = copy(h[(3, 0)]);
961	let mut beta = copy(v[0]);
962	let tail = v.rb_mut().subrows_mut(1, 2);
963	let HouseholderInfo { tau, .. } = make_householder_in_place(&mut beta, tail);
964	v[0] = recip(tau);
965	if h[(3, 0)] != zero() || h[(3, 1)] != zero() || h[(3, 2)] != zero() {
966		h[(1, 0)] = beta;
967		h[(2, 0)] = zero();
968		h[(3, 0)] = zero();
969	} else {
970		stack_mat!(vt, 3, 1, T);
971		let mut vt = vt.rb_mut().col_mut(0);
972		let h2 = h.rb().submatrix(1, 1, 3, 3);
973		lahqr_shiftcolumn(h2, vt.rb_mut(), s1, s2);
974		let mut beta_unused = copy(vt[0]);
975		let tail = vt.rb_mut().subrows_mut(1, 2);
976		let HouseholderInfo { tau, .. } = make_householder_in_place(&mut beta_unused, tail);
977		vt[0] = recip(tau);
978		let vt0 = copy(vt[0]);
979		let vt1 = copy(vt[1]);
980		let vt2 = copy(vt[2]);
981		let refsum = conj(vt0) * h[(1, 0)] + conj(vt1) * h[(2, 0)];
982		if (abs1((h[(2, 0)] - (refsum * vt1))) + abs1((refsum * vt2))) > (epsilon * ((abs1(h[(0, 0)]) + abs1(h[(1, 1)])) + abs1(h[(2, 2)]))) {
983			h[(1, 0)] = beta;
984			h[(2, 0)] = zero();
985			h[(3, 0)] = zero();
986		} else {
987			h[(1, 0)] = h[(1, 0)] - refsum;
988			h[(2, 0)] = zero();
989			h[(3, 0)] = zero();
990			v[0] = copy(vt[0]);
991			v[1] = copy(vt[1]);
992			v[2] = copy(vt[2]);
993		}
994	}
995}

1088#[math]
1089fn introduce_bulges<T: RealField>(
1090	ilo: usize,
1091	ihi: usize,
1092	n_block_desired: usize,
1093	n_bulges: usize,
1094	n_shifts: usize,
1095	want_t: bool,
1096	mut a: MatMut<'_, T>,
1097	mut z: Option<MatMut<'_, T>>,
1098	mut u: MatMut<'_, T>,
1099	mut v: MatMut<'_, T>,
1100	mut wh: MatMut<'_, T>,
1101	mut wv: MatMut<'_, T>,
1102	s_re: ColRef<'_, T>,
1103	s_im: ColRef<'_, T>,
1104	i_pos_block: &mut usize,
1105	parallelism: Par,
1106) {
1107	let n = a.nrows();
1108	let eps = eps::<T>();
1109	let small_num = min_positive::<T>() / eps * from_f64::<T>(n as f64);
1110	let n_block = Ord::min(n_block_desired, ihi - ilo);
1111	let mut istart_m = ilo;
1112	let mut istop_m = ilo + n_block;
1113	let mut u2 = u.rb_mut().submatrix_mut(0, 0, n_block, n_block);
1114	u2.fill(zero());
1115	u2.rb_mut().diagonal_mut().fill(one::<T>());
1116	for i_pos_last in ilo..ilo + n_block - 2 {
1117		let n_active_bulges = Ord::min(n_bulges, ((i_pos_last - ilo) / 2) + 1);
1118		for i_bulge in 0..n_active_bulges {
1119			let i_pos = i_pos_last - 2 * i_bulge;
1120			let mut v = v.rb_mut().col_mut(i_bulge);
1121			if i_pos == ilo {
1122				let h = a.rb().submatrix(ilo, ilo, 3, 3);
1123				let s1_re = copy(s_re[s_re.nrows() - 1 - 2 * i_bulge]);
1124				let s1_im = copy(s_im[s_im.nrows() - 1 - 2 * i_bulge]);
1125				let s2_re = copy(s_re[s_re.nrows() - 1 - 2 * i_bulge - 1]);
1126				let s2_im = copy(s_im[s_im.nrows() - 1 - 2 * i_bulge - 1]);
1127				lahqr_shiftcolumn(h, v.rb_mut(), (s1_re, s1_im), (s2_re, s2_im));
1128				debug_assert!(v.nrows() == 3);
1129				let mut head = copy(v[0]);
1130				let tail = v.rb_mut().subrows_mut(1, 2);
1131				let HouseholderInfo { tau, .. } = make_householder_in_place(&mut head, tail);
1132				v[0] = recip(tau);
1133			} else {
1134				let mut h = a.rb_mut().submatrix_mut(i_pos - 1, i_pos - 1, 4, 4);
1135				let s1_re = copy(s_re[s_re.nrows() - 1 - 2 * i_bulge]);
1136				let s1_im = copy(s_im[s_im.nrows() - 1 - 2 * i_bulge]);
1137				let s2_re = copy(s_re[s_re.nrows() - 1 - 2 * i_bulge - 1]);
1138				let s2_im = copy(s_im[s_im.nrows() - 1 - 2 * i_bulge - 1]);
1139				move_bulge(h.rb_mut(), v.rb_mut(), (s1_re, s1_im), (s2_re, s2_im));
1140			}
1141			let v0 = real(v[0]);
1142			let v1 = copy(v[1]);
1143			let v2 = copy(v[2]);
1144			for j in istart_m..i_pos + 3 {
1145				let sum = a[(j, i_pos)] + v1 * a[(j, i_pos + 1)] + v2 * a[(j, i_pos + 2)];
1146				a[(j, i_pos)] = a[(j, i_pos)] - mul_real(sum, v0);
1147				a[(j, i_pos + 1)] = a[(j, i_pos + 1)] - mul_real(sum, v0) * conj(v1);
1148				a[(j, i_pos + 2)] = a[(j, i_pos + 2)] - mul_real(sum, v0) * conj(v2);
1149			}
1150			let sum = a[(i_pos, i_pos)] + conj(v1) * a[(i_pos + 1, i_pos)] + conj(v2) * a[(i_pos + 2, i_pos)];
1151			a[(i_pos, i_pos)] = a[(i_pos, i_pos)] - mul_real(sum, v0);
1152			a[(i_pos + 1, i_pos)] = a[(i_pos + 1, i_pos)] - (mul_real(sum, v0) * v1);
1153			a[(i_pos + 2, i_pos)] = a[(i_pos + 2, i_pos)] - mul_real(sum, v0) * v2;
1154			if i_pos > ilo && (a[(i_pos, i_pos - 1)] != zero()) {
1155				let mut tst1 = abs1(a[(i_pos - 1, i_pos - 1)]) + abs1(a[(i_pos, i_pos)]);
1156				if tst1 == zero() {
1157					if i_pos > ilo + 1 {
1158						tst1 = tst1 + abs1(a[(i_pos - 1, i_pos - 2)]);
1159					}
1160					if i_pos > ilo + 2 {
1161						tst1 = tst1 + abs1(a[(i_pos - 1, i_pos - 3)]);
1162					}
1163					if i_pos > ilo + 3 {
1164						tst1 = tst1 + abs1(a[(i_pos - 1, i_pos - 4)]);
1165					}
1166					if i_pos < ihi - 1 {
1167						tst1 = tst1 + abs1(a[(i_pos + 1, i_pos)]);
1168					}
1169					if i_pos < ihi - 2 {
1170						tst1 = tst1 + abs1(a[(i_pos + 2, i_pos)]);
1171					}
1172					if i_pos < ihi - 3 {
1173						tst1 = tst1 + abs1(a[(i_pos + 3, i_pos)]);
1174					}
1175				}
1176				if abs1(a[(i_pos, i_pos - 1)]) < max(small_num, (eps * tst1)) {
1177					let ab = max(abs1(a[(i_pos, i_pos - 1)]), abs1(a[(i_pos - 1, i_pos)]));
1178					let ba = min(abs1(a[(i_pos, i_pos - 1)]), abs1(a[(i_pos - 1, i_pos)]));
1179					let aa = max(abs1(a[(i_pos, i_pos)]), abs1((a[(i_pos, i_pos)] - a[(i_pos - 1, i_pos - 1)])));
1180					let bb = min(abs1(a[(i_pos, i_pos)]), abs1((a[(i_pos, i_pos)] - a[(i_pos - 1, i_pos - 1)])));
1181					let s = aa + ab;
1182					if (ba * (ab / s)) <= max(small_num, (eps * (bb * (aa / s)))) {
1183						a[(i_pos, i_pos - 1)] = zero();
1184					}
1185				}
1186			}
1187		}
1188		for i_bulge in 0..n_active_bulges {
1189			let i_pos = i_pos_last - 2 * i_bulge;
1190			let v = v.rb_mut().col_mut(i_bulge);
1191			let v0 = real(v[0]);
1192			let v1 = copy(v[1]);
1193			let v2 = copy(v[2]);
1194			for j in i_pos + 1..istop_m {
1195				let sum = a[(i_pos, j)] + conj(v1) * a[(i_pos + 1, j)] + conj(v2) * a[(i_pos + 2, j)];
1196				a[(i_pos, j)] = a[(i_pos, j)] - mul_real(sum, v0);
1197				a[(i_pos + 1, j)] = a[(i_pos + 1, j)] - mul_real(sum, v0) * v1;
1198				a[(i_pos + 2, j)] = a[(i_pos + 2, j)] - mul_real(sum, v0) * v2;
1199			}
1200		}
1201		for i_bulge in 0..n_active_bulges {
1202			let i_pos = i_pos_last - 2 * i_bulge;
1203			let v = v.rb_mut().col_mut(i_bulge);
1204			let v0 = real(v[0]);
1205			let v1 = copy(v[1]);
1206			let v2 = copy(v[2]);
1207			let i1 = 0;
1208			let i2 = Ord::min(u2.nrows(), (i_pos_last - ilo) + (i_pos_last - ilo) + 3);
1209			for j in i1..i2 {
1210				let sum = u2[(j, i_pos - ilo)] + v1 * u2[(j, i_pos - ilo + 1)] + v2 * u2[(j, i_pos - ilo + 2)];
1211				u2[(j, i_pos - ilo)] = u2[(j, i_pos - ilo)] - mul_real(sum, v0);
1212				u2[(j, i_pos - ilo + 1)] = u2[(j, i_pos - ilo + 1)] - mul_real(sum, v0) * conj(v1);
1213				u2[(j, i_pos - ilo + 2)] = u2[(j, i_pos - ilo + 2)] - mul_real(sum, v0) * conj(v2);
1214			}
1215		}
1216	}
1217	if want_t {
1218		istart_m = 0;
1219		istop_m = n;
1220	} else {
1221		istart_m = ilo;
1222		istop_m = ihi;
1223	}
1224	if ilo + n_shifts + 1 < istop_m {
1225		let mut i = ilo + n_block;
1226		while i < istop_m {
1227			let iblock = Ord::min(istop_m - i, wh.ncols());
1228			let mut a_slice = a.rb_mut().submatrix_mut(ilo, i, n_block, iblock);
1229			let mut wh_slice = wh.rb_mut().submatrix_mut(0, 0, a_slice.nrows(), a_slice.ncols());
1230			matmul(
1231				wh_slice.rb_mut(),
1232				Accum::Replace,
1233				u2.rb().adjoint(),
1234				a_slice.rb(),
1235				one::<T>(),
1236				parallelism,
1237			);
1238			a_slice.copy_from(wh_slice.rb());
1239			i += iblock;
1240		}
1241	}
1242	if istart_m < ilo {
1243		let mut i = istart_m;
1244		while i < ilo {
1245			let iblock = Ord::min(ilo - i, wv.nrows());
1246			let mut a_slice = a.rb_mut().submatrix_mut(i, ilo, iblock, n_block);
1247			let mut wv_slice = wv.rb_mut().submatrix_mut(0, 0, a_slice.nrows(), a_slice.ncols());
1248			matmul(wv_slice.rb_mut(), Accum::Replace, a_slice.rb(), u2.rb(), one::<T>(), parallelism);
1249			a_slice.copy_from(wv_slice.rb());
1250			i += iblock;
1251		}
1252	}
1253	if let Some(mut z) = z.rb_mut() {
1254		let mut i = 0;
1255		while i < n {
1256			let iblock = Ord::min(n - i, wv.nrows());
1257			let mut z_slice = z.rb_mut().submatrix_mut(i, ilo, iblock, n_block);
1258			let mut wv_slice = wv.rb_mut().submatrix_mut(0, 0, z_slice.nrows(), z_slice.ncols());
1259			matmul(wv_slice.rb_mut(), Accum::Replace, z_slice.rb(), u2.rb(), one::<T>(), parallelism);
1260			z_slice.copy_from(wv_slice.rb());
1261			i += iblock;
1262		}
1263	}
1264	*i_pos_block = ilo + n_block - n_shifts;
1265}

1267#[math]
1268fn move_bulges_down<T: RealField>(
1269	ilo: usize,
1270	ihi: usize,
1271	n_block_desired: usize,
1272	n_bulges: usize,
1273	n_shifts: usize,
1274	want_t: bool,
1275	mut a: MatMut<'_, T>,
1276	mut z: Option<MatMut<'_, T>>,
1277	mut u: MatMut<'_, T>,
1278	mut v: MatMut<'_, T>,
1279	mut wh: MatMut<'_, T>,
1280	mut wv: MatMut<'_, T>,
1281	s_re: ColRef<'_, T>,
1282	s_im: ColRef<'_, T>,
1283	i_pos_block: &mut usize,
1284	parallelism: Par,
1285) {
1286	let n = a.nrows();
1287	let eps = eps::<T>();
1288	let small_num = min_positive::<T>() / eps * from_f64::<T>(n as f64);
1289	while *i_pos_block + n_block_desired < ihi {
1290		let n_pos = Ord::min(n_block_desired - n_shifts, ihi - n_shifts - 1 - *i_pos_block);
1291		let n_block = n_shifts + n_pos;
1292		let mut u2 = u.rb_mut().submatrix_mut(0, 0, n_block, n_block);
1293		u2.fill(zero());
1294		u2.rb_mut().diagonal_mut().fill(one::<T>());
1295		let mut istart_m = *i_pos_block;
1296		let mut istop_m = *i_pos_block + n_block;
1297		for i_pos_last in *i_pos_block + n_shifts - 2..*i_pos_block + n_shifts - 2 + n_pos {
1298			for i_bulge in 0..n_bulges {
1299				let i_pos = i_pos_last - 2 * i_bulge;
1300				let mut v = v.rb_mut().col_mut(i_bulge);
1301				let mut h = a.rb_mut().submatrix_mut(i_pos - 1, i_pos - 1, 4, 4);
1302				let s1_re = copy(s_re[s_re.nrows() - 1 - 2 * i_bulge]);
1303				let s1_im = copy(s_im[s_im.nrows() - 1 - 2 * i_bulge]);
1304				let s2_re = copy(s_re[s_re.nrows() - 1 - 2 * i_bulge - 1]);
1305				let s2_im = copy(s_im[s_im.nrows() - 1 - 2 * i_bulge - 1]);
1306				move_bulge(h.rb_mut(), v.rb_mut(), (s1_re, s1_im), (s2_re, s2_im));
1307				let v0 = real(v[0]);
1308				let v1 = copy(v[1]);
1309				let v2 = copy(v[2]);
1310				for j in istart_m..i_pos + 3 {
1311					let sum = a[(j, i_pos)] + (v1 * a[(j, i_pos + 1)]) + v2 * a[(j, i_pos + 2)];
1312					a[(j, i_pos)] = a[(j, i_pos)] - mul_real(sum, v0);
1313					a[(j, i_pos + 1)] = a[(j, i_pos + 1)] - mul_real(sum, v0) * conj(v1);
1314					a[(j, i_pos + 2)] = a[(j, i_pos + 2)] - mul_real(sum, v0) * conj(v2);
1315				}
1316				let sum = a[(i_pos, i_pos)] + conj(v1) * a[(i_pos + 1, i_pos)] + conj(v2) * a[(i_pos + 2, i_pos)];
1317				a[(i_pos, i_pos)] = a[(i_pos, i_pos)] - mul_real(sum, v0);
1318				a[(i_pos + 1, i_pos)] = a[(i_pos + 1, i_pos)] - mul_real(sum, v0) * v1;
1319				a[(i_pos + 2, i_pos)] = a[(i_pos + 2, i_pos)] - mul_real(sum, v0) * v2;
1320				if i_pos > ilo && a[(i_pos, i_pos - 1)] != zero() {
1321					let mut tst1 = abs1(a[(i_pos - 1, i_pos - 1)]) + abs1(a[(i_pos, i_pos)]);
1322					if tst1 == zero() {
1323						if i_pos > ilo + 1 {
1324							tst1 = tst1 + abs1(a[(i_pos - 1, i_pos - 2)]);

1728			}
1729		}
1730		let (ls, ld) = aggressive_early_deflation(
1731			want_t,
1732			a.rb_mut(),
1733			z.rb_mut(),
1734			w_re.rb_mut(),
1735			w_im.rb_mut(),
1736			istart,
1737			istop,
1738			nw,
1739			parallelism,
1740			stack.rb_mut(),
1741			params,
1742		);
1743		count_aed += 1;
1744		istop -= ld;
1745		if ld > 0 {
1746			k_defl = 0;
1747		}
1748		if ld > 0 && (100 * ld > nwr * nibble || (istop - istart) <= Ord::min(nmin, nw_max)) {
1749			continue;
1750		}
1751		k_defl += 1;
1752		let mut ns = Ord::min(nh - 1, Ord::min(Ord::max(2, ls), nsr));
1753		ns = ns / 2 * 2;
1754		let mut i_shifts = istop - ls;
1755		if k_defl % non_convergence_limit_shift == 0 {
1756			for i in (i_shifts + 1..istop).rev().step_by(2) {
1757				if i >= ilo + 2 {
1758					let ss = abs1(a[(i, i - 1)]) + abs1(a[(i - 1, i - 2)]);
1759					let aa = dat1 * ss + a[(i, i)];
1760					let bb = copy(ss);
1761					let cc = dat2 * ss;
1762					let dd = copy(aa);
1763					let (s1, s2) = lahqr_eig22(aa, bb, cc, dd);
1764					w_re[i - 1] = s1.0;
1765					w_im[i - 1] = s1.1;
1766					w_re[i] = s2.0;
1767					w_im[i] = s2.1;
1768				} else {
1769					w_re[i] = copy(a[(i, i)]);
1770					w_re[i - 1] = copy(a[(i, i)]);
1771					w_im[i] = zero();
1772					w_im[i - 1] = zero();
1773				}
1774			}
1775		} else {
1776			if ls <= ns / 2 {
1777				let mut temp = a.rb_mut().submatrix_mut(n - ns, 0, ns, ns);
1778				let mut shifts_re = w_re.rb_mut().subrows_mut(istop - ns, ns);
1779				let mut shifts_im = w_im.rb_mut().subrows_mut(istop - ns, ns);
1780				let ierr = lahqr(false, temp.rb_mut(), None, shifts_re.rb_mut(), shifts_im.rb_mut(), 0, ns) as usize;
1781				ns = ns - ierr;
1782				if ns < 2 {
1783					let aa = copy(a[(istop - 2, istop - 2)]);
1784					let bb = copy(a[(istop - 2, istop - 1)]);
1785					let cc = copy(a[(istop - 1, istop - 2)]);
1786					let dd = copy(a[(istop - 1, istop - 1)]);
1787					let (s1, s2) = lahqr_eig22(aa, bb, cc, dd);
1788					w_re[istop - 2] = s1.0;
1789					w_im[istop - 2] = s1.1;
1790					w_re[istop - 1] = s2.0;
1791					w_im[istop - 1] = s2.1;
1792					ns = 2;
1793				}
1794				i_shifts = istop - ns;
1795			}
1796			let mut sorted = false;
1797			let mut k = istop;
1798			while !sorted && k > i_shifts {
1799				sorted = true;
1800				for i in i_shifts..k - 1 {
1801					if (abs(w_re[i]) + abs(w_im[i])) < (abs(w_re[i + 1]) + abs(w_im[i + 1])) {
1802						sorted = false;
1803						let wi = (copy(w_re[i]), copy(w_im[i]));
1804						let wip1 = (copy(w_re[i + 1]), copy(w_im[i + 1]));
1805						w_re[i] = wip1.0;
1806						w_im[i] = wip1.1;
1807						w_re[i + 1] = wi.0;
1808						w_im[i + 1] = wi.1;
1809					}
1810				}
1811				k -= 1;
1812			}
1813			for i in (i_shifts + 2..istop).rev().step_by(2) {
1814				if w_im[i] != (-w_im[i - 1]) {
1815					let tmp = (copy(w_re[i]), copy(w_im[i]));
1816					w_re[i] = copy(w_re[i - 1]);
1817					w_im[i] = copy(w_im[i - 1]);
1818					w_re[i - 1] = copy(w_re[i - 2]);
1819					w_im[i - 1] = copy(w_im[i - 2]);
1820					w_re[i - 2] = tmp.0;
1821					w_im[i - 2] = tmp.1;
1822				}
1823			}
1824			if ns > 1 && ns % 2 == 1 {
1825				ns -= 1;
1826			}
1827			i_shifts = istop - ns;
1828		}
1829		if ns == 2 && (w_im[i_shifts] == zero()) {
1830			if abs((w_re[i_shifts] - a[(istop - 1, istop - 1)])) < abs((w_re[i_shifts + 1] - a[(istop - 1, istop - 1)])) {
1831				w_re[i_shifts + 1] = copy(w_re[i_shifts]);
1832				w_im[i_shifts + 1] = copy(w_im[i_shifts]);
1833			} else {
1834				w_re[i_shifts] = copy(w_re[i_shifts + 1]);
1835				w_im[i_shifts] = copy(w_im[i_shifts + 1]);
1836			}
1837		}
1838		let mut shifts_re = w_re.rb_mut().subrows_mut(i_shifts, ns);
1839		let mut shifts_im = w_im.rb_mut().subrows_mut(i_shifts, ns);
1840		multishift_qr_sweep(
1841			want_t,
1842			a.rb_mut(),
1843			z.rb_mut(),
1844			shifts_re.rb_mut(),
1845			shifts_im.rb_mut(),
1846			istart,
1847			istop,
1848			parallelism,
1849			stack.rb_mut(),
1850		);
1851		count_sweep += 1;
1852	}
1853	(info, count_aed, count_sweep)
1854}

1856pub fn lahqr<T: RealField>(
1857	want_t: bool,
1858	a: MatMut<'_, T>,
1859	z: Option<MatMut<'_, T>>,
1860	w_re: ColMut<'_, T>,
1861	w_im: ColMut<'_, T>,
1862	ilo: usize,
1863	ihi: usize,
1864) -> isize {
1865	let epsilon = eps::<T>();
1866	let zero_threshold = min_positive::<T>();
1867	assert!(a.nrows() == a.ncols());
1868	assert!(ilo <= ihi);
1869	let n = a.nrows();
1870	let nh = ihi - ilo;
1871	assert!(w_re.nrows() == n);
1872	assert!(w_im.nrows() == n);
1873	assert!(w_re.ncols() == 1);
1874	assert!(w_im.ncols() == 1);
1875	if let Some(z) = z.rb() {
1876		assert!(z.nrows() == n);
1877		assert!(z.ncols() == n);
1878	}
1879	let mut a = a;
1880	let mut z = z;
1881	let mut w_re = w_re;
1882	let mut w_im = w_im;
1883	let one = one::<T>();
1884	let eps = epsilon;
1885	let small_num = zero_threshold / eps;
1886	let non_convergence_limit = 10;
1887	let dat1 = from_f64::<T>(0.75);
1888	let dat2 = from_f64::<T>(-0.4375);
1889	if nh == 0 {
1890		return 0;
1891	}
1892	if nh == 1 {
1893		w_re[ilo] = copy(a[(ilo, ilo)]);
1894		w_im[ilo] = zero();
1895	}
1896	let itmax = 30 * Ord::max(10, nh);
1897	let mut k_defl = 0usize;
1898	let mut istop = ihi;
1899	let mut istart = ilo;
1900	stack_mat!(v, 3, 1, T);
1901	let mut v = v.rb_mut().col_mut(0);
1902	for iter in 0..itmax + 1 {
1903		if iter == itmax {
1904			return istop as isize;
1905		}
1906		if istart + 1 >= istop {
1907			if istart + 1 == istop {
1908				w_re[istart] = copy(a[(istart, istart)]);
1909				w_im[istart] = zero();
1910			}
1911			break;
1912		}
1913		let istart_m;
1914		let istop_m;
1915		if !want_t {
1916			istart_m = istart;
1917			istop_m = istop;
1918		} else {
1919			istart_m = 0;
1920			istop_m = n;
1921		}
1922		for i in (istart + 1..istop).rev() {
1923			if abs(a[(i, i - 1)]) < small_num {
1924				a[(i, i - 1)] = zero();
1925				istart = i;
1926				break;
1927			}
1928			let mut tst = abs(a[(i - 1, i - 1)]) + abs(a[(i, i)]);
1929			if tst == zero() {
1930				if i >= ilo + 2 {
1931					tst = tst + abs(a[(i - 1, i - 2)]);
1932				}
1933				if i + 1 < ihi {
1934					tst = tst + abs(a[(i + 1, i)]);
1935				}
1936			}
1937			if abs(a[(i, i - 1)]) <= (eps * tst) {
1938				let ab = max(abs(a[(i, i - 1)]), abs(a[(i - 1, i)]));
1939				let ba = min(abs(a[(i, i - 1)]), abs(a[(i - 1, i)]));
1940				let aa = max(abs(a[(i, i)]), abs((a[(i, i)] - a[(i - 1, i - 1)])));
1941				let bb = min(abs(a[(i, i)]), abs((a[(i, i)] - a[(i - 1, i - 1)])));
1942				let s = aa + ab;
1943				if ba * (ab / s) <= max(small_num, eps * (bb * (aa / s))) {
1944					a[(i, i - 1)] = zero();
1945					istart = i;
1946					break;
1947				}
1948			}
1949		}
1950		if istart + 2 >= istop {
1951			if istart + 1 == istop {
1952				k_defl = 0;
1953				w_re[istart] = copy(a[(istart, istart)]);
1954				w_im[istart] = zero();
1955				istop = istart;
1956				istart = ilo;
1957				continue;
1958			}
1959			if istart + 2 == istop {
1960				let ((a00, a01, a10, a11), (s1_re, s1_im), (s2_re, s2_im), (cs, sn)) = lahqr_schur22(
1961					copy(a[(istart, istart)]),
1962					copy(a[(istart, istart + 1)]),
1963					copy(a[(istart + 1, istart)]),
1964					copy(a[(istart + 1, istart + 1)]),
1965				);
1966				let rot = JacobiRotation { c: cs, s: sn };
1967				a[(istart, istart)] = a00;
1968				a[(istart, istart + 1)] = a01;
1969				a[(istart + 1, istart)] = a10;
1970				a[(istart + 1, istart + 1)] = a11;
1971				w_re[istart] = s1_re;
1972				w_im[istart] = s1_im;
1973				w_re[istart + 1] = s2_re;
1974				w_im[istart + 1] = s2_im;
1975				if want_t {
1976					if istart + 2 < istop_m {
1977						rot.apply_on_the_right_in_place(
1978							a.rb_mut()
1979								.transpose_mut()
1980								.get_mut(istart + 2..istop_m, ..)
1981								.two_cols_mut(istart, istart + 1),
1982						);
1983					}
1984					rot.apply_on_the_right_in_place(a.rb_mut().get_mut(istart_m..istart, ..).two_cols_mut(istart, istart + 1));
1985				}
1986				if let Some(z) = z.rb_mut() {
1987					rot.apply_on_the_right_in_place(z.two_cols_mut(istart, istart + 1));
1988				}
1989				k_defl = 0;
1990				istop = istart;
1991				istart = ilo;
1992				continue;
1993			}
1994		}
1995		let (a00, a01, a10, a11);
1996		k_defl += 1;
1997		if k_defl % non_convergence_limit == 0 {
1998			let mut s = abs(a[(istop - 1, istop - 2)]);
1999			if istop > ilo + 2 {
2000				s = s + abs(a[(istop - 2, istop - 3)]);
2001			};
2002			a00 = dat1 * s + a[(istop - 1, istop - 1)];
2003			a01 = dat2 * s;
2004			a10 = s;
2005			a11 = copy(a00);
2006		} else {
2007			a00 = copy(a[(istop - 2, istop - 2)]);
2008			a10 = copy(a[(istop - 1, istop - 2)]);
2009			a01 = copy(a[(istop - 2, istop - 1)]);
2010			a11 = copy(a[(istop - 1, istop - 1)]);
2011		}
2012		let (mut s1, mut s2) = lahqr_eig22(a00, a01, a10, a11);
2013		if s1.1 == zero() && s2.1 == zero() {
2014			if abs((s1.0 - a[(istop - 1, istop - 1)])) <= abs((s2.0 - a[(istop - 1, istop - 1)])) {
2015				s2 = (copy(s1.0), copy(s1.1));
2016			} else {
2017				s1 = (copy(s2.0), copy(s2.1));
2018			}
2019		}
2020		let mut istart2 = istart;
2021		if istart + 3 < istop {
2022			for i in (istart + 1..istop - 2).rev() {
2023				let h = a.rb().submatrix(i, i, 3, 3);
2024				lahqr_shiftcolumn(h, v.rb_mut(), (copy(s1.0), copy(s1.1)), (copy(s2.0), copy(s2.1)));
2025				let mut head = copy(v[0]);
2026				let HouseholderInfo { tau, .. } = make_householder_in_place(&mut head, v.rb_mut().subrows_mut(1, 2));
2027				let tau = recip(tau);
2028				let v0 = tau;
2029				let v1 = copy(v[1]);
2030				let v2 = copy(v[2]);
2031				let refsum = v0 * a[(i, i - 1)] + v1 * a[(i + 1, i - 1)];
2032				if (abs((a[(i + 1, i - 1)] - (refsum * v1))) + abs((refsum * v2)))
2033					<= (eps * ((abs(a[(i, i - 1)]) + abs(a[(i, i + 1)])) + abs(a[(i + 1, i + 2)])))
2034				{
2035					istart2 = i;
2036					break;
2037				}
2038			}
2039		}
2040		for i in istart2..istop - 1 {
2041			let nr = Ord::min(3, istop - i);
2042			let mut t1;
2043			if i == istart2 {
2044				let h = a.rb().submatrix(i, i, nr, nr);
2045				let mut x = v.rb_mut().subrows_mut(0, nr);
2046				lahqr_shiftcolumn(h, x.rb_mut(), (copy(s1.0), copy(s1.1)), (copy(s2.0), copy(s2.1)));
2047				let mut beta = copy(x[0]);
2048				let tail = x.rb_mut().subrows_mut(1, nr - 1);
2049				HouseholderInfo { tau: t1, .. } = make_householder_in_place(&mut beta, tail);
2050				v[0] = beta;
2051				t1 = recip(t1);
2052				if i > istart {
2053					a[(i, i - 1)] = a[(i, i - 1)] * (one - t1);
2054				}
2055			} else {
2056				v[0] = copy(a[(i, i - 1)]);
2057				v[1] = copy(a[(i + 1, i - 1)]);
2058				if nr == 3 {
2059					v[2] = copy(a[(i + 2, i - 1)]);
2060				}
2061				let mut beta = copy(v[0]);
2062				let tail = v.rb_mut().subrows_mut(1, nr - 1);
2063				HouseholderInfo { tau: t1, .. } = make_householder_in_place(&mut beta, tail);
2064				t1 = recip(t1);
2065				v[0] = copy(beta);
2066				a[(i, i - 1)] = copy(beta);
2067				a[(i + 1, i - 1)] = zero();
2068				if nr == 3 {
2069					a[(i + 2, i - 1)] = zero();
2070				}
2071			}
2072			let v2 = copy(v[1]);
2073			let t2 = t1 * v2;
2074			if nr == 3 {
2075				let v3 = copy(v[2]);
2076				let t3 = t1 * v[2];
2077				for j in i..istop_m {
2078					let sum = a[(i, j)] + v2 * a[(i + 1, j)] + v3 * a[(i + 2, j)];
2079					a[(i, j)] = a[(i, j)] - (sum * t1);
2080					a[(i + 1, j)] = a[(i + 1, j)] - sum * t2;
2081					a[(i + 2, j)] = a[(i + 2, j)] - sum * t3;
2082				}
2083				for j in istart_m..Ord::min(i + 4, istop) {
2084					let sum = a[(j, i)] + v2 * a[(j, i + 1)] + v3 * a[(j, i + 2)];
2085					a[(j, i)] = a[(j, i)] - sum * t1;
2086					a[(j, i + 1)] = a[(j, i + 1)] - sum * t2;
2087					a[(j, i + 2)] = a[(j, i + 2)] - sum * t3;
2088				}
2089				if let Some(mut z) = z.rb_mut() {
2090					for j in 0..n {
2091						let sum = z[(j, i)] + v2 * z[(j, i + 1)] + v3 * z[(j, i + 2)];
2092						z[(j, i)] = z[(j, i)] - sum * t1;
2093						z[(j, i + 1)] = z[(j, i + 1)] - sum * t2;
2094						z[(j, i + 2)] = z[(j, i + 2)] - sum * t3;
2095					}
2096				}
2097			} else {
2098				for j in i..istop_m {
2099					let sum = a[(i, j)] + v2 * a[(i + 1, j)];
2100					a[(i, j)] = a[(i, j)] - sum * t1;
2101					a[(i + 1, j)] = a[(i + 1, j)] - sum * t2;
2102				}
2103				for j in istart_m..Ord::min(i + 3, istop) {
2104					let sum = a[(j, i)] + v2 * a[(j, i + 1)];
2105					a[(j, i)] = a[(j, i)] - sum * t1;
2106					a[(j, i + 1)] = a[(j, i + 1)] - sum * t2;
2107				}
2108				if let Some(mut z) = z.rb_mut() {
2109					for j in 0..n {
2110						let sum = z[(j, i)] + v2 * z[(j, i + 1)];
2111						z[(j, i)] = z[(j, i)] - sum * t1;
2112						z[(j, i + 1)] = z[(j, i + 1)] - sum * t2;
2113					}
2114				}
2115			}
2116		}
2117	}
2118	0
