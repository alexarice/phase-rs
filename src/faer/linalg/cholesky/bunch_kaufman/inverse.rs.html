17	par: Par,
18	stack: &mut MemStack,
19) {

21	assert!(all(
22		out.nrows() == n,

24		L.nrows() == n,
25		L.ncols() == n,

27		subdiagonal.dim() == n,
28		perm.len() == n,

30
31	let mut out = out;
