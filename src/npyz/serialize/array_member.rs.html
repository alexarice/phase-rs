6impl DType {
7    /// Expect an array dtype, get the inner dtype.
8    fn array_inner_dtype(&self, expected_len: u64) -> Result<&Self, DTypeError> {
9        match *self {
10            DType::Record { .. } => Err(DTypeError(ErrorKind::ExpectedArray { got: "a record" })),
11            DType::Plain { .. } => Err(DTypeError(ErrorKind::ExpectedArray { got: "a scalar" })),
12            DType::Array(len, ref ty) => {
13                if len != expected_len {
14                    return Err(DTypeError(ErrorKind::WrongArrayLen {
15                        actual: len,
16                        expected: expected_len,
17                    }));
18                }
19
20                Ok(ty)
21            },
22        }
23    }
24}

26pub struct ArrayReader<I, const N: usize>{ inner: I }
27pub struct ArrayWriter<I, const N: usize>{ inner: I }
28
29impl<I: TypeRead, const N: usize> TypeRead for ArrayReader<I, N>
30where I::Value: Copy + Default,
31{
32    type Value = [I::Value; N];
33
34    #[inline]
35    fn read_one<R: io::Read>(&self, mut reader: R) -> io::Result<Self::Value> {
36        let mut value = [I::Value::default(); N];
37        for place in &mut value {
38            *place = self.inner.read_one(&mut reader)?;
39        }
40
41        Ok(value)
42    }
43}
44
45impl<I: TypeWrite, const N: usize> TypeWrite for ArrayWriter<I, N>
46where I::Value: Sized,
47{
48    type Value = [I::Value; N];
49
50    #[inline]
51    fn write_one<W: io::Write>(&self, mut writer: W, value: &Self::Value) -> io::Result<()>
52    where Self: Sized,
53    {
54        for item in value {
55            self.inner.write_one(&mut writer, item)?;
56        }
57        Ok(())
58    }
59}

68impl<T: Deserialize + Default + Copy, const N: usize> Deserialize for [T; N] {
69    type TypeReader = ArrayReader<<T as Deserialize>::TypeReader, N>;
70
71    #[inline]
72    fn reader(dtype: &DType) -> Result<Self::TypeReader, DTypeError> {
73        let inner_dtype = dtype.array_inner_dtype(N as u64)?;
74        let inner = <T>::reader(inner_dtype)?;
75        Ok(ArrayReader { inner })
76    }
77}
