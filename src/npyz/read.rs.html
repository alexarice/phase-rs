7/// Object for reading an `npy` file.
8///
9/// This type represents a partially read `npy` file, where the header has been parsed
10/// and we are ready to begin parsing data.
11/// ```
12/// # fn main() -> std::io::Result<()> {
13/// use std::fs::File;
14/// use std::io;
15///
16/// let file = io::BufReader::new(File::open("./test-data/c-order.npy")?);
17/// let npy = npyz::NpyFile::new(file)?;
18///
19/// // Helper methods for inspecting the layout of the data.
20/// assert_eq!(npy.shape(), &[2, 3, 4]);
21/// assert_eq!(npy.strides(), &[12, 4, 1]);
22/// assert_eq!(npy.order(), npyz::Order::C);
23///
24/// // Get the data!
25/// let data: Vec<i64> = npy.into_vec()?;
26/// assert_eq!(data.len(), 24);
27/// # Ok(()) }
28/// ```
29///
30/// # Working with large files
31///
32/// For large files, it may be undesirable to read all of the data into a Vec.
33/// The [`NpyFile::data`] method allows you to iterate over the data instead.
34///
35/// ```
36/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
37/// # use std::fs::File;
38/// # use std::io;
39/// #
40/// # let file = io::BufReader::new(File::open("./test-data/c-order.npy")?);
41/// let npy = npyz::NpyFile::new(file)?;
42///
43/// let mut sum = 0;
44/// for x in npy.data::<i64>()? {
45///     sum += x?;  // items are Result
46/// }
47/// assert_eq!(sum, 84);
48/// # Ok(()) }
49/// ```
50///
51/// # Related types
52///
53/// Is a read adaptor too heavy for you?  [`NpyFile`] is ultimately just a
54/// [`NpyHeader`] paired with its input stream, so you may consider parsing
55/// [`NpyHeader`] instead if you need something easier to clone or send.
56///
57/// ```
58/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
59/// # use std::fs::File;
60/// # use std::io;
61/// #
62/// let mut file = io::BufReader::new(File::open("./test-data/c-order.npy")?);
63/// let header = npyz::NpyHeader::from_reader(&mut file)?;
64/// assert_eq!(header.shape(), &[2, 3, 4]);
65///
66/// // now you can store `header` somewhere
67/// // ...
68/// // and later in your program you can construct an NpyFile to read the data
69///
70/// let npy = npyz::NpyFile::with_header(header, file);
71/// let data: Vec<i64> = npy.into_vec()?;
72/// assert_eq!(data.len(), 24);
73/// # Ok(()) }
74/// ```
75///
76/// # Migrating from `npy-rs 0.4.0`
77///
78/// [`NpyData`] is still provided, but it is deprecated in favor of [`NpyFile`].
79/// At construction, since `&[u8]` impls `Read`, you can still use them as input.
80///
81/// ```text
82/// was:
83///     npyz::NpyData::<i64>::from_bytes(&bytes).to_vec()
84/// now:
85///     npyz::NpyFile::new(&bytes[..]).into_vec::<i64>()
86/// ```
87///
88/// If you were using the iterator API of `NpyData`, this is now on [`NpyReader`], which
89/// requires us to call [`NpyFile::data`]:
90///
91/// ```text
92/// was:
93///     let iter = npyz::NpyData::<i64>::new(&bytes)?;
94/// now:
95///     let iter = npyz::NpyFile::new(&bytes[..]).data::<i64>.map_err(invalid_data)?;
96/// ```
97///
98/// where the following function has been used to paper over the fact that [`NpyFile::data`]
99/// has a different Error type:
100/// ```rust
101/// # #[allow(unused)]
102/// fn invalid_data<S: ToString>(err: S) -> std::io::Error {
103///     std::io::Error::new(std::io::ErrorKind::InvalidData, err.to_string())
104/// }
105/// ```
106///
107/// [`NpyData::is_empty`] is gone due to possible ambiguity between [`NpyReader::len`] and [`NpyReader::total_len`].
108/// Use the one that is appropriate for what you are doing.
109///
110/// If you were using [`NpyData::get`]... well, honestly, you should first consider whether
111/// you could just iterate over the reader instead.  But if you were using [`NpyData::get`]
112/// because you *genuinely need* random access, then there is [`NpyReader::read_at`].
113///
114/// ```text
115/// was:
116///     // note:  0 <= i < arr.len()
117///     arr.get(i)
118/// now:
119///     // note:  0 <= i < reader.total_len()
120///     reader.read_at(i)?
121/// ```
122#[derive(Clone)]
123pub struct NpyFile<R: io::Read> {
124    header: NpyHeader,
125    reader: R,
126}

128/// Represents the parsed header portion of an `npy` file.
129///
130/// The header contains all of the information necessary to interpret the raw
131/// data stream.  It provides the datatype, dimensions, and axis ordering of
132/// the data.
133///
134/// ```
135/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
136/// # use std::fs::File;
137/// # use std::io;
138/// #
139/// let mut file = io::BufReader::new(File::open("./test-data/c-order.npy")?);
140/// let header = npyz::NpyHeader::from_reader(&mut file)?;
141/// assert_eq!(header.shape(), &[2, 3, 4]);
142///
143/// // now you can store `header` somewhere
144/// // ...
145/// // and later in your program you can construct an NpyFile to read the data
146///
147/// let npy = npyz::NpyFile::with_header(header, file);
148/// let data: Vec<i64> = npy.into_vec()?;
149/// assert_eq!(data.len(), 24);
150/// # Ok(()) }
151/// ```
152#[derive(Clone)]
153pub struct NpyHeader {
154    dtype: DType,
155    shape: Vec<u64>,
156    /// Strides of each axis, pre-computed from the shape/order.
157    strides: Vec<u64>,
158    order: Order,
159    /// Total number of elements, pre-computed from the shape.
160    n_records: u64,
161    /// Item size in bytes.
162    item_size: usize,
163}
164
165impl NpyHeader {
166    /// Parse a header from the reader for an NPY file.
167    ///
168    /// The reader must initially be at the beginning of an NPY file. After this
169    /// function returns `Ok(_)`, the reader will have been advanced to the
170    /// beginning of the raw data bytes.
171    pub fn from_reader(r: impl io::Read) -> io::Result<NpyHeader> {
172        NpyHeader::read_and_interpret(r)
173    }
174}

201/// Order of axes in a file.
202#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
203pub enum Order {
204    /// Indices are arranged from slowest to fastest, so that the last dimension has a stride of 1.
205    C,
206    /// Indices are arranged from fastest to slowest, so that the first dimension has a stride of 1.
207    Fortran,
208}
209
210impl Order {
211    pub(crate) fn from_fortran_order(fortran_order: bool) -> Order {
212        if fortran_order { Order::Fortran } else { Order::C }
213    }
214}

274impl<R: io::Read> NpyFile<R> {
275    /// Read all elements into a flat `Vec`, in the order they are stored as.
276    ///
277    /// This is a convenience wrapper around [`Self::data`] and [`Iterator::collect`].
278    pub fn into_vec<T: Deserialize>(self) -> io::Result<Vec<T>> {
279        match self.data() {
280            Ok(r) => r.collect(),
281            Err(e) => Err(invalid_data(e)),
282        }

285    /// Produce an [`NpyReader`] to begin reading elements, if `T` can be deserialized from the file's dtype.
286    ///
287    /// The returned type implements [`Iterator`]`<Item=io::Result<T>>`, and provides additional methods
288    /// for random access when `R: Seek`.  See [`NpyReader`] for more details.
289    pub fn data<T: Deserialize>(self) -> Result<NpyReader<T, R>, DTypeError> {
290        let NpyFile { reader, header } = self;
291        let type_reader = T::reader(&header.dtype)?;
292        Ok(NpyReader { type_reader, header, reader_and_current_index: (reader, 0) })
293    }
294
295    /// Produce an [`NpyReader`] to begin reading elements, if `T` can be deserialized from the file's dtype.
296    ///
297    /// This fallible form of the function returns `self` on error, so that you can try again with a different `T`.
298    pub fn try_data<T: Deserialize>(self) -> Result<NpyReader<T, R>, Self> {
299        let type_reader = match T::reader(&self.header.dtype) {
300            Ok(r) => r,
301            Err(_) => return Err(self),
302        };
303        let NpyFile { reader, header } = self;
304        Ok(NpyReader { type_reader, header, reader_and_current_index: (reader, 0) })
305    }

308impl NpyHeader {
309    fn read_and_interpret(mut r: impl io::Read) -> io::Result<NpyHeader> {
310        let header = read_header(&mut r)?;
311
312        let dict = match header {
313            Value::Dict(dict) => dict
314                .into_iter()
315                .map(|(k, v)| Ok((k.as_string().ok_or(invalid_data("key is not string"))?.to_owned(), v)))
316                .collect::<io::Result<HashMap<String, Value>>>()?,
317            _ => return Err(invalid_data("expected a python dict literal")),
318        };
319
320        let expect_key = |key: &str| {
321            dict.get(key).ok_or_else(|| invalid_data(format_args!("dict is missing key '{}'", key)))
322        };
323
324        let order = match expect_key("fortran_order")? {
325            &Value::Boolean(b) => Order::from_fortran_order(b),
326            _ => return Err(invalid_data(format_args!("'fortran_order' value is not a bool"))),
327        };
328
329        let shape = convert_value_to_shape(expect_key("shape")?)?;
330
331        let descr: &Value = expect_key("descr")?;
332        let dtype = DType::from_descr(descr)?;
333
334        Self::from_parts(dtype, shape, order)
335    }
336
337    fn from_parts(dtype: DType, shape: Vec<u64>, order: Order) -> io::Result<NpyHeader> {
338        let n_records = shape.iter().product();
339        let item_size = dtype.num_bytes().ok_or_else(|| {
340            invalid_data(format_args!("dtype is larger than usize!"))
341        })?;
342        let strides = strides(order, &shape);
343        Ok(NpyHeader { dtype, shape, strides, order, n_records, item_size })
344    }
345}

375/// # Random access methods
376impl<R: io::Read, T: Deserialize> NpyReader<T, R> where R: io::Seek {
377    /// Move the read cursor to the item at the given index.
378    ///
379    /// Be aware that this will affect [`Self::len`], which is always computed
380    /// from the current position.
381    /// Seeking to [`Self::total_len`] is well defined.
382    ///
383    /// # Panics
384    ///
385    /// Panics if the index is greater than [`Self::total_len`].
386    pub fn seek_to(&mut self, index: u64) -> io::Result<()> {
387        let len = self.total_len();
388        assert!(index <= len, "index out of bounds for seeking (the index is {} but the len is {})", index, len);
389
390        let (reader, current_index) = &mut self.reader_and_current_index;
391        let delta = index as i64 - *current_index as i64;
392        if delta != 0 {
393            reader.seek(io::SeekFrom::Current(delta * self.header.item_size as i64))?;
394            *current_index = index;
395        }
396        Ok(())
397    }
398
399    /// Read a single item at the given position, leaving the cursor at the position after it.
400    ///
401    /// # Panics
402    ///
403    /// Panics if the index is out of bounds. (`>=` to [`Self::total_len`]).
404    pub fn read_at(&mut self, index: u64) -> io::Result<T> {
405        let len = self.total_len();
406        assert!(index < len, "index out of bounds for reading (the index is {} but the len is {})", index, len);
407
408        self.seek_to(index)?;
409        self.next().unwrap()
410    }
411}

414impl<'a, T: Deserialize> NpyData<'a, T> {
415    /// Deserialize a NPY file represented as bytes
416    pub fn from_bytes(bytes: &'a [u8]) -> io::Result<NpyData<'a, T>> {
417        let inner = NpyFile::new(bytes)?.data().map_err(invalid_data)?;
418
419        assert_eq!(inner.header.item_size as u64 * inner.header.n_records, inner.reader().len() as u64);
420        Ok(NpyData { inner })
421    }
422
423    #[inline(always)] // this should optimize into just a copy of a pointer-sized field
424    fn get_data_slice(&self) -> &'a [u8] {
425        self.inner.reader()
426    }
427
428    /// Returns the total number of elements.
429    pub fn len(&self) -> usize {
430        // we can safely assume that the len is <= usize::MAX since the entire file has already
431        // been mapped into the address space (even if not occupying physical memory)
432        self.inner.total_len() as usize
433    }
434
435    /// Returns whether there are zero records in this NpyData structure.
436    pub fn is_empty(&self) -> bool { self.len() == 0 }
437
438    /// Gets a single data-record with the specified flat index.
439    ///
440    /// Returns None if the index is out of bounds.
441    ///
442    /// # Panics
443    ///
444    /// Panics if the bytes stored for the element are invalid for the dtype.
445    pub fn get(&self, i: usize) -> Option<T> {
446        if i < self.len() {
447            Some(self.get_unchecked(i))
448        } else {
449            None
450        }
451    }
452
453    /// Gets a single data-record with the specified index.
454    ///
455    /// # Panics
456    ///
457    /// Panics if the bytes stored for the element are invalid for the dtype,
458    /// or if the index is out of bounds.
459    pub fn get_unchecked(&self, i: usize) -> T {
460        let item_bytes = &self.get_data_slice()[i * self.inner.header.item_size..];
461        self.inner.type_reader.read_one(item_bytes).unwrap()
462    }
463
464    /// Construct a vector with the deserialized contents of the whole file.
465    ///
466    /// The output is a flat vector with the elements in the same order that they are in the file.
467    /// [`NpyData`] is deprecated and does not provide any tools for inspecting the shape and
468    /// layout of the data, so if you want to correctly read multi-dimensional arrays you should
469    /// switch to [`NpyFile`].
470    pub fn to_vec(&self) -> Vec<T> {
471        let &(mut reader) = self.inner.reader();
472        (0..self.len()).map(|_| self.inner.type_reader.read_one(&mut reader).unwrap()).collect()
473    }
474}

476fn strides(order: Order, shape: &[u64]) -> Vec<u64> {
477    match order {
478        Order::C => {
479            let mut strides = prefix_products(shape.iter().rev().copied()).collect::<Vec<_>>();
480            strides.reverse();
481            strides
482        },
483        Order::Fortran => prefix_products(shape.iter().copied()).collect(),
484    }
485}
486
487fn prefix_products<I: IntoIterator<Item=u64>>(iter: I) -> impl Iterator<Item=u64> {
488    iter.into_iter().scan(1, |acc, x| { let old = *acc; *acc *= x; Some(old) })
489}

495impl<R, T> Iterator for NpyReader<T, R> where T: Deserialize, R: io::Read {
496    type Item = io::Result<T>;
497
498    fn next(&mut self) -> Option<Self::Item> {
499        let (reader, current_index) = &mut self.reader_and_current_index;
500        if *current_index < self.header.n_records {
501            *current_index += 1;
502            return Some(self.type_reader.read_one(reader));
503        }
504        None
505    }
506
507    fn size_hint(&self) -> (usize, Option<usize>) {
508        let u64_len = self.len();
509        if u64_len > usize::MAX as u64 {
510            (usize::MAX, None)
511        } else {
512            (u64_len as usize, Some(u64_len as usize))
513        }
514    }
515}
