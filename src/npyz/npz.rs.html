1//! Utilities for working with `npz` files.
2//!
3//! NPZ files are a format used by numpy to store archives of multiple arrays.
4//! The format itself is just a `.zip` file containing files with a `.npy` extension.
5//!
6//! When the `"npz"` feature is enabled, this module provides adapters around the
7//! [`zip`][::zip] crate that read and write NPZ files.
8//!
9//! Even without the `"npz"` feature, this module always provides a set of utility functions
10//! for converting between array names and filenames inside the zip file.
11//! This allows you to use your own zip library. The [`examples` directory] includes an example
12//! of how to read and write an npz without the feature.
13//!
14//! [`examples` directory]: https://github.com/ExpHP/npyz/tree/master/examples
15
16#[cfg(feature = "npz")]
17pub use crate::npz_feature::*;

19/// Get the name of the array that would correspond to the given filename inside a zip file.
20///
21/// This tries to match the behavior of numpy's own npz-loading behavior:
22///
23/// * Case sensitive (only `.npy` files, not `.NPY` files)
24/// * Allows weird characters like `/` and `.`.  Makes no attempt to normalize paths.
25/// * Treats an interior null as the end of the path.  Notice that this means that multiple
26///   different filenames could produce the same array name in a maliciously constructed zip.
27///
28/// Returns `None` if numpy would not consider the file to be an array.
29///
30/// _This function does not require any cargo features._
31pub fn array_name_from_file_name(path_in_zip: &str) -> Option<&str> {
32    let mut path = path_in_zip;
33    if let Some(idx) = path.find("\0") {
34        path = &path[..idx];
35    }
36
37    if path.ends_with(".npy") {
38        Some(&path[..path.len() - 4])
39    } else {
40        None
41    }
42}

44/// Get the filename in a zip that `np.savez` would use for a keyword argument.
45///
46/// **Note:** This does accept the name `"file"`, even though this cannot normally be used in `np.savez`
47/// due to technical limitations. (numpy can read the file just fine)
48///
49/// _This function does not require any cargo features._
50pub fn file_name_from_array_name(name: &str) -> String {
51    format!("{}.npy", name)
52}
53
54/// Get the filename in a zip that would be used for an array supplied as a positional
55/// argument to `np.savez`.
56///
57/// _This function does not require any cargo features._
58pub fn file_name_from_index(index: i32) -> String {
59    format!("arr_{}.npy", index)
60}
