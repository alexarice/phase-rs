8use crate::{combinator::syntax::normal::{AtomN, Buildable, PatternN}, ket::KetState, phase::Phase};

10/// A unitary type "qn <-> qn"
11#[derive(Clone, Copy, Debug, PartialEq)]
12pub struct TermType(pub usize);
13
14impl Sum for TermType {
15    fn sum<I: Iterator<Item = Self>>(iter: I) -> Self {
16        TermType(iter.map(|x| x.0).sum())
17    }
18}
19
20impl TermType {
21    /// Convert a unitary type qn <-> qn to pattern type qn < qn
22    pub fn to_pattern_type(self) -> PatternType {
23        PatternType(self.0, self.0)
24    }
25}

27/// Syntax of typed terms
28#[derive(Clone, Debug, PartialEq)]
29pub enum TermT {
30    /// A non-empty composition "t_1 ; ... ; t_n"
31    Comp(Vec<TermT>),
32    /// A tensor "t_1 x ... x t_n"
33    Tensor(Vec<TermT>),
34    /// An identity "id(n)"
35    Id(TermType),
36    /// A (global) phase operator, e.g. "-1" or "ph(0.1pi)"
37    Phase(Phase),
38    /// An "if let" statement, "if let pattern then inner"
39    IfLet {
40        /// Pattern to match on in "if let"
41        pattern: PatternT,
42        /// Body of the "if let"
43        inner: Box<TermT>,
44    },
45    /// Top level symbol, a named gate
46    Gate {
47        /// Name of symbol/gate
48        name: String,
49        /// Definition of symbol
50        def: Box<TermT>,
51    },
52    /// Inverse of a term "t ^ -1"
53    Inverse(Box<TermT>),
54    /// Square root of a term "sqrt(t)"
55    Sqrt(Box<TermT>),
56}
57
58impl TermT {
59    /// Returns the type of this term
60    pub fn get_type(&self) -> TermType {
61        match self {
62            TermT::Comp(terms) => terms.first().unwrap().get_type(),
63            TermT::Tensor(terms) => terms.iter().map(TermT::get_type).sum(),
64            TermT::Id(ty) => *ty,
65            TermT::Phase(_) => TermType(0),
66            TermT::IfLet { pattern, .. } => TermType(pattern.get_type().0),
67            TermT::Gate { def, .. } => def.get_type(),
68            TermT::Inverse(inner) => inner.get_type(),
69            TermT::Sqrt(inner) => inner.get_type(),
70        }
71    }
72}

74/// A pattern type "qn < qm"
75#[derive(Clone, Copy, Debug, PartialEq)]
76pub struct PatternType(pub usize, pub usize);
77
78impl Sum for PatternType {
79    fn sum<I: Iterator<Item = Self>>(iter: I) -> Self {
80        iter.fold(PatternType(0, 0), |PatternType(a, b), PatternType(c, d)| {
81            PatternType(a + c, b + d)
82        })
83    }
84}

86/// Syntax of typed patterns
87#[derive(Clone, Debug, PartialEq)]
88pub enum PatternT {
89    /// A non-empty composition "p_1 . ... . p_n"
90    Comp(Vec<PatternT>),
91    /// A tensor "p_1 x ... x p_n"
92    Tensor(Vec<PatternT>),
93    /// A sequence of ket states "|xyz>", equivalent to "|x> x |y> x |z>"
94    Ket(Vec<KetState>),
95    /// A unitary pattern
96    Unitary(Box<TermT>),
97}
98
99impl PatternT {
100    /// Returns the type of this pattern
101    pub fn get_type(&self) -> PatternType {
102        match self {
103            PatternT::Comp(patterns) => PatternType(
104                patterns.first().unwrap().get_type().0,
105                patterns.last().unwrap().get_type().1,
106            ),
107            PatternT::Tensor(patterns) => patterns.iter().map(PatternT::get_type).sum(),
108            PatternT::Ket(states) => PatternType(states.len(), 0),
109            PatternT::Unitary(inner) => inner.get_type().to_pattern_type(),
110        }
111    }
112
113    /// Evaluate a term to a `PatternN`, expanding top level definitions
114    /// and evaluating inverse and sqrt macros.
115    fn eval(&self) -> PatternN {
116        match self {
117            PatternT::Comp(patterns) => {
118                if patterns.len() == 1 {
119                    patterns[0].eval()
120                } else {
121                    PatternN::Comp(
122                        patterns.iter().map(PatternT::eval).collect(),
123                        self.get_type(),
124                    )
125                }
126            }
127            PatternT::Tensor(patterns) => {
128                if patterns.len() == 1 {
129                    patterns[0].eval()
130                } else {
131                    PatternN::Tensor(patterns.iter().map(PatternT::eval).collect())
132                }
133            }
134            PatternT::Ket(states) => {
135                PatternN::Tensor(states.iter().map(|&state| PatternN::Ket(state)).collect())
136            }
137            PatternT::Unitary(inner) => inner.eval(),
138        }
139    }
140}

143impl TermT {
144    /// Evaluate a term to a given `Buildable` type, expanding top level definitions
145    /// and evaluating inverse and sqrt macros.
146    /// In particular this can be used to generate a `TermN` from a `TermT`.
147    pub fn eval<B: Buildable>(&self) -> B {
148        self.eval_with_phase_mul(1.0)
149    }
150
151    fn eval_with_phase_mul<B: Buildable>(&self, phase_mul: f64) -> B {
152        match self {
153            TermT::Comp(terms) => {
154                let mut mapped_terms = terms.iter().map(|t| t.eval_with_phase_mul(phase_mul));
155                if terms.len() == 1 {
156                    mapped_terms.next().unwrap()
157                } else if phase_mul > 0.0 {
158                    B::comp(mapped_terms, &terms.first().unwrap().get_type())
159                } else {
160                    B::comp(mapped_terms.rev(), &terms.first().unwrap().get_type())
161                }
162            }
163            TermT::Tensor(terms) => {
164                if terms.len() == 1 {
165                    terms[0].eval_with_phase_mul(phase_mul)
166                } else {
167                    B::tensor(terms.iter().map(|t| t.eval_with_phase_mul(phase_mul)))
168                }
169            }
170            TermT::Id(ty) => B::comp(std::iter::empty(), ty),
171            TermT::Phase(phase) => B::atom(AtomN::Phase(phase_mul * phase.eval())),
172            TermT::IfLet { pattern, inner } => B::atom(AtomN::IfLet(
173                pattern.eval(),
174                Box::new(inner.eval_with_phase_mul(phase_mul)),
175                TermType(pattern.get_type().0),
176            )),
177            TermT::Gate { def, .. } => def.eval_with_phase_mul(phase_mul),
178            TermT::Inverse(inner) => inner.eval_with_phase_mul(-phase_mul),
179            TermT::Sqrt(inner) => inner.eval_with_phase_mul(phase_mul / 2.0),
180        }
181    }
182}
