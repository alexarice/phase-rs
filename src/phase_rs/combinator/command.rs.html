1//! A syntax of commands and their parsing.
2//!
3//! A `Command` is the top level structure accepted by the executable
4//! They allow a sequence of gates to be defined before taking a term to evaluate.
5
6use std::ops::Range;
7
8use winnow::{
9    LocatingSlice, ModalResult, Parser,
10    ascii::{multispace0, multispace1},
11    combinator::{cut_err, preceded, repeat, seq, terminated},
12    error::{StrContext, StrContextValue},
13};
14
15use super::{
16    typecheck::{Env, TypeCheckError},
17    typed_syntax::TermT,
18};
19use crate::{
20    combinator::raw_syntax::TermR,
21    text::{HasParser, Name, comment_parser},
22};
23
24/// The Command structure: a runnable program.
25#[derive(Clone, Debug)]
26pub struct Command<S> {
27    /// List of gates to define, with the name to bind them to.
28    pub gates: Vec<(Name, TermR<S>)>,
29    /// Final term to evaluate.
30    pub term: TermR<S>,
31}
32
33impl<S: Clone> Command<S> {
34    /// Typecheck a command, building an `Env` with gate definitions.
35    pub fn check(&self) -> Result<(Env, TermT), TypeCheckError<S>> {
36        let mut env = Env::default();
37        for (name, tm) in &self.gates {
38            let t = tm.check(&env, None)?;
39            env.0.insert(name.clone(), t);
40        }
41        let tm = self.term.check(&env, None)?;
42        Ok((env, tm))
43    }
44}
45
46impl HasParser for Command<Range<usize>> {
47    fn parser(input: &mut LocatingSlice<&str>) -> ModalResult<Self> {
48        let gate = preceded(
49	"gate",
50	cut_err(seq!(_: multispace1,
51		     Name::parser,
52		     _: (multispace0, "=", multispace0).context(StrContext::Expected(StrContextValue::CharLiteral('='))),
53		     TermR::parser,
54		     _: (multispace0, ","))).context(StrContext::Label("gate definition"))
55	);
56
57        comment_parser.parse_next(input)?;
58        let gates = repeat(0.., terminated(gate, comment_parser)).parse_next(input)?;
59        let term = TermR::parser
60            .context(StrContext::Label("Term"))
61            .parse_next(input)?;
62        comment_parser.parse_next(input)?;
63        Ok(Command { gates, term })
64    }
65}
