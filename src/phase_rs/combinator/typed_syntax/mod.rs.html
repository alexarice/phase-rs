23/// A unitary type "qn <-> qn"
24#[derive(Clone, Copy, Debug, PartialEq)]
25pub struct TermType(pub usize);
26
27impl Sum for TermType {
28    fn sum<I: Iterator<Item = Self>>(iter: I) -> Self {
29        TermType(iter.map(|x| x.0).sum())
30    }
31}

33impl TermType {
34    /// Convert a unitary type qn <-> qn to pattern type qn < qn
35    pub fn to_pattern_type(self) -> PatternType {
36        PatternType(self.0, self.0)
37    }
38}

40/// Syntax of typed terms
41#[derive(Clone, Debug, PartialEq)]
42pub enum TermT {
43    /// A non-empty composition "t_1 ; ... ; t_n"
44    Comp(Vec<TermT>),
45    /// A tensor "t_1 x ... x t_n"
46    Tensor(Vec<TermT>),
47    /// An identity "id(n)"
48    Id(TermType),
49    /// A (global) phase operator, e.g. "-1" or "ph(0.1pi)"
50    Phase(Phase),
51    /// An "if let" statement, "if let pattern then inner"
52    IfLet {
53        /// Pattern to match on in "if let"
54        pattern: PatternT,
55        /// Body of the "if let"
56        inner: Box<TermT>,
57    },
58    /// Top level symbol, a named gate
59    Gate {
60        /// Name of symbol/gate
61        name: Name,
62        /// Definition of symbol
63        def: Box<TermT>,
64    },
65    /// Inverse of a term "t ^ -1"
66    Inverse(Box<TermT>),
67    /// Square root of a term "sqrt(t)"
68    Sqrt(Box<TermT>),
69}
70
71impl TermT {
72    /// Returns the type of this term
73    pub fn get_type(&self) -> TermType {
74        match self {
75            TermT::Comp(terms) => terms.first().unwrap().get_type(),
76            TermT::Tensor(terms) => terms.iter().map(TermT::get_type).sum(),
77            TermT::Id(ty) => *ty,
78            TermT::Phase(_) => TermType(0),
79            TermT::IfLet { pattern, .. } => TermType(pattern.get_type().0),
80            TermT::Gate { def, .. } => def.get_type(),
81            TermT::Inverse(inner) => inner.get_type(),
82            TermT::Sqrt(inner) => inner.get_type(),
83        }
84    }
85}

87/// A pattern type "qn < qm"
88#[derive(Clone, Copy, Debug, PartialEq)]
89pub struct PatternType(pub usize, pub usize);
90
91impl Sum for PatternType {
92    fn sum<I: Iterator<Item = Self>>(iter: I) -> Self {
93        iter.fold(PatternType(0, 0), |PatternType(a, b), PatternType(c, d)| {
94            PatternType(a + c, b + d)
95        })
96    }
97}

99/// Syntax of typed patterns
100#[derive(Clone, Debug, PartialEq)]
101pub enum PatternT {
102    /// A non-empty composition "p_1 . ... . p_n"
103    Comp(Vec<PatternT>),
104    /// A tensor "p_1 x ... x p_n"
105    Tensor(Vec<PatternT>),
106    /// A sequence of ket states "|xyz>", equivalent to "|x> x |y> x |z>"
107    Ket(CompKetState),
108    /// A unitary pattern
109    Unitary(Box<TermT>),
110}
111
112impl PatternT {
113    /// Returns the type of this pattern
114    pub fn get_type(&self) -> PatternType {
115        match self {
116            PatternT::Comp(patterns) => PatternType(
117                patterns.first().unwrap().get_type().0,
118                patterns.last().unwrap().get_type().1,
119            ),
120            PatternT::Tensor(patterns) => patterns.iter().map(PatternT::get_type).sum(),
121            PatternT::Ket(states) => PatternType(states.qubits(), 0),
122            PatternT::Unitary(inner) => inner.get_type().to_pattern_type(),
123        }
124    }
125
126    /// Evaluate a term to a `PatternN`, expanding top level definitions
127    /// and evaluating inverse and sqrt macros.
128    fn eval(&self) -> PatternN {
129        match self {
130            PatternT::Comp(patterns) => {
131                if patterns.len() == 1 {
132                    patterns[0].eval()
133                } else {
134                    PatternN::Comp(
135                        patterns.iter().map(PatternT::eval).collect(),
136                        self.get_type(),
137                    )
138                }
139            }
140            PatternT::Tensor(patterns) => {
141                if patterns.len() == 1 {
142                    patterns[0].eval()
143                } else {
144                    PatternN::Tensor(patterns.iter().map(PatternT::eval).collect())
145                }
146            }
147            PatternT::Ket(states) => {
148                PatternN::Tensor(states.iter().map(|&state| PatternN::Ket(state)).collect())
149            }
150            PatternT::Unitary(inner) => inner.eval(),
151        }
152    }
153
154    fn eval_circ(&self, pattern: &mut PatternC, inj: &mut Vec<usize>, clauses: &mut Vec<ClauseC>) {
155        match self {
156            PatternT::Comp(patterns) => {
157                for p in patterns {
158                    p.eval_circ(pattern, inj, clauses);
159                }
160            }
161            PatternT::Tensor(patterns) => {
162                let mut stack: Vec<Vec<usize>> = Vec::new();
163                for p in patterns.iter().rev() {
164                    let size = p.get_type().0;
165                    let mut i = inj.split_off(inj.len() - size);
166                    p.eval_circ(pattern, &mut i, clauses);
167                    stack.push(i);
168                }
169                while let Some(i) = stack.pop() {
170                    inj.extend(i);
171                }
172            }
173            PatternT::Ket(states) => {
174                for (state, i) in states.iter().zip(inj.drain(0..states.qubits())) {
175                    pattern.parts[i] = Some(*state)
176                }
177            }
178            PatternT::Unitary(inner) => {
179                inner.eval_circ_clause(pattern, inj, -1.0, clauses);
180            }
181        }
182    }
183}

185impl TermT {
186    /// Evaluate a term to a given `Buildable` type, expanding top level definitions
187    /// and evaluating inverse and sqrt macros.
188    /// In particular this can be used to generate a `TermN` from a `TermT`.
189    pub fn eval<B: Buildable>(&self) -> B {
190        self.eval_with_phase_mul(1.0)
191    }
192
193    fn eval_with_phase_mul<B: Buildable>(&self, phase_mul: f64) -> B {
194        match self {
195            TermT::Comp(terms) => {
196                let mut mapped_terms = terms.iter().map(|t| t.eval_with_phase_mul(phase_mul));
197                if terms.len() == 1 {
198                    mapped_terms.next().unwrap()
199                } else if phase_mul > 0.0 {
200                    B::comp(mapped_terms, &terms.first().unwrap().get_type())
201                } else {
202                    B::comp(mapped_terms.rev(), &terms.first().unwrap().get_type())
203                }
204            }
205            TermT::Tensor(terms) => {
206                if terms.len() == 1 {
207                    terms[0].eval_with_phase_mul(phase_mul)
208                } else {
209                    B::tensor(terms.iter().map(|t| t.eval_with_phase_mul(phase_mul)))
210                }
211            }
212            TermT::Id(ty) => B::comp(std::iter::empty(), ty),
213            TermT::Phase(phase) => B::atom(AtomN::Phase(phase_mul * phase.eval())),
214            TermT::IfLet { pattern, inner } => B::atom(AtomN::IfLet(
215                pattern.eval(),
216                Box::new(inner.eval_with_phase_mul(phase_mul)),
217                TermType(pattern.get_type().0),
218            )),
219            TermT::Gate { def, .. } => def.eval_with_phase_mul(phase_mul),
220            TermT::Inverse(inner) => inner.eval_with_phase_mul(-phase_mul),
221            TermT::Sqrt(inner) => inner.eval_with_phase_mul(phase_mul / 2.0),
222        }
223    }
224
225    /// Returns a `TermC` representing the "circuit-normal-form" of the term.
226    pub fn eval_circ(&self) -> TermC {
227        let mut clauses = vec![];
228        let size = self.get_type().0;
229        let inj = (0..size).collect::<Vec<_>>();
230        self.eval_circ_clause(&PatternC::id(size), &inj, 1.0, &mut clauses);
231        TermC {
232            clauses,
233            ty: self.get_type(),
234        }
235    }
236    fn eval_circ_clause(
237        &self,
238        pattern: &PatternC,
239        inj: &[usize],
240        phase_mul: f64,
241        clauses: &mut Vec<ClauseC>,
242    ) {
243        match self {
244            TermT::Comp(terms) => {
245                if phase_mul < 0.0 {
246                    for t in terms.iter().rev() {
247                        t.eval_circ_clause(pattern, inj, phase_mul, clauses);
248                    }
249                } else {
250                    for t in terms {
251                        t.eval_circ_clause(pattern, inj, phase_mul, clauses);
252                    }
253                }
254            }
255            TermT::Tensor(terms) => {
256                let mut start = 0;
257                for t in terms {
258                    let size = t.get_type().0;
259                    let end = start + size;
260                    t.eval_circ_clause(pattern, &inj[start..end], phase_mul, clauses);
261                    start = end;
262                }
263            }
264            TermT::Id(_) => {
265                // Intentionally blank
266            }
267            TermT::Phase(phase) => {
268                clauses.push(ClauseC {
269                    pattern: pattern.clone(),
270                    phase: phase_mul * phase.eval(),
271                });
272            }
273            TermT::IfLet {
274                pattern: if_pattern,
275                inner,
276            } => {
277                let mut unitary_clauses = Vec::new();
278                let mut inner_pattern = pattern.clone();
279                let mut inner_inj = inj.to_vec();
280                if_pattern.eval_circ(&mut inner_pattern, &mut inner_inj, &mut unitary_clauses);
281                let temp: Vec<_> = unitary_clauses.iter().rev().map(ClauseC::invert).collect();
282                clauses.extend(unitary_clauses);
283
284                inner.eval_circ_clause(&inner_pattern, &inner_inj, phase_mul, clauses);
285
286                clauses.extend(temp)
287            }
288            TermT::Gate { def, .. } => {
289                def.eval_circ_clause(pattern, inj, phase_mul, clauses);
290            }
291            TermT::Inverse(inner) => {
292                inner.eval_circ_clause(pattern, inj, -phase_mul, clauses);
293            }
294            TermT::Sqrt(inner) => {
295                inner.eval_circ_clause(pattern, inj, phase_mul / 2.0, clauses);
296            }
297        }
298    }
299}

301impl TermT {
302    /// Convert to a raw term.
303    pub fn to_raw(&self) -> TermR<()> {
304        let terms = if let TermT::Comp(terms) = self {
305            terms.iter().map(|t| t.to_raw_tensor()).collect()
306        } else {
307            vec![self.to_raw_tensor()]
308        };
309        TermRInner { terms }.into()
310    }
311
312    fn to_raw_tensor(&self) -> TensorR<()> {
313        let terms = if let TermT::Tensor(terms) = self {
314            terms.iter().map(|t| t.to_raw_atom()).collect()
315        } else {
316            vec![self.to_raw_atom()]
317        };
318        TensorRInner { terms }.into()
319    }
320
321    fn to_raw_atom(&self) -> AtomR<()> {
322        match self {
323            TermT::Id(ty) => AtomRInner::Id(ty.0),
324            TermT::Phase(phase) => AtomRInner::Phase(*phase),
325            TermT::IfLet { pattern, inner } => AtomRInner::IfLet {
326                pattern: pattern.to_raw(),
327                inner: Box::new(inner.to_raw_atom()),
328            },
329            TermT::Gate { name, .. } => AtomRInner::Gate(name.to_owned()),
330            TermT::Inverse(inner) => AtomRInner::Inverse(Box::new(inner.to_raw_atom())),
331            TermT::Sqrt(inner) => AtomRInner::Sqrt(Box::new(inner.to_raw_atom())),
332            t => AtomRInner::Brackets(t.to_raw()),
333        }
334        .into()
335    }
336}

338impl PatternT {
339    /// Convert to a raw pattern.
340    pub fn to_raw(&self) -> PatternR<()> {
341        let patterns = if let PatternT::Comp(patterns) = self {
342            patterns.iter().map(|t| t.to_raw_tensor()).collect()
343        } else {
344            vec![self.to_raw_tensor()]
345        };
346        PatternRInner { patterns }.into()
347    }
348
349    fn to_raw_tensor(&self) -> PatTensorR<()> {
350        let patterns = if let PatternT::Tensor(patterns) = self {
351            patterns.iter().map(|t| t.to_raw_atom()).collect()
352        } else {
353            vec![self.to_raw_atom()]
354        };
355        PatTensorRInner { patterns }.into()
356    }
357
358    fn to_raw_atom(&self) -> PatAtomR<()> {
359        match self {
360            PatternT::Ket(states) => PatAtomRInner::Ket(states.clone()),
361            PatternT::Unitary(inner) => PatAtomRInner::Unitary(Box::new(inner.to_raw())),
362            p => PatAtomRInner::Brackets(p.to_raw()),
363        }
364        .into()
365    }
