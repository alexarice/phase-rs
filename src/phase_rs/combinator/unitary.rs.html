1//! Contains functions for converting normal-form syntax to matrices.
2
3use std::f64::consts::{FRAC_1_SQRT_2, PI};
4
5use faer::{Mat, mat};
6use num_complex::Complex;
7
8const CISQRT2: Complex<f64> = Complex::new(FRAC_1_SQRT_2, 0.0);
9
10use super::syntax::{
11    KetState,
12    normal::{AtomN, PatternN, TermN},
13};
14
15impl TermN {
16    /// Convert a normal-form term of type qn <-> qn to an n x n unitary matrix.
17    pub fn to_unitary(&self) -> Mat<Complex<f64>> {
18        match self {
19            TermN::Comp(terms, ty) => {
20                let mut terms_iter = terms.iter().map(TermN::to_unitary);
21                match terms_iter.next() {
22                    None => Mat::identity(1 << ty.0, 1 << ty.0),
23                    Some(u) => terms_iter.fold(u, |x, y| y * x),
24                }
25            }
26            TermN::Tensor(terms) => {
27                let mut terms_iter = terms.iter().map(TermN::to_unitary);
28                match terms_iter.next() {
29                    None => Mat::identity(1, 1),
30                    Some(u) => terms_iter.fold(u, |x, y| x.kron(y)),
31                }
32            }
33            TermN::Atom(atom) => atom.to_unitary(),
34        }
35    }
36}
37
38impl AtomN {
39    /// Convert a normal-form atom of type qn <-> qn to an n x n unitary matrix.
40    pub fn to_unitary(&self) -> Mat<Complex<f64>> {
41        match self {
42            AtomN::Phase(angle) => mat![[Complex::cis(angle * PI)]],
43            AtomN::IfLet(pattern, inner, _) => {
44                let (inj, proj) = pattern.to_inj_and_proj();
45                let u = inner.to_unitary();
46                proj + &inj * u * inj.adjoint()
47            }
48        }
49    }
50}
51
52impl KetState {
53    /// Returns the vector this `KetState` represents.
54    pub fn to_state(self) -> Mat<Complex<f64>> {
55        match self {
56            KetState::Zero => mat![[Complex::ONE], [Complex::ZERO]],
57            KetState::One => mat![[Complex::ZERO], [Complex::ONE]],
58            KetState::Plus => mat![[CISQRT2], [CISQRT2]],
59            KetState::Minus => mat![[CISQRT2], [-CISQRT2]],
60        }
61    }
62}
63
64impl PatternN {
65    /// Convert a normal-form pattern of type qm < qn to an m x n isometry matrix `i`
66    /// and an n x n projector `p` such that
67    /// p + ii^dagger = id
68    pub fn to_inj_and_proj(&self) -> (Mat<Complex<f64>>, Mat<Complex<f64>>) {
69        match self {
70            PatternN::Comp(patterns, ty) => {
71                let mut patterns_iter = patterns.iter().map(PatternN::to_inj_and_proj);
72                if let Some(i) = patterns_iter.next() {
73                    patterns_iter.fold(i, |(i1, p1), (i2, p2)| {
74                        (&i1 * i2, p1 + &i1 * p2 * i1.adjoint())
75                    })
76                } else {
77                    (
78                        Mat::identity(1 << ty.0, 1 << ty.0),
79                        Mat::zeros(1 << ty.0, 1 << ty.0),
80                    )
81                }
82            }
83            PatternN::Tensor(patterns) => {
84                let mut patterns_iter = patterns.iter().map(PatternN::to_inj_and_proj);
85                let i = patterns_iter.next().unwrap();
86                patterns_iter.fold(i, |(i1, p1), (i2, p2)| {
87                    (
88                        i1.kron(i2),
89                        p1.kron(Mat::<Complex<f64>>::identity(p2.nrows(), p2.nrows()))
90                            + (&i1 * i1.adjoint()).kron(p2),
91                    )
92                })
93            }
94            PatternN::Ket(state) => {
95                let m = state.to_state();
96                let cm = state.compl().to_state();
97                (m, cm.as_ref() * cm.adjoint())
98            }
99            PatternN::Unitary(inner) => {
100                let size = inner.get_type().0;
101                (inner.to_unitary(), Mat::zeros(1 << size, 1 << size))
102            }
103        }
104    }
