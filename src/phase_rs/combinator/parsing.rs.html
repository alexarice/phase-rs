5use winnow::{

10};

12use super::raw_syntax::{PatternR, TermR};
13use crate::{

15        pattern::{PatAtomR, PatAtomRInner, PatTensorR, PatTensorRInner, PatternRInner},

23/// Parser for terms.
24pub fn tm(input: &mut LocatingSlice<&str>) -> ModalResult<TermR<Range<usize>>> {
25    parse_spanned(
26        separated(1.., tensor, (multispace0, ';', multispace0))
27            .context(StrContext::Label("term"))
28            .map(|terms| TermRInner { terms }),
29    )
30    .parse_next(input)
31}
32
33fn tensor(input: &mut LocatingSlice<&str>) -> ModalResult<TensorR<Range<usize>>> {
34    parse_spanned(
35        separated(1.., atom, (multispace0, 'x', multispace0))
36            .context(StrContext::Label("term"))
37            .map(|terms| TensorRInner { terms }),
38    )
39    .parse_next(input)
40}

42fn atom(input: &mut LocatingSlice<&str>) -> ModalResult<AtomR<Range<usize>>> {
43    parse_spanned((
44	alt((
45	    delimited(("(", multispace0),
46		      cut_err(tm),
47		      cut_err((multispace0, ")").context(StrContext::Expected(StrContextValue::CharLiteral(')')))))
48		.map(AtomRInner::Brackets),
49	    preceded(("sqrt", multispace0), cut_err(atom))
50		.map(|inner| AtomRInner::Sqrt(Box::new(inner))),
51	    preceded("id", opt(dec_uint))
52		.map(|qubits| AtomRInner::Id(qubits.unwrap_or(1))),
53	    preceded("if", cut_err(seq!(_: multispace1, _: "let".context(StrContext::Expected(StrContextValue::StringLiteral("let"))), _: multispace1, pattern, _: multispace1, _: "then".context(StrContext::Expected(StrContextValue::StringLiteral("then"))), _: multispace1, atom)))
54	    .map(|(pattern, inner)| AtomRInner::IfLet{ pattern, inner: Box::new(inner) }),
55	    phase.map(AtomRInner::Phase),
56	    identifier.map(AtomRInner::Gate)
57	)).context(StrContext::Expected(StrContextValue::CharLiteral('(')))
58	    .context(StrContext::Expected(StrContextValue::StringLiteral("sqrt")))
59	    .context(StrContext::Expected(StrContextValue::StringLiteral("id")))
60	    .context(StrContext::Expected(StrContextValue::StringLiteral("if")))
61	    .context(StrContext::Expected(StrContextValue::CharLiteral('H')))
62	    .context(StrContext::Expected(StrContextValue::Description("identifier"))),
63	opt((
64	    multispace0,
65	    "^",
66	    multispace0,
67	    cut_err("-1").context(StrContext::Expected(StrContextValue::StringLiteral("-1"))),
68	)).context(StrContext::Label("term"))
69    )
70	.with_span()
71	.map(|((inner, invert), span)| {
72	    if invert.is_some() {
73		AtomRInner::Inverse ( Box::new(Spanned { inner, span }) )
74	    } else {
75		inner
76	    }}))
77	.parse_next(input)
78}

80/// Parse a pattern
81fn pattern(input: &mut LocatingSlice<&str>) -> ModalResult<PatternR<Range<usize>>> {
82    parse_spanned(
83        separated(1.., pattern_tensor, (multispace0, '.', multispace0))
84            .map(|patterns| PatternRInner { patterns }),
85    )
86    .parse_next(input)
87}
88
89fn pattern_tensor(input: &mut LocatingSlice<&str>) -> ModalResult<PatTensorR<Range<usize>>> {
90    parse_spanned(
91        separated(1.., pattern_atom, (multispace0, 'x', multispace0))
92            .map(|patterns| PatTensorRInner { patterns }),
93    )
94    .parse_next(input)
95}
96
97fn pattern_atom(input: &mut LocatingSlice<&str>) -> ModalResult<PatAtomR<Range<usize>>> {
98    parse_spanned(alt((
99        delimited(("(", multispace0), pattern, (multispace0, ")")).map(PatAtomRInner::Brackets),
100        ket.map(PatAtomRInner::Ket),
101        tm.map(|x| PatAtomRInner::Unitary(Box::new(x))),
102    )))
103    .parse_next(input)
