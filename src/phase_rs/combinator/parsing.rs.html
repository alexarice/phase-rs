24/// Parser for terms.
25pub fn tm(input: &mut LocatingSlice<&str>) -> ModalResult<TermR<Range<usize>>> {
26    parse_spanned(
27        separated(1.., tensor, (multispace0, ';', multispace0))
28            .context(StrContext::Label("term"))
29            .map(|terms| TermRInner { terms }),
30    )
31    .parse_next(input)
32}
33
34fn tensor(input: &mut LocatingSlice<&str>) -> ModalResult<TensorR<Range<usize>>> {
35    parse_spanned(
36        separated(1.., atom, (multispace0, 'x', multispace0))
37            .context(StrContext::Label("term"))
38            .map(|terms| TensorRInner { terms }),
39    )
40    .parse_next(input)
41}

43/// Parser for phases.
44pub fn phase(input: &mut LocatingSlice<&str>) -> ModalResult<Phase> {
45    alt((
46        "-1".value(Phase::MinusOne),
47        "i".value(Phase::Imag),
48        "-i".value(Phase::MinusImag),
49        delimited(
50            ("ph(", multispace0),
51            float,
52            (multispace0, "pi", multispace0, ")"),
53        )
54        .map(Phase::Angle),
55    ))
56    .parse_next(input)
57}

59fn atom(input: &mut LocatingSlice<&str>) -> ModalResult<AtomR<Range<usize>>> {
60    parse_spanned((
61	alt((
62	    delimited(("(", multispace0),
63		      cut_err(tm),
64		      cut_err((multispace0, ")").context(StrContext::Expected(StrContextValue::CharLiteral(')')))))
65		.map(AtomRInner::Brackets),
66	    preceded(("sqrt", multispace0), cut_err(atom))
67		.map(|inner| AtomRInner::Sqrt(Box::new(inner))),
68	    preceded("id", opt(dec_uint))
69		.map(|qubits| AtomRInner::Id(qubits.unwrap_or(1))),
70	    preceded("if", cut_err(seq!(_: multispace1, _: "let".context(StrContext::Expected(StrContextValue::StringLiteral("let"))), _: multispace1, pattern, _: multispace1, _: "then".context(StrContext::Expected(StrContextValue::StringLiteral("then"))), _: multispace1, atom)))
71	    .map(|(pattern, inner)| AtomRInner::IfLet{ pattern, inner: Box::new(inner) }),
72	    phase.map(AtomRInner::Phase),
73	    identifier.map(AtomRInner::Gate)
74	)).context(StrContext::Expected(StrContextValue::CharLiteral('(')))
75	    .context(StrContext::Expected(StrContextValue::StringLiteral("sqrt")))
76	    .context(StrContext::Expected(StrContextValue::StringLiteral("id")))
77	    .context(StrContext::Expected(StrContextValue::StringLiteral("if")))
78	    .context(StrContext::Expected(StrContextValue::CharLiteral('H')))
79	    .context(StrContext::Expected(StrContextValue::Description("identifier"))),
80	opt((
81	    multispace0,
82	    "^",
83	    multispace0,
84	    cut_err("-1").context(StrContext::Expected(StrContextValue::StringLiteral("-1"))),
85	)).context(StrContext::Label("term"))
86    )
87	.with_span()
88	.map(|((inner, invert), span)| {
89	    if invert.is_some() {
90		AtomRInner::Inverse ( Box::new(Spanned { inner, span }) )
91	    } else {
92		inner
93	    }}))
94	.parse_next(input)
95}

97/// Parse a pattern
98fn pattern(input: &mut LocatingSlice<&str>) -> ModalResult<PatternR<Range<usize>>> {
99    parse_spanned(
100        separated(1.., pattern_tensor, (multispace0, '.', multispace0))
101            .map(|patterns| PatternRInner { patterns }),
102    )
103    .parse_next(input)
104}
105
106fn pattern_tensor(input: &mut LocatingSlice<&str>) -> ModalResult<PatTensorR<Range<usize>>> {
107    parse_spanned(
108        separated(1.., pattern_atom, (multispace0, 'x', multispace0))
109            .map(|patterns| PatTensorRInner { patterns }),
110    )
111    .parse_next(input)
112}
113
114fn ket(input: &mut LocatingSlice<&str>) -> ModalResult<PatAtomRInner<Range<usize>>> {
115    delimited(
116        "|",
117        repeat(
118            1..,
119            alt((
120                "0".value(KetState::Zero),
121                "1".value(KetState::One),
122                "+".value(KetState::Plus),
123                "-".value(KetState::Minus),
124            )),
125        ),
126        ">",
127    )
128    .map(PatAtomRInner::Ket)
129    .parse_next(input)
130}
131
132fn pattern_atom(input: &mut LocatingSlice<&str>) -> ModalResult<PatAtomR<Range<usize>>> {
133    parse_spanned(alt((
134        delimited(("(", multispace0), pattern, (multispace0, ")")).map(PatAtomRInner::Brackets),
135        ket,
136        tm.map(|x| PatAtomRInner::Unitary(Box::new(x))),
137    )))
138    .parse_next(input)
139}
140
141fn identifier(input: &mut LocatingSlice<&str>) -> ModalResult<String> {
142    alphanumeric1
143        .map(|s: &str| s.to_owned())
144        .context(StrContext::Label("identifier"))
145        .context(StrContext::Expected(StrContextValue::Description(
146            "alphanumeric string",
147        )))
148        .parse_next(input)
149}
150
151fn gate(input: &mut LocatingSlice<&str>) -> ModalResult<(String, TermR<Range<usize>>)> {
152    preceded(
153	"gate",
154	cut_err(seq!(_: multispace1,
155		     identifier,
156		     _: (multispace0, "=", multispace0).context(StrContext::Expected(StrContextValue::CharLiteral('='))),
157		     tm,
158		     _: (multispace0, ","))).context(StrContext::Label("gate definition"))
159    ).parse_next(input)
160}

162/// Parse a comment
163pub fn comment(input: &mut LocatingSlice<&str>) -> ModalResult<()> {
164    (
165        multispace0,
166        repeat::<_, _, (), _, _>(0.., ("//", take_until(0.., "\n"), multispace0).value(())),
167    )
168        .parse_next(input)?;
169    Ok(())
170}
171
172/// Parse a command
173pub fn command(input: &mut LocatingSlice<&str>) -> ModalResult<Command<Range<usize>>> {
174    comment.parse_next(input)?;
175    let gates = repeat(0.., terminated(gate, comment)).parse_next(input)?;
176    let term = tm.context(StrContext::Label("Term")).parse_next(input)?;
177    comment.parse_next(input)?;
178    Ok(Command { gates, term })
