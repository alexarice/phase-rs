1//! Raw syntax definitions
2//!
3//! Raw syntax is used primarily for parsing and printed.
4//! It is not assumed to be typechecked/well-formed.
5
6use crate::{ket::CompKetState, phase::Phase, text::{Name, Spanned}};
7
8pub type TypeR<S> = Spanned<S, TypeRInner<S>>;
9
10pub enum TypeRInner<S> {
11    Bool,
12    QReg(usize),
13    Func {
14	unitary: bool,
15	params: Vec<ParamR<S>>,
16	ret: Box<TypeR<S>>,
17    },
18    Tuple(Vec<TypeR<S>>),
19}
20
21pub type ParamR<S> = Spanned<S, ParamRInner<S>>;
22pub struct ParamRInner<S> {
23    pub name: Name,
24    pub inout: bool,
25    pub ty: TypeR<S>,
26}
27
28
29pub type ExprR<S> = Spanned<S, ExprRInner<S>>;
30
31pub enum ExprRInner<S> {
32    MeasureCompBasis(Box<ExprR<S>>),
33    FunCall(Box<ExprR<S>>, Vec<ExprR<S>>, Vec<(Name, ExprR<S>)>),
34    Var(Name),
35    Ket(CompKetState),
36    Tensor(Box<ExprR<S>>, Box<ExprR<S>>),
37    Ap(Box<ExprR<S>>, Box<ExprR<S>>),
38}
39
40pub type ClauseR<S> = Spanned<S, ClauseRInner<S>>;
41
42pub enum ClauseRInner<S> {
43    Let(Vec<Name>, Option<TypeR<S>>, ExprR<S>),
44    Mutate(Vec<Name>, ExprR<S>),
45    Expr(ExprR<S>),
46    Phase(Phase),
47    IfLet {
48	pattern: ExprR<S>,
49	copattern: ExprR<S>,
50	body: TermR<S>,
51    },
52}
53
54pub type TermR<S> = Spanned<S, TermRInner<S>>;
55
56pub struct TermRInner<S>(Vec<ClauseR<S>>);
