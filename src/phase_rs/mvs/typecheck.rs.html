17/// Errors that can occur during typechecking.
18#[derive(Debug, Clone)]
19pub enum TypeCheckError<S> {
20    /// An unknown top-level symbol.
21    UnknownSymbol {
22        /// The unknown symbol encountered
23        name: Name,
24        /// Span of symbol
25        span: S,
26    },
27    /// Square root applied to a non-rootable term.
28    TermNotRootable {
29        /// The square root term.
30        unitary: UnitaryR<S>,
31    },
32    /// Wrong number of arguments given to unitary
33    WrongNumberOfArgs {
34        /// Unitary term
35        unitary: UnitaryR<S>,
36        ty: UnitaryTypeT,
37        expected_args: usize,
38    },
39    /// Argument to unitary's type does not match expected type
40    UnitaryArgTypeMismatch {
41        /// Unitary term
42        unitary: UnitaryR<S>,
43        /// Argument
44        argument: CopatternR<S>,
45        /// Argument type
46        arg_type: TypeT,
47        /// Expected type
48        expected_type: TypeT,
49    },
50    /// Named argument has unknown name
51    UnitaryUnknownNamedArg {
52        /// Unitary term
53        unitary: UnitaryR<S>,
54        /// Argument
55        argument: CopatternR<S>,
56        /// Name of argument
57        name: Name,
58    },
59    /// Argument is given both by position and by name
60    UnitaryArgNamedAndPosition {
61        /// Unitary term
62        unitary: UnitaryR<S>,
63        /// Position argument
64        pos_arg: CopatternR<S>,
65        /// Name
66        name: Name,
67        /// Named argument
68        named_arg: CopatternR<S>,
69    },
70    /// Clash of support in Copattern
71    CopatternSupportClash {
72        /// Subterm 1
73        copattern_1: CopatternR<S>,
74        /// Subterm 2
75        copattern_2: CopatternR<S>,
76        /// Name of resued variable
77        name: Name,
78    },
79    /// Argument is given by two different named arguments.
80    UnitaryArgNamedTwice {
81        unitary: UnitaryR<S>,
82        /// Name
83        name: Name,
84        /// Argument 1,
85        arg_1: CopatternR<S>,
86        /// Argument 2,
87        arg_2: CopatternR<S>,
88    },
89}

100#[derive(Debug, Clone, PartialEq, Eq)]
101pub struct SupportAtom {
102    pub(crate) var: usize,
103    pub(crate) range: Option<Range<usize>>,
104}
105
106impl PartialOrd for SupportAtom {
107    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
108        Some(self.cmp(other))
109    }
110}
111
112impl Ord for SupportAtom {
113    fn cmp(&self, other: &Self) -> Ordering {
114        let fst = self.var.cmp(&other.var);
115        if fst.is_ne() {
116            return fst;
117        }
118        match (&self.range, &other.range) {
119            (None, None) => Ordering::Equal,
120            (None, Some(_)) => Ordering::Less,
121            (Some(_), None) => Ordering::Greater,
122            (Some(lhs), Some(rhs)) => lhs.start.cmp(&rhs.start).then(lhs.end.cmp(&rhs.end)),
123        }
124    }
125}
126
127#[derive(Debug, Clone)]
128pub struct Support<T>(BTreeMap<SupportAtom, T>);
129
130impl<T> Default for Support<T> {
131    fn default() -> Self {
132        Support(BTreeMap::default())
133    }
134}
135
136impl<T> IntoIterator for Support<T> {
137    type Item = (SupportAtom, T);
138
139    type IntoIter = <BTreeMap<SupportAtom, T> as IntoIterator>::IntoIter;
140
141    fn into_iter(self) -> Self::IntoIter {
142        self.0.into_iter()
143    }
144}
145
146fn cmp_range(lhs: &Option<Range<usize>>, rhs: &Option<Range<usize>>) -> Ordering {
147    match (lhs, rhs) {
148        (None, None) => Ordering::Equal,
149        (None, Some(_)) => Ordering::Less,
150        (Some(_), None) => Ordering::Greater,
151        (Some(r1), Some(r2)) => r1.start.cmp(&r2.start).then(r1.end.cmp(&r2.end)),
152    }
153}
154
155impl<T> Support<T> {
156    pub fn insert(&mut self, key: SupportAtom, value: T) {
157        self.0.insert(key, value);
158    }
159
160    pub fn new(key: SupportAtom, value: T) -> Self {
161        Support(BTreeMap::from([(key, value)]))
162    }
163
164    pub fn get_clash(&self, key: &SupportAtom) -> Option<&T> {
165        let lower_bound = SupportAtom {
166            var: key.var,
167            range: None,
168        };
169        let upper_bound = SupportAtom {
170            var: key.var,
171            range: None,
172        };
173        let before = self.0.range(&lower_bound..key).next_back();
174        let after = self.0.range(key..&upper_bound).next();
175        before
176            .and_then(|(x, t)| {
177                if x.var != key.var || cmp_range(&x.range, &key.range).is_lt() {
178                    None
179                } else {
180                    Some(t)
181                }
182            })
183            .or(after.and_then(|(x, t)| {
184                if x.var != key.var || cmp_range(&key.range, &x.range).is_lt() {
185                    None
186                } else {
187                    Some(t)
188                }
189            }))
190    }
