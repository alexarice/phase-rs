Enum AtomRInner

pub enum AtomRInner<S> {
    Brackets(TermR<S>),
    Id(usize),
    Phase(Phase),
    IfLet {
        pattern: PatternR<S>,
        inner: Box<AtomR<S>>,
    },
    Gate(String),
    Inverse(Box<AtomR<S>>),
    Sqrt(Box<AtomR<S>>),
}
Expand description

Raw syntax atom without text span. Represents a term other than a tensor or composition (or a composition/tensor in brackets)
